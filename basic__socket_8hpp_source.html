1871 template <typename ConnectHandler>
1872 void operator()(ConnectHandler&& handler,
1873 const endpoint_type& peer_endpoint,
1874 const asio::error_code& open_ec) const
1875 {
1876 // If you get an error on the following line it means that your handler
1877 // does not meet the documented type requirements for a ConnectHandler.
1878 ASIO_CONNECT_HANDLER_CHECK(ConnectHandler, handler) type_check;
1879
1880 if (open_ec)
1881 {
1882 asio::post(self_->impl_.get_executor(),
1883 asio::detail::bind_handler(
1884 static_cast<ConnectHandler&&>(handler), open_ec));
1885 }
1886 else
1887 {
1888 detail::non_const_lvalue<ConnectHandler> handler2(handler);
1889 self_->impl_.get_service().async_connect(
1890 self_->impl_.get_implementation(), peer_endpoint,
1891 handler2.value, self_->impl_.get_executor());
1892 }
1893 }
