234 // Send a datagram to the specified endpoint. Returns the number of bytes
235 // sent.
236 template <typename ConstBufferSequence>
237 size_t send_to(implementation_type& impl, const ConstBufferSequence& buffers,
238 const endpoint_type& destination, socket_base::message_flags flags,
239 asio::error_code& ec)
240 {
241 typedef buffer_sequence_adapter<asio::const_buffer,
242 ConstBufferSequence> bufs_type;
243
244 size_t n;
245 if (bufs_type::is_single_buffer)
246 {
247 n = socket_ops::sync_sendto1(impl.socket_, impl.state_,
248 bufs_type::first(buffers).data(),
249 bufs_type::first(buffers).size(), flags,
250 destination.data(), destination.size(), ec);
251 }
252 else
253 {
254 bufs_type bufs(buffers);
255 n = socket_ops::sync_sendto(impl.socket_, impl.state_,
256 bufs.buffers(), bufs.count(), flags,
257 destination.data(), destination.size(), ec);
258 }
259
260 ASIO_ERROR_LOCATION(ec);
261 return n;
262 }

278 template <typename ConstBufferSequence, typename Handler, typename IoExecutor>
279 void async_send_to(implementation_type& impl,
280 const ConstBufferSequence& buffers,
281 const endpoint_type& destination, socket_base::message_flags flags,
282 Handler& handler, const IoExecutor& io_ex)
283 {
284 bool is_continuation =
285 asio_handler_cont_helpers::is_continuation(handler);
286
287 associated_cancellation_slot_t<Handler> slot
288 = asio::get_associated_cancellation_slot(handler);
289
290 // Allocate and construct an operation to wrap the handler.
291 typedef reactive_socket_sendto_op<ConstBufferSequence,
292 endpoint_type, Handler, IoExecutor> op;
293 typename op::ptr p = { asio::detail::addressof(handler),
294 op::ptr::allocate(handler), 0 };
295 p.p = new (p.v) op(success_ec_, impl.socket_,
296 buffers, destination, flags, handler, io_ex);
297
298 // Optionally register for per-operation cancellation.
299 if (slot.is_connected())
300 {
301 p.p->cancellation_key_ =
302 &slot.template emplace<reactor_op_cancellation>(
303 &reactor_, &impl.reactor_data_, impl.socket_, reactor::write_op);
304 }
305
306 ASIO_HANDLER_CREATION((reactor_.context(), *p.p, "socket",
307 &impl, impl.socket_, "async_send_to"));
308
309 start_op(impl, reactor::write_op, p.p,
310 is_continuation, true, false, &io_ex, 0);
311 p.v = p.p = 0;
312 }

348 // Receive a datagram with the endpoint of the sender. Returns the number of
349 // bytes received.
350 template <typename MutableBufferSequence>
351 size_t receive_from(implementation_type& impl,
352 const MutableBufferSequence& buffers,
353 endpoint_type& sender_endpoint, socket_base::message_flags flags,
354 asio::error_code& ec)
355 {
356 typedef buffer_sequence_adapter<asio::mutable_buffer,
357 MutableBufferSequence> bufs_type;
358
359 std::size_t addr_len = sender_endpoint.capacity();
360 std::size_t n;
361 if (bufs_type::is_single_buffer)
362 {
363 n = socket_ops::sync_recvfrom1(impl.socket_, impl.state_,
364 bufs_type::first(buffers).data(), bufs_type::first(buffers).size(),
365 flags, sender_endpoint.data(), &addr_len, ec);
366 }
367 else
368 {
369 bufs_type bufs(buffers);
370 n = socket_ops::sync_recvfrom(impl.socket_, impl.state_, bufs.buffers(),
371 bufs.count(), flags, sender_endpoint.data(), &addr_len, ec);
372 }
373
374 if (!ec)
375 sender_endpoint.resize(addr_len);
376
377 ASIO_ERROR_LOCATION(ec);
378 return n;
379 }

478 // Accept a new connection.
479 template <typename Socket>
480 asio::error_code accept(implementation_type& impl,
481 Socket& peer, endpoint_type* peer_endpoint, asio::error_code& ec)
482 {
483 // We cannot accept a socket that is already open.
484 if (peer.is_open())
485 {
486 ec = asio::error::already_open;
487 ASIO_ERROR_LOCATION(ec);
488 return ec;
489 }
490
491 std::size_t addr_len = peer_endpoint ? peer_endpoint->capacity() : 0;
492 socket_holder new_socket(socket_ops::sync_accept(impl.socket_,
493 impl.state_, peer_endpoint ? peer_endpoint->data() : 0,
494 peer_endpoint ? &addr_len : 0, ec));
495
496 // On success, assign new connection to peer socket object.
497 if (new_socket.get() != invalid_socket)
498 {
499 if (peer_endpoint)
500 peer_endpoint->resize(addr_len);
501 peer.assign(impl.protocol_, new_socket.get(), ec);
502 if (!ec)
503 new_socket.release();
504 }
505
506 ASIO_ERROR_LOCATION(ec);
507 return ec;
508 }
