38 template <typename SyncWriteStream, typename ConstBufferSequence,
39 typename ConstBufferIterator, typename CompletionCondition>
40 std::size_t write(SyncWriteStream& s,
41 const ConstBufferSequence& buffers, const ConstBufferIterator&,
42 CompletionCondition completion_condition, asio::error_code& ec)
43 {
44 ec = asio::error_code();
45 asio::detail::consuming_buffers<const_buffer,
46 ConstBufferSequence, ConstBufferIterator> tmp(buffers);
47 while (!tmp.empty())
48 {
49 if (std::size_t max_size = detail::adapt_completion_condition_result(
50 completion_condition(ec, tmp.total_consumed())))
51 tmp.consume(s.write_some(tmp.prepare(max_size), ec));
52 else
53 break;
54 }
55 return tmp.total_consumed();
56 }
57} // namespace detail

287 class write_op
288 : public base_from_cancellation_state<WriteHandler>,
289 base_from_completion_cond<CompletionCondition>
290 {
291 public:
292 write_op(AsyncWriteStream& stream, const ConstBufferSequence& buffers,
293 CompletionCondition& completion_condition, WriteHandler& handler)
294 : base_from_cancellation_state<WriteHandler>(
295 handler, enable_partial_cancellation()),
296 base_from_completion_cond<CompletionCondition>(completion_condition),
297 stream_(stream),
298 buffers_(buffers),
299 start_(0),
300 handler_(static_cast<WriteHandler&&>(handler))
301 {

304 write_op(const write_op& other)
305 : base_from_cancellation_state<WriteHandler>(other),
306 base_from_completion_cond<CompletionCondition>(other),
307 stream_(other.stream_),
308 buffers_(other.buffers_),
309 start_(other.start_),
310 handler_(other.handler_)
311 {
312 }
313
314 write_op(write_op&& other)
315 : base_from_cancellation_state<WriteHandler>(
316 static_cast<base_from_cancellation_state<WriteHandler>&&>(other)),
317 base_from_completion_cond<CompletionCondition>(
318 static_cast<base_from_completion_cond<CompletionCondition>&&>(other)),
319 stream_(other.stream_),
320 buffers_(static_cast<buffers_type&&>(other.buffers_)),
321 start_(other.start_),
322 handler_(static_cast<WriteHandler&&>(other.handler_))
323 {
324 }
325
326 void operator()(asio::error_code ec,
327 std::size_t bytes_transferred, int start = 0)
328 {
329 std::size_t max_size;
330 switch (start_ = start)
331 {
332 case 1:
333 max_size = this->check_for_completion(ec, buffers_.total_consumed());
334 for (;;)
335 {
336 {
337 ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_write"));
338 stream_.async_write_some(buffers_.prepare(max_size),
339 static_cast<write_op&&>(*this));
340 }
341 return; default:
342 buffers_.consume(bytes_transferred);
343 if ((!ec && bytes_transferred == 0) || buffers_.empty())
344 break;
345 max_size = this->check_for_completion(ec, buffers_.total_consumed());
346 if (max_size == 0)
347 break;
348 if (this->cancelled() != cancellation_type::none)
349 {
350 ec = error::operation_aborted;
351 break;
352 }
353 }
354
355 static_cast<WriteHandler&&>(handler_)(
356 static_cast<const asio::error_code&>(ec),
357 static_cast<const std::size_t&>(buffers_.total_consumed()));
358 }
359 }

368 WriteHandler handler_;
