407template <typename Handler, typename IoExecutor, typename = void>
408class handler_work :
409 handler_work_base<IoExecutor>,
410 handler_work_base<associated_executor_t<Handler, IoExecutor>, IoExecutor>
411{
412public:
413 typedef handler_work_base<IoExecutor> base1_type;
414 typedef handler_work_base<associated_executor_t<Handler, IoExecutor>,
415 IoExecutor> base2_type;
416
417 handler_work(Handler& handler, const IoExecutor& io_ex) noexcept
418 : base1_type(0, 0, io_ex),
419 base2_type(base1_type::owns_work(),
420 asio::get_associated_executor(handler, io_ex), io_ex)
421 {
422 }
423
424 template <typename Function>
425 void complete(Function& function, Handler& handler)
426 {
427 if (!base1_type::owns_work() && !base2_type::owns_work())
428 {
429 // When using a native implementation, I/O completion handlers are
430 // already dispatched according to the execution context's executor's
431 // rules. We can call the function directly.
432 static_cast<Function&&>(function)();
433 }
434 else
435 {
436 base2_type::dispatch(function, handler);
437 }
438 }
439};
