372protected:
373 int_type underflow()
374 {
375#if defined(ASIO_WINDOWS_RUNTIME)
376 ec_ = asio::error::operation_not_supported;
377 return traits_type::eof();
378#else // defined(ASIO_WINDOWS_RUNTIME)
379 if (gptr() != egptr())
380 return traits_type::eof();
381
382 for (;;)
383 {
384 // Check if we are past the expiry time.
385 if (traits_helper::less_than(expiry_time_, traits_helper::now()))
386 {
387 ec_ = asio::error::timed_out;
388 return traits_type::eof();
389 }
390
391 // Try to complete the operation without blocking.
392 if (!socket().native_non_blocking())
393 socket().native_non_blocking(true, ec_);
394 detail::buffer_sequence_adapter<mutable_buffer, mutable_buffer>
395 bufs(asio::buffer(get_buffer_) + putback_max);
396 detail::signed_size_type bytes = detail::socket_ops::recv(
397 socket().native_handle(), bufs.buffers(), bufs.count(), 0, ec_);
398
399 // Check if operation succeeded.
400 if (bytes > 0)
401 {
402 setg(&get_buffer_[0], &get_buffer_[0] + putback_max,
403 &get_buffer_[0] + putback_max + bytes);
404 return traits_type::to_int_type(*gptr());
405 }
406
407 // Check for EOF.
408 if (bytes == 0)
409 {
410 ec_ = asio::error::eof;
411 return traits_type::eof();
412 }
413
414 // Operation failed.
415 if (ec_ != asio::error::would_block
416 && ec_ != asio::error::try_again)
417 return traits_type::eof();
418
419 // Wait for socket to become ready.
420 if (detail::socket_ops::poll_read(
421 socket().native_handle(), 0, timeout(), ec_) < 0)
422 return traits_type::eof();
423 }
424#endif // defined(ASIO_WINDOWS_RUNTIME)
425 }
426
427 int_type overflow(int_type c)
428 {
429#if defined(ASIO_WINDOWS_RUNTIME)
430 ec_ = asio::error::operation_not_supported;
431 return traits_type::eof();
432#else // defined(ASIO_WINDOWS_RUNTIME)
433 char_type ch = traits_type::to_char_type(c);
434
435 // Determine what needs to be sent.
436 const_buffer output_buffer;
437 if (put_buffer_.empty())
438 {
439 if (traits_type::eq_int_type(c, traits_type::eof()))
440 return traits_type::not_eof(c); // Nothing to do.
441 output_buffer = asio::buffer(&ch, sizeof(char_type));
442 }
443 else
444 {
445 output_buffer = asio::buffer(pbase(),
446 (pptr() - pbase()) * sizeof(char_type));
447 }
448
449 while (output_buffer.size() > 0)
450 {
451 // Check if we are past the expiry time.
452 if (traits_helper::less_than(expiry_time_, traits_helper::now()))
453 {
454 ec_ = asio::error::timed_out;
455 return traits_type::eof();
456 }
457
458 // Try to complete the operation without blocking.
459 if (!socket().native_non_blocking())
460 socket().native_non_blocking(true, ec_);
461 detail::buffer_sequence_adapter<
462 const_buffer, const_buffer> bufs(output_buffer);
463 detail::signed_size_type bytes = detail::socket_ops::send(
464 socket().native_handle(), bufs.buffers(), bufs.count(), 0, ec_);
465
466 // Check if operation succeeded.
467 if (bytes > 0)
468 {
469 output_buffer += static_cast<std::size_t>(bytes);
470 continue;
471 }
472
473 // Operation failed.
474 if (ec_ != asio::error::would_block
475 && ec_ != asio::error::try_again)
476 return traits_type::eof();
477
478 // Wait for socket to become ready.
479 if (detail::socket_ops::poll_write(
480 socket().native_handle(), 0, timeout(), ec_) < 0)
481 return traits_type::eof();
482 }
483
484 if (!put_buffer_.empty())
485 {
486 setp(&put_buffer_[0], &put_buffer_[0] + put_buffer_.size());
487
488 // If the new character is eof then our work here is done.
489 if (traits_type::eq_int_type(c, traits_type::eof()))
490 return traits_type::not_eof(c);
491
492 // Add the new character to the output buffer.
493 *pptr() = ch;
494 pbump(1);
495 }
496
497 return c;
498#endif // defined(ASIO_WINDOWS_RUNTIME)
499 }

537 int timeout() const
538 {
539 int64_t msec = traits_helper::to_posix_duration(
540 traits_helper::subtract(expiry_time_,
541 traits_helper::now())).total_milliseconds();
542 if (msec > (std::numeric_limits<int>::max)())
543 msec = (std::numeric_limits<int>::max)();
544 else if (msec < 0)
545 msec = 0;
546 return static_cast<int>(msec);
547 }

554
555 template <typename EndpointIterator>
556 void connect_to_endpoints(EndpointIterator begin, EndpointIterator end)
557 {
558#if defined(ASIO_WINDOWS_RUNTIME)
559 ec_ = asio::error::operation_not_supported;
560#else // defined(ASIO_WINDOWS_RUNTIME)
561 if (ec_)
562 return;
563
564 ec_ = asio::error::not_found;
565 for (EndpointIterator i = begin; i != end; ++i)
566 {
567 // Check if we are past the expiry time.
568 if (traits_helper::less_than(expiry_time_, traits_helper::now()))
569 {
570 ec_ = asio::error::timed_out;
571 return;
572 }
573
574 // Close and reopen the socket.
575 typename Protocol::endpoint ep(*i);
576 socket().close(ec_);
577 socket().open(ep.protocol(), ec_);
578 if (ec_)
579 continue;
580
581 // Try to complete the operation without blocking.
582 if (!socket().native_non_blocking())
583 socket().native_non_blocking(true, ec_);
584 detail::socket_ops::connect(socket().native_handle(),
585 ep.data(), ep.size(), ec_);
586
587 // Check if operation succeeded.
588 if (!ec_)
589 return;
590
591 // Operation failed.
592 if (ec_ != asio::error::in_progress
593 && ec_ != asio::error::would_block)
594 continue;
595
596 // Wait for socket to become ready.
597 if (detail::socket_ops::poll_connect(
598 socket().native_handle(), timeout(), ec_) < 0)
599 continue;
600
601 // Get the error code from the connect operation.
602 int connect_error = 0;
603 size_t connect_error_len = sizeof(connect_error);
604 if (detail::socket_ops::getsockopt(socket().native_handle(), 0,
605 SOL_SOCKET, SO_ERROR, &connect_error, &connect_error_len, ec_)
606 == detail::socket_error_retval)
607 return;
608
609 // Check the result of the connect operation.
610 ec_ = asio::error_code(connect_error,
611 asio::error::get_system_category());
612 if (!ec_)
613 return;
614 }
615#endif // defined(ASIO_WINDOWS_RUNTIME)
616 }
