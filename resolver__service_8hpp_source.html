73 // Resolve a query to a list of entries.
74 results_type resolve(implementation_type&, const query_type& qry,
75 asio::error_code& ec)
76 {
77 asio::detail::addrinfo_type* address_info = 0;
78
79 socket_ops::getaddrinfo(qry.host_name().c_str(),
80 qry.service_name().c_str(), qry.hints(), &address_info, ec);
81 auto_addrinfo auto_address_info(address_info);
82
83 ASIO_ERROR_LOCATION(ec);
84 return ec ? results_type() : results_type::create(
85 address_info, qry.host_name(), qry.service_name());
86 }
87
88 // Asynchronously resolve a query to a list of entries.
89 template <typename Handler, typename IoExecutor>
90 void async_resolve(implementation_type& impl, const query_type& qry,
91 Handler& handler, const IoExecutor& io_ex)
92 {
93 // Allocate and construct an operation to wrap the handler.
94 typedef resolve_query_op<Protocol, Handler, IoExecutor> op;
95 typename op::ptr p = { asio::detail::addressof(handler),
96 op::ptr::allocate(handler), 0 };
97 p.p = new (p.v) op(impl, qry, scheduler_, handler, io_ex);
98
99 ASIO_HANDLER_CREATION((scheduler_.context(),
100 *p.p, "resolver", &impl, 0, "async_resolve"));
101
102 start_resolve_op(p.p);
103 p.v = p.p = 0;
104 }
105
106 // Resolve an endpoint to a list of entries.
107 results_type resolve(implementation_type&,
108 const endpoint_type& endpoint, asio::error_code& ec)
109 {
110 char host_name[NI_MAXHOST];
111 char service_name[NI_MAXSERV];
112 socket_ops::sync_getnameinfo(endpoint.data(), endpoint.size(),
113 host_name, NI_MAXHOST, service_name, NI_MAXSERV,
114 endpoint.protocol().type(), ec);
115
116 ASIO_ERROR_LOCATION(ec);
117 return ec ? results_type() : results_type::create(
118 endpoint, host_name, service_name);
119 }
120
121 // Asynchronously resolve an endpoint to a list of entries.
122 template <typename Handler, typename IoExecutor>
123 void async_resolve(implementation_type& impl, const endpoint_type& endpoint,
124 Handler& handler, const IoExecutor& io_ex)

133 *p.p, "resolver", &impl, 0, "async_resolve"));
134
135 start_resolve_op(p.p);
136 p.v = p.p = 0;
137 }
138};
