33template <typename T, typename U, typename Executor>
34struct coro_spawn_op
35{
36 coro<T, U, Executor> c;
37
38 void operator()(auto& self)
39 {
40 auto op = c.async_resume(deferred);
41 std::move(op)((prepend)(std::move(self), 0));
42 }
43
44 void operator()(auto& self, int, auto... res)
45 {
46 self.complete(std::move(res)...);
47 }
48};
