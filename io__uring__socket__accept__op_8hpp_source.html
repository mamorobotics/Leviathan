36template <typename Socket, typename Protocol>
37class io_uring_socket_accept_op_base : public io_uring_operation
38{
39public:
40 io_uring_socket_accept_op_base(const asio::error_code& success_ec,
41 socket_type socket, socket_ops::state_type state, Socket& peer,
42 const Protocol& protocol, typename Protocol::endpoint* peer_endpoint,
43 func_type complete_func)
44 : io_uring_operation(success_ec,
45 &io_uring_socket_accept_op_base::do_prepare,
46 &io_uring_socket_accept_op_base::do_perform, complete_func),
47 socket_(socket),
48 state_(state),
49 peer_(peer),
50 protocol_(protocol),
51 peer_endpoint_(peer_endpoint),
52 addrlen_(peer_endpoint ? peer_endpoint->capacity() : 0)
53 {
54 }
55
56 static void do_prepare(io_uring_operation* base, ::io_uring_sqe* sqe)
57 {
58 ASIO_ASSUME(base != 0);
59 io_uring_socket_accept_op_base* o(
60 static_cast<io_uring_socket_accept_op_base*>(base));
61
62 if ((o->state_ & socket_ops::internal_non_blocking) != 0)
63 {
64 ::io_uring_prep_poll_add(sqe, o->socket_, POLLIN);
65 }
66 else
67 {
68 ::io_uring_prep_accept(sqe, o->socket_,
69 o->peer_endpoint_ ? o->peer_endpoint_->data() : 0,
70 o->peer_endpoint_ ? &o->addrlen_ : 0, 0);
71 }
72 }
73
74 static bool do_perform(io_uring_operation* base, bool after_completion)
75 {
76 ASIO_ASSUME(base != 0);
77 io_uring_socket_accept_op_base* o(
78 static_cast<io_uring_socket_accept_op_base*>(base));
79
80 if ((o->state_ & socket_ops::internal_non_blocking) != 0)
81 {
82 socket_type new_socket = invalid_socket;
83 std::size_t addrlen = static_cast<std::size_t>(o->addrlen_);
84 bool result = socket_ops::non_blocking_accept(o->socket_,
85 o->state_, o->peer_endpoint_ ? o->peer_endpoint_->data() : 0,
86 o->peer_endpoint_ ? &addrlen : 0, o->ec_, new_socket);
87 o->new_socket_.reset(new_socket);
88 o->addrlen_ = static_cast<socklen_t>(addrlen);
89 return result;
90 }
91
92 if (o->ec_ && o->ec_ == asio::error::would_block)
93 {
94 o->state_ |= socket_ops::internal_non_blocking;
95 return false;
96 }
97
98 if (after_completion && !o->ec_)
99 o->new_socket_.reset(static_cast<int>(o->bytes_transferred_));
100
101 return after_completion;
102 }
103
104 void do_assign()
105 {
106 if (new_socket_.get() != invalid_socket)
107 {
108 if (peer_endpoint_)
109 peer_endpoint_->resize(addrlen_);
110 peer_.assign(protocol_, new_socket_.get(), ec_);
111 if (!ec_)
112 new_socket_.release();
113 }
114 }
115
116private:
117 socket_type socket_;
118 socket_ops::state_type state_;
119 socket_holder new_socket_;
120 Socket& peer_;
121 Protocol protocol_;
122 typename Protocol::endpoint* peer_endpoint_;
123 socklen_t addrlen_;
124};

126template <typename Socket, typename Protocol,
127 typename Handler, typename IoExecutor>
128class io_uring_socket_accept_op :
129 public io_uring_socket_accept_op_base<Socket, Protocol>
130{
131public:
132 ASIO_DEFINE_HANDLER_PTR(io_uring_socket_accept_op);
133
134 io_uring_socket_accept_op(const asio::error_code& success_ec,
135 socket_type socket, socket_ops::state_type state, Socket& peer,
136 const Protocol& protocol, typename Protocol::endpoint* peer_endpoint,
137 Handler& handler, const IoExecutor& io_ex)
138 : io_uring_socket_accept_op_base<Socket, Protocol>(
139 success_ec, socket, state, peer, protocol, peer_endpoint,
140 &io_uring_socket_accept_op::do_complete),
141 handler_(static_cast<Handler&&>(handler)),
142 work_(handler_, io_ex)
143 {
144 }
145
146 static void do_complete(void* owner, operation* base,
147 const asio::error_code& /*ec*/,
148 std::size_t /*bytes_transferred*/)
149 {
150 // Take ownership of the handler object.
151 ASIO_ASSUME(base != 0);
152 io_uring_socket_accept_op* o(static_cast<io_uring_socket_accept_op*>(base));
153 ptr p = { asio::detail::addressof(o->handler_), o, o };
154
155 // On success, assign new connection to peer socket object.
156 if (owner)
157 o->do_assign();
158
159 ASIO_HANDLER_COMPLETION((*o));
160
161 // Take ownership of the operation's outstanding work.
162 handler_work<Handler, IoExecutor> w(
163 static_cast<handler_work<Handler, IoExecutor>&&>(
164 o->work_));
165
166 ASIO_ERROR_LOCATION(o->ec_);
167
168 // Make a copy of the handler so that the memory can be deallocated before
169 // the upcall is made. Even if we're not about to make an upcall, a
170 // sub-object of the handler may be the true owner of the memory associated
171 // with the handler. Consequently, a local copy of the handler is required
172 // to ensure that any owning sub-object remains valid until after we have
173 // deallocated the memory here.
174 detail::binder1<Handler, asio::error_code>
175 handler(o->handler_, o->ec_);
176 p.h = asio::detail::addressof(handler.handler_);
177 p.reset();
178
179 // Make the upcall if required.
180 if (owner)
181 {
182 fenced_block b(fenced_block::half);
183 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
184 w.complete(handler, handler.handler_);
185 ASIO_HANDLER_INVOCATION_END;
186 }
187 }
188
189private:
190 Handler handler_;
191 handler_work<Handler, IoExecutor> work_;
192};
