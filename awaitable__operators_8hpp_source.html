73template <typename Executor>
74awaitable<void, Executor> operator&&(
75 awaitable<void, Executor> t, awaitable<void, Executor> u)
76{
77 auto ex = co_await this_coro::executor;
78
79 auto [order, ex0, ex1] =
80 co_await make_parallel_group(
81 co_spawn(ex, std::move(t), deferred),
82 co_spawn(ex, std::move(u), deferred)
83 ).async_wait(
84 wait_for_one_error(),
85 deferred
86 );
87
88 if (ex0 && ex1)
89 throw multiple_exceptions(ex0);
90 if (ex0)
91 std::rethrow_exception(ex0);
92 if (ex1)
93 std::rethrow_exception(ex1);
94 co_return;
95}

424template <typename... T>
425struct widen_variant
426{
427 template <std::size_t I, typename SourceVariant>
428 static std::variant<T...> call(SourceVariant& source)
429 {
430 if (source.index() == I)
431 return std::variant<T...>{
432 std::in_place_index<I>, std::move(std::get<I>(source))};
433 else if constexpr (I + 1 < std::variant_size_v<SourceVariant>)
434 return call<I + 1>(source);
435 else
436 throw std::logic_error("empty variant");
437 }
438};
