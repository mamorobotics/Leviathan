53template <typename Allocator>
54struct partial_promise_base
55{
56 template <typename Executor, typename Token, typename... Args>
57 void* operator new(const std::size_t size, Executor&, Token& tk, Args&...)
58 {
59 return allocate_coroutine<Allocator>(size, get_associated_allocator(tk));
60 }
61
62 void operator delete(void* raw, const std::size_t size)
63 {
64 deallocate_coroutine<Allocator>(raw, size);
65 }
66};
67
68template <>
69struct partial_promise_base<std::allocator<void>>
70{
71};

73template <typename Allocator>
74struct partial_promise : partial_promise_base<Allocator>
75{
76 auto initial_suspend() noexcept
77 {
78 return asio::detail::suspend_always{};
79 }
80
81 auto final_suspend() noexcept
82 {
83 struct awaitable_t
84 {
85 partial_promise *p;
86
87 constexpr bool await_ready() noexcept { return true; }
88
89 auto await_suspend(asio::detail::coroutine_handle<>) noexcept
90 {
91 p->get_return_object().handle.destroy();
92 }
93
94 constexpr void await_resume() noexcept {}
95 };
96
97 return awaitable_t{this};
98 }
99
100 void return_void() {}
101
102 partial_coro get_return_object()
103 {
104 return partial_coro{coroutine_handle<partial_promise>::from_promise(*this)};
105 }
106
107 void unhandled_exception()

109 assert(false);
110 }
111};
