28template <typename Stream>
29std::size_t buffered_read_stream<Stream>::fill()
30{
31 detail::buffer_resize_guard<detail::buffered_stream_storage>
32 resize_guard(storage_);
33 std::size_t previous_size = storage_.size();
34 storage_.resize(storage_.capacity());
35 storage_.resize(previous_size + next_layer_.read_some(buffer(
36 storage_.data() + previous_size,
37 storage_.size() - previous_size)));
38 resize_guard.commit();
39 return storage_.size() - previous_size;
40}

190template <typename Stream>
191template <typename MutableBufferSequence>
192std::size_t buffered_read_stream<Stream>::read_some(
193 const MutableBufferSequence& buffers)
194{
195 using asio::buffer_size;
196 if (buffer_size(buffers) == 0)
197 return 0;
198
199 if (storage_.empty())
200 this->fill();
201
202 return this->copy(buffers);
203}

207std::size_t buffered_read_stream<Stream>::read_some(
208 const MutableBufferSequence& buffers, asio::error_code& ec)
209{
210 ec = asio::error_code();
211
212 using asio::buffer_size;
213 if (buffer_size(buffers) == 0)
214 return 0;
215
216 if (storage_.empty() && !this->fill(ec))
217 return 0;
218
219 return this->copy(buffers);

251 {
252 if (ec || storage_.empty())
253 {

303 {
304 // If you get an error on the following line it means that your handler
305 // does not meet the documented type requirements for a ReadHandler.
306 ASIO_READ_HANDLER_CHECK(ReadHandler, handler) type_check;
307
308 using asio::buffer_size;
309 non_const_lvalue<ReadHandler> handler2(handler);
310 if (buffer_size(buffers) == 0 || !storage->empty())
311 {
312 next_layer_.async_read_some(ASIO_MUTABLE_BUFFER(0, 0),
313 buffered_read_some_handler<MutableBufferSequence,
314 decay_t<ReadHandler>>(
315 *storage, buffers, handler2.value));

317 else
318 {
319 initiate_async_buffered_fill<Stream>(this->next_layer_)(
320 buffered_read_some_handler<MutableBufferSequence,
321 decay_t<ReadHandler>>(
