50template <typename T, typename Executor = any_io_executor>
51class ASIO_NODISCARD awaitable
52{
53public:
55 typedef T value_type;
56
58 typedef Executor executor_type;
59
61 constexpr awaitable() noexcept
62 : frame_(nullptr)
63 {
64 }
65
67 awaitable(awaitable&& other) noexcept
68 : frame_(std::exchange(other.frame_, nullptr))
69 {
70 }
71
73 ~awaitable()
74 {
75 if (frame_)
76 frame_->destroy();
77 }
78
80 awaitable& operator=(awaitable&& other) noexcept
81 {
82 if (this != &other)
83 frame_ = std::exchange(other.frame_, nullptr);
84 return *this;
85 }
86
88 bool valid() const noexcept
89 {
90 return !!frame_;
91 }
