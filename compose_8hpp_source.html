34template <typename Impl, typename Work, typename Handler,
35 typename R, typename... Args>
36class composed_op<Impl, Work, Handler, R(Args...)>
37 : public base_from_cancellation_state<Handler>
38{
39public:
40 template <typename I, typename W, typename H>
41 composed_op(I&& impl,
42 W&& work,
43 H&& handler)
44 : base_from_cancellation_state<Handler>(
45 handler, enable_terminal_cancellation()),
46 impl_(static_cast<I&&>(impl)),
47 work_(static_cast<W&&>(work)),
48 handler_(static_cast<H&&>(handler)),
49 invocations_(0)
50 {
51 }
52
53 composed_op(composed_op&& other)
54 : base_from_cancellation_state<Handler>(
55 static_cast<base_from_cancellation_state<Handler>&&>(other)),
56 impl_(static_cast<Impl&&>(other.impl_)),
57 work_(static_cast<Work&&>(other.work_)),
58 handler_(static_cast<Handler&&>(other.handler_)),
59 invocations_(other.invocations_)
60 {
61 }
62
63 typedef typename composed_work_guard<
64 typename Work::head_type>::executor_type io_executor_type;
65
66 io_executor_type get_io_executor() const noexcept
67 {
68 return work_.head_.get_executor();
69 }
70
71 typedef associated_executor_t<Handler, io_executor_type> executor_type;
72
73 executor_type get_executor() const noexcept
74 {
75 return (get_associated_executor)(handler_, work_.head_.get_executor());
76 }
77
78 typedef associated_allocator_t<Handler, std::allocator<void>> allocator_type;
79
80 allocator_type get_allocator() const noexcept
81 {
82 return (get_associated_allocator)(handler_, std::allocator<void>());
83 }
84
85 template<typename... T>
86 void operator()(T&&... t)
87 {
88 if (invocations_ < ~0u)
89 ++invocations_;
90 this->get_cancellation_state().slot().clear();
91 impl_(*this, static_cast<T&&>(t)...);
92 }
93
94 void complete(Args... args)
95 {
96 this->work_.reset();
97 static_cast<Handler&&>(this->handler_)(static_cast<Args&&>(args)...);
98 }
99
100 void reset_cancellation_state()
101 {
102 base_from_cancellation_state<Handler>::reset_cancellation_state(handler_);
103 }
104
105 template <typename Filter>
106 void reset_cancellation_state(Filter&& filter)
107 {
108 base_from_cancellation_state<Handler>::reset_cancellation_state(handler_,
109 static_cast<Filter&&>(filter));
110 }
111
112 template <typename InFilter, typename OutFilter>
113 void reset_cancellation_state(InFilter&& in_filter,
114 OutFilter&& out_filter)
115 {
116 base_from_cancellation_state<Handler>::reset_cancellation_state(handler_,
117 static_cast<InFilter&&>(in_filter),
118 static_cast<OutFilter&&>(out_filter));
119 }
120
121 cancellation_type_t cancelled() const noexcept
122 {
123 return base_from_cancellation_state<Handler>::cancelled();
124 }
125
126//private:
127 Impl impl_;
128 Work work_;
129 Handler handler_;
130 unsigned invocations_;
131};

133template <typename Impl, typename Work, typename Handler, typename Signature>
134inline bool asio_handler_is_continuation(
135 composed_op<Impl, Work, Handler, Signature>* this_handler)
136{
137 return this_handler->invocations_ > 1 ? true
138 : asio_handler_cont_helpers::is_continuation(
139 this_handler->handler_);

142template <typename Signature, typename Executors>
143class initiate_composed_op
144{
145public:
146 typedef typename composed_io_executors<Executors>::head_type executor_type;
147
148 template <typename T>
149 explicit initiate_composed_op(int, T&& executors)
150 : executors_(static_cast<T&&>(executors))
151 {
152 }
153
154 executor_type get_executor() const noexcept
155 {
156 return executors_.head_;
157 }
158
159 template <typename Handler, typename Impl>
160 void operator()(Handler&& handler,
161 Impl&& impl) const
162 {
163 composed_op<decay_t<Impl>, composed_work<Executors>,
164 decay_t<Handler>, Signature>(
165 static_cast<Impl&&>(impl),
166 composed_work<Executors>(executors_),
167 static_cast<Handler&&>(handler))();
168 }
169
170private:
171 composed_io_executors<Executors> executors_;
172};
173
174template <typename Signature, typename Executors>
175inline initiate_composed_op<Signature, Executors> make_initiate_composed_op(
176 composed_io_executors<Executors>&& executors)
177{
178 return initiate_composed_op<Signature, Executors>(0,
179 static_cast<composed_io_executors<Executors>&&>(executors));
180}
