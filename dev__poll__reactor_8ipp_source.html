55void dev_poll_reactor::shutdown()
56{
57 asio::detail::mutex::scoped_lock lock(mutex_);
58 shutdown_ = true;
59 lock.unlock();
60
61 op_queue<operation> ops;
62
63 for (int i = 0; i < max_ops; ++i)
64 op_queue_[i].get_all_operations(ops);
65
66 timer_queues_.get_all_timers(ops);
67
68 scheduler_.abandon_operations(ops);

71void dev_poll_reactor::notify_fork(
72 asio::execution_context::fork_event fork_ev)
73{
74 if (fork_ev == asio::execution_context::fork_child)
75 {
76 detail::mutex::scoped_lock lock(mutex_);
77
78 if (dev_poll_fd_ != -1)
79 ::close(dev_poll_fd_);
80 dev_poll_fd_ = -1;
81 dev_poll_fd_ = do_dev_poll_create();
82
83 interrupter_.recreate();
84
85 // Add the interrupter's descriptor to /dev/poll.
86 ::pollfd ev = { 0, 0, 0 };
87 ev.fd = interrupter_.read_descriptor();
88 ev.events = POLLIN | POLLERR;
89 ev.revents = 0;
90 ::write(dev_poll_fd_, &ev, sizeof(ev));
91
92 // Re-register all descriptors with /dev/poll. The changes will be written
93 // to the /dev/poll descriptor the next time the reactor is run.
94 for (int i = 0; i < max_ops; ++i)
95 {
96 reactor_op_queue<socket_type>::iterator iter = op_queue_[i].begin();
97 reactor_op_queue<socket_type>::iterator end = op_queue_[i].end();
98 for (; iter != end; ++iter)
99 {
100 ::pollfd& pending_ev = add_pending_event_change(iter->first);
101 pending_ev.events |= POLLERR | POLLHUP;
102 switch (i)
103 {
104 case read_op: pending_ev.events |= POLLIN; break;
105 case write_op: pending_ev.events |= POLLOUT; break;
106 case except_op: pending_ev.events |= POLLPRI; break;
107 default: break;
108 }
109 }
110 }
111 interrupter_.interrupt();
112 }
113}

158void dev_poll_reactor::start_op(int op_type, socket_type descriptor,
159 dev_poll_reactor::per_descriptor_data&, reactor_op* op,
160 bool is_continuation, bool allow_speculative,
161 void (*on_immediate)(operation*, bool, const void*),
162 const void* immediate_arg)
163{
164 asio::detail::mutex::scoped_lock lock(mutex_);
165
166 if (shutdown_)
167 {
168 on_immediate(op, is_continuation, immediate_arg);
169 return;
170 }
171
172 if (allow_speculative)
173 {
174 if (op_type != read_op || !op_queue_[except_op].has_operation(descriptor))
175 {
176 if (!op_queue_[op_type].has_operation(descriptor))
177 {
178 if (op->perform())
179 {
180 lock.unlock();
181 on_immediate(op, is_continuation, immediate_arg);
182 return;
183 }
184 }
185 }
186 }
187
188 bool first = op_queue_[op_type].enqueue_operation(descriptor, op);
189 scheduler_.work_started();
190 if (first)
191 {
192 ::pollfd& ev = add_pending_event_change(descriptor);
193 ev.events = POLLERR | POLLHUP;
194 if (op_type == read_op
195 || op_queue_[read_op].has_operation(descriptor))
196 ev.events |= POLLIN;
197 if (op_type == write_op
198 || op_queue_[write_op].has_operation(descriptor))
199 ev.events |= POLLOUT;
200 if (op_type == except_op
201 || op_queue_[except_op].has_operation(descriptor))
202 ev.events |= POLLPRI;
203 interrupter_.interrupt();
204 }
205}

227void dev_poll_reactor::deregister_descriptor(socket_type descriptor,
228 dev_poll_reactor::per_descriptor_data&, bool)
229{
230 asio::detail::mutex::scoped_lock lock(mutex_);
231
232 // Remove the descriptor from /dev/poll.
233 ::pollfd& ev = add_pending_event_change(descriptor);
234 ev.events = POLLREMOVE;
235 interrupter_.interrupt();
236
237 // Cancel any outstanding operations associated with the descriptor.
238 cancel_ops_unlocked(descriptor, asio::error::operation_aborted);
239}

241void dev_poll_reactor::deregister_internal_descriptor(
242 socket_type descriptor, dev_poll_reactor::per_descriptor_data&)
243{
244 asio::detail::mutex::scoped_lock lock(mutex_);
245
246 // Remove the descriptor from /dev/poll. Since this function is only called
247 // during a fork, we can apply the change immediately.
248 ::pollfd ev = { 0, 0, 0 };
249 ev.fd = descriptor;
250 ev.events = POLLREMOVE;
251 ev.revents = 0;
252 ::write(dev_poll_fd_, &ev, sizeof(ev));
253
254 // Destroy all operations associated with the descriptor.
255 op_queue<operation> ops;
256 asio::error_code ec;
257 for (int i = 0; i < max_ops; ++i)
258 op_queue_[i].cancel_operations(descriptor, ops, ec);
259}

266void dev_poll_reactor::run(long usec, op_queue<operation>& ops)
267{
268 asio::detail::mutex::scoped_lock lock(mutex_);
269
270 // We can return immediately if there's no work to do and the reactor is
271 // not supposed to block.
272 if (usec == 0 && op_queue_[read_op].empty() && op_queue_[write_op].empty()
273 && op_queue_[except_op].empty() && timer_queues_.all_empty())
274 return;
275
276 // Write the pending event registration changes to the /dev/poll descriptor.
277 std::size_t events_size = sizeof(::pollfd) * pending_event_changes_.size();
278 if (events_size > 0)
279 {
280 errno = 0;
281 int result = ::write(dev_poll_fd_,
282 &pending_event_changes_[0], events_size);
283 if (result != static_cast<int>(events_size))
284 {
285 asio::error_code ec = asio::error_code(
286 errno, asio::error::get_system_category());
287 for (std::size_t i = 0; i < pending_event_changes_.size(); ++i)
288 {
289 int descriptor = pending_event_changes_[i].fd;
290 for (int j = 0; j < max_ops; ++j)
291 op_queue_[j].cancel_operations(descriptor, ops, ec);
292 }
293 }
294 pending_event_changes_.clear();
295 pending_event_change_index_.clear();
296 }
297
298 // Calculate timeout.
299 int timeout;
300 if (usec == 0)
301 timeout = 0;
302 else
303 {
304 timeout = (usec < 0) ? -1 : ((usec - 1) / 1000 + 1);
305 timeout = get_timeout(timeout);
306 }
307 lock.unlock();
308
309 // Block on the /dev/poll descriptor.
310 ::pollfd events[128] = { { 0, 0, 0 } };
311 ::dvpoll dp = { 0, 0, 0 };
312 dp.dp_fds = events;
313 dp.dp_nfds = 128;
314 dp.dp_timeout = timeout;
315 int num_events = ::ioctl(dev_poll_fd_, DP_POLL, &dp);
316
317 lock.lock();
318
319 // Dispatch the waiting events.
320 for (int i = 0; i < num_events; ++i)
321 {
322 int descriptor = events[i].fd;
323 if (descriptor == interrupter_.read_descriptor())
324 {
325 interrupter_.reset();
326 }
327 else
328 {
329 bool more_reads = false;
330 bool more_writes = false;
331 bool more_except = false;
332
333 // Exception operations must be processed first to ensure that any
334 // out-of-band data is read before normal data.
335 if (events[i].events & (POLLPRI | POLLERR | POLLHUP))
336 more_except =
337 op_queue_[except_op].perform_operations(descriptor, ops);
338 else
339 more_except = op_queue_[except_op].has_operation(descriptor);
340
341 if (events[i].events & (POLLIN | POLLERR | POLLHUP))
342 more_reads = op_queue_[read_op].perform_operations(descriptor, ops);
343 else
344 more_reads = op_queue_[read_op].has_operation(descriptor);
345
346 if (events[i].events & (POLLOUT | POLLERR | POLLHUP))
347 more_writes = op_queue_[write_op].perform_operations(descriptor, ops);
348 else
349 more_writes = op_queue_[write_op].has_operation(descriptor);
350
351 if ((events[i].events & (POLLERR | POLLHUP)) != 0
352 && !more_except && !more_reads && !more_writes)
353 {
354 // If we have an event and no operations associated with the
355 // descriptor then we need to delete the descriptor from /dev/poll.
356 // The poll operation can produce POLLHUP or POLLERR events when there
357 // is no operation pending, so if we do not remove the descriptor we
358 // can end up in a tight polling loop.
359 ::pollfd ev = { 0, 0, 0 };
360 ev.fd = descriptor;
361 ev.events = POLLREMOVE;
362 ev.revents = 0;
363 ::write(dev_poll_fd_, &ev, sizeof(ev));
364 }
365 else
366 {
367 ::pollfd ev = { 0, 0, 0 };
368 ev.fd = descriptor;
369 ev.events = POLLERR | POLLHUP;
370 if (more_reads)
371 ev.events |= POLLIN;
372 if (more_writes)
373 ev.events |= POLLOUT;
374 if (more_except)
375 ev.events |= POLLPRI;
376 ev.revents = 0;
377 int result = ::write(dev_poll_fd_, &ev, sizeof(ev));
378 if (result != sizeof(ev))
379 {
380 asio::error_code ec(errno,
381 asio::error::get_system_category());
382 for (int j = 0; j < max_ops; ++j)
383 op_queue_[j].cancel_operations(descriptor, ops, ec);
384 }
385 }
386 }
387 }
388 timer_queues_.get_ready_timers(ops);
389}

396int dev_poll_reactor::do_dev_poll_create()
397{
398 int fd = ::open("/dev/poll", O_RDWR);

400 {
401 asio::error_code ec(errno,
402 asio::error::get_system_category());
403 asio::detail::throw_error(ec, "/dev/poll");

420int dev_poll_reactor::get_timeout(int msec)
421{
422 // By default we will wait no longer than 5 minutes. This will ensure that
423 // any changes to the system clock are detected after no longer than this.
424 const int max_msec = 5 * 60 * 1000;
425 return timer_queues_.wait_duration_msec(
426 (msec < 0 || max_msec < msec) ? max_msec : msec);
427}

429void dev_poll_reactor::cancel_ops_unlocked(socket_type descriptor,
430 const asio::error_code& ec)
431{
432 bool need_interrupt = false;
433 op_queue<operation> ops;
434 for (int i = 0; i < max_ops; ++i)
435 need_interrupt = op_queue_[i].cancel_operations(
436 descriptor, ops, ec) || need_interrupt;
437 scheduler_.post_deferred_completions(ops);
438 if (need_interrupt)
439 interrupter_.interrupt();
440}

442::pollfd& dev_poll_reactor::add_pending_event_change(int descriptor)
443{
444 hash_map<int, std::size_t>::iterator iter
445 = pending_event_change_index_.find(descriptor);
446 if (iter == pending_event_change_index_.end())
447 {
448 std::size_t index = pending_event_changes_.size();
449 pending_event_changes_.reserve(pending_event_changes_.size() + 1);
450 pending_event_change_index_.insert(std::make_pair(descriptor, index));
451 pending_event_changes_.push_back(::pollfd());
452 pending_event_changes_[index].fd = descriptor;
453 pending_event_changes_[index].revents = 0;
454 return pending_event_changes_[index];
455 }
456 else
457 {
458 return pending_event_changes_[iter->second];
459 }
