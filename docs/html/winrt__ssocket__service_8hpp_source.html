110 // Open a new socket implementation.
111 asio::error_code open(implementation_type& impl,
112 const protocol_type& protocol, asio::error_code& ec)
113 {
114 if (is_open(impl))
115 {
116 ec = asio::error::already_open;
117 return ec;
118 }
119
120 try
121 {
122 impl.socket_ = ref new Windows::Networking::Sockets::StreamSocket;
123 impl.protocol_ = protocol;
124 ec = asio::error_code();
125 }
126 catch (Platform::Exception^ e)
127 {
128 ec = asio::error_code(e->HResult,
129 asio::system_category());
130 }
131
132 return ec;
133 }

135 // Assign a native socket to a socket implementation.

199 // Get a socket option.
200 template <typename Option>
201 asio::error_code get_option(const implementation_type& impl,
202 Option& option, asio::error_code& ec) const
203 {
204 std::size_t size = option.size(impl.protocol_);
205 do_get_option(impl, option.level(impl.protocol_),
206 option.name(impl.protocol_),
207 option.data(impl.protocol_), &size, ec);
208 if (!ec)
209 option.resize(impl.protocol_, size);
210 return ec;
211 }

220 // Start an asynchronous connect.
221 template <typename Handler, typename IoExecutor>
222 void async_connect(implementation_type& impl,
223 const endpoint_type& peer_endpoint,
224 Handler& handler, const IoExecutor& io_ex)
225 {
226 bool is_continuation =
227 asio_handler_cont_helpers::is_continuation(handler);
228
229 // Allocate and construct an operation to wrap the handler.
230 typedef winrt_socket_connect_op<Handler, IoExecutor> op;
231 typename op::ptr p = { asio::detail::addressof(handler),
232 op::ptr::allocate(handler), 0 };
233 p.p = new (p.v) op(handler, io_ex);
234
235 ASIO_HANDLER_CREATION((scheduler_.context(),
236 *p.p, "socket", &impl, 0, "async_connect"));
237
238 start_connect_op(impl, peer_endpoint.data(), p.p, is_continuation);
239 p.v = p.p = 0;
240 }
241};
