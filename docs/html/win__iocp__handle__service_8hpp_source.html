148 // Start an asynchronous write. The data being written must be valid for the
149 // lifetime of the asynchronous operation.
150 template <typename ConstBufferSequence, typename Handler, typename IoExecutor>
151 void async_write_some(implementation_type& impl,
152 const ConstBufferSequence& buffers,
153 Handler& handler, const IoExecutor& io_ex)
154 {
155 associated_cancellation_slot_t<Handler> slot
156 = asio::get_associated_cancellation_slot(handler);
157
158 // Allocate and construct an operation to wrap the handler.
159 typedef win_iocp_handle_write_op<
160 ConstBufferSequence, Handler, IoExecutor> op;
161 typename op::ptr p = { asio::detail::addressof(handler),
162 op::ptr::allocate(handler), 0 };
163 operation* o = p.p = new (p.v) op(buffers, handler, io_ex);
164
165 ASIO_HANDLER_CREATION((iocp_service_.context(), *p.p, "handle", &impl,
166 reinterpret_cast<uintmax_t>(impl.handle_), "async_write_some"));
167
168 // Optionally register for per-operation cancellation.
169 if (slot.is_connected())
170 o = &slot.template emplace<iocp_op_cancellation>(impl.handle_, o);
171
172 start_write_op(impl, 0,
173 buffer_sequence_adapter<asio::const_buffer,
174 ConstBufferSequence>::first(buffers), o);
175 p.v = p.p = 0;
176 }

367 // Helper class used to implement per operation cancellation.
368 class iocp_op_cancellation : public operation
369 {
370 public:
371 iocp_op_cancellation(HANDLE h, operation* target)
372 : operation(&iocp_op_cancellation::do_complete),
373 handle_(h),
374 target_(target)
375 {
376 }
377
378 static void do_complete(void* owner, operation* base,
379 const asio::error_code& result_ec,
380 std::size_t bytes_transferred)
381 {
382 iocp_op_cancellation* o = static_cast<iocp_op_cancellation*>(base);
383 o->target_->complete(owner, result_ec, bytes_transferred);
384 }
385
386 void operator()(cancellation_type_t type)
387 {
388#if defined(_WIN32_WINNT) && (_WIN32_WINNT >= 0x0600)
389 if (!!(type &
390 (cancellation_type::terminal
391 | cancellation_type::partial
392 | cancellation_type::total)))
393 {
394 ::CancelIoEx(handle_, this);
395 }
396#else // defined(_WIN32_WINNT) && (_WIN32_WINNT >= 0x0600)
397 (void)type;
398#endif // defined(_WIN32_WINNT) && (_WIN32_WINNT >= 0x0600)
399 }
400
401 private:
402 HANDLE handle_;
403 operation* target_;
404 };
