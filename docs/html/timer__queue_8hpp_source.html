76 // Add a new timer to the queue. Returns true if this is the timer that is
77 // earliest in the queue, in which case the reactor's event demultiplexing
78 // function call may need to be interrupted and restarted.
79 bool enqueue_timer(const time_type& time, per_timer_data& timer, wait_op* op)
80 {
81 // Enqueue the timer object.
82 if (timer.prev_ == 0 && &timer != timers_)
83 {
84 if (this->is_positive_infinity(time))
85 {
86 // No heap entry is required for timers that never expire.
87 timer.heap_index_ = (std::numeric_limits<std::size_t>::max)();
88 }
89 else
90 {
91 // Put the new timer at the correct position in the heap. This is done
92 // first since push_back() can throw due to allocation failure.
93 timer.heap_index_ = heap_.size();
94 heap_entry entry = { time, &timer };
95 heap_.push_back(entry);
96 up_heap(heap_.size() - 1);
97 }
98
99 // Insert the new timer into the linked list of active timers.
100 timer.next_ = timers_;
101 timer.prev_ = 0;
102 if (timers_)
103 timers_->prev_ = &timer;
104 timers_ = &timer;
105 }
106
107 // Enqueue the individual timer operation.
108 timer.op_queue_.push(op);
109
110 // Interrupt reactor only if newly added timer is first to expire.
111 return timer.heap_index_ == 0 && timer.op_queue_.front() == op;
112 }

144 // Dequeue all timers not later than the current time.
145 virtual void get_ready_timers(op_queue<operation>& ops)
146 {
147 if (!heap_.empty())
148 {
149 const time_type now = Time_Traits::now();
150 while (!heap_.empty() && !Time_Traits::less_than(now, heap_[0].time_))
151 {
152 per_timer_data* timer = heap_[0].timer_;
153 while (wait_op* op = timer->op_queue_.front())
154 {
155 timer->op_queue_.pop();
156 op->ec_ = asio::error_code();
157 ops.push(op);
158 }
159 remove_timer(*timer);
160 }
161 }
162 }
163
164 // Dequeue all timers.
165 virtual void get_all_timers(op_queue<operation>& ops)
166 {
167 while (timers_)
168 {
169 per_timer_data* timer = timers_;
170 timers_ = timers_->next_;
171 ops.push(timer->op_queue_);
172 timer->next_ = 0;
173 timer->prev_ = 0;
174 }
175
176 heap_.clear();
177 }

179 // Cancel and dequeue operations for the given timer.
180 std::size_t cancel_timer(per_timer_data& timer, op_queue<operation>& ops,
181 std::size_t max_cancelled = (std::numeric_limits<std::size_t>::max)())
182 {
183 std::size_t num_cancelled = 0;
184 if (timer.prev_ != 0 || &timer == timers_)
185 {
186 while (wait_op* op = (num_cancelled != max_cancelled)
187 ? timer.op_queue_.front() : 0)
188 {
189 op->ec_ = asio::error::operation_aborted;
190 timer.op_queue_.pop();
191 ops.push(op);
192 ++num_cancelled;
193 }
194 if (timer.op_queue_.empty())
195 remove_timer(timer);
196 }
197 return num_cancelled;
198 }

200 // Cancel and dequeue a specific operation for the given timer.
201 void cancel_timer_by_key(per_timer_data* timer,
202 op_queue<operation>& ops, void* cancellation_key)
203 {
204 if (timer->prev_ != 0 || timer == timers_)
205 {
206 op_queue<wait_op> other_ops;
207 while (wait_op* op = timer->op_queue_.front())
208 {
209 timer->op_queue_.pop();
210 if (op->cancellation_key_ == cancellation_key)
211 {
212 op->ec_ = asio::error::operation_aborted;
213 ops.push(op);
214 }
215 else
216 other_ops.push(op);
217 }
218 timer->op_queue_.push(other_ops);
219 if (timer->op_queue_.empty())
220 remove_timer(*timer);
221 }
222 }

224 // Move operations from one timer to another, empty timer.
225 void move_timer(per_timer_data& target, per_timer_data& source)
226 {
227 target.op_queue_.push(source.op_queue_);
228
229 target.heap_index_ = source.heap_index_;
230 source.heap_index_ = (std::numeric_limits<std::size_t>::max)();
231
232 if (target.heap_index_ < heap_.size())
233 heap_[target.heap_index_].timer_ = &target;
234
235 if (timers_ == &source)
236 timers_ = &target;
237 if (source.prev_)
238 source.prev_->next_ = &target;
239 if (source.next_)
240 source.next_->prev_= &target;
241 target.next_ = source.next_;
242 target.prev_ = source.prev_;
243 source.next_ = 0;
244 source.prev_ = 0;
245 }

247private:
248 // Move the item at the given index up the heap to its correct position.
249 void up_heap(std::size_t index)
250 {
251 while (index > 0)
252 {
253 std::size_t parent = (index - 1) / 2;
254 if (!Time_Traits::less_than(heap_[index].time_, heap_[parent].time_))
255 break;
256 swap_heap(index, parent);
257 index = parent;
258 }
259 }
260
261 // Move the item at the given index down the heap to its correct position.
262 void down_heap(std::size_t index)
263 {
264 std::size_t child = index * 2 + 1;
265 while (child < heap_.size())
266 {
267 std::size_t min_child = (child + 1 == heap_.size()
268 || Time_Traits::less_than(
269 heap_[child].time_, heap_[child + 1].time_))
270 ? child : child + 1;
271 if (Time_Traits::less_than(heap_[index].time_, heap_[min_child].time_))
272 break;
273 swap_heap(index, min_child);
274 index = min_child;
275 child = index * 2 + 1;
276 }
277 }
278
279 // Swap two entries in the heap.
280 void swap_heap(std::size_t index1, std::size_t index2)
281 {
282 heap_entry tmp = heap_[index1];
283 heap_[index1] = heap_[index2];
284 heap_[index2] = tmp;
285 heap_[index1].timer_->heap_index_ = index1;
286 heap_[index2].timer_->heap_index_ = index2;
287 }
288
289 // Remove a timer from the heap and list of timers.
290 void remove_timer(per_timer_data& timer)
291 {
292 // Remove the timer from the heap.
293 std::size_t index = timer.heap_index_;
294 if (!heap_.empty() && index < heap_.size())
295 {
296 if (index == heap_.size() - 1)
297 {
298 timer.heap_index_ = (std::numeric_limits<std::size_t>::max)();
299 heap_.pop_back();
300 }
301 else
302 {
303 swap_heap(index, heap_.size() - 1);
304 timer.heap_index_ = (std::numeric_limits<std::size_t>::max)();
305 heap_.pop_back();
306 if (index > 0 && Time_Traits::less_than(
307 heap_[index].time_, heap_[(index - 1) / 2].time_))
308 up_heap(index);
309 else
310 down_heap(index);
311 }
312 }
313
314 // Remove the timer from the linked list of active timers.
315 if (timers_ == &timer)
316 timers_ = timer.next_;
317 if (timer.prev_)
318 timer.prev_->next_ = timer.next_;
319 if (timer.next_)
320 timer.next_->prev_= timer.prev_;
321 timer.next_ = 0;
322 timer.prev_ = 0;
323 }

340 // Helper function to convert a duration into milliseconds.
341 template <typename Duration>
342 long to_msec(const Duration& d, long max_duration) const
343 {
344 if (d.ticks() <= 0)
345 return 0;
346 int64_t msec = d.total_milliseconds();
347 if (msec == 0)
348 return 1;
349 if (msec > max_duration)
350 return max_duration;
351 return static_cast<long>(msec);
352 }
353
354 // Helper function to convert a duration into microseconds.
355 template <typename Duration>
356 long to_usec(const Duration& d, long max_duration) const
357 {
358 if (d.ticks() <= 0)
359 return 0;
360 int64_t usec = d.total_microseconds();
361 if (usec == 0)
362 return 1;
363 if (usec > max_duration)
364 return max_duration;
365 return static_cast<long>(usec);
366 }
