59 static void do_complete(void* owner, operation* base,
60 const asio::error_code& result_ec,
61 std::size_t /*bytes_transferred*/)
62 {
63 asio::error_code ec(result_ec);
64
65 // Take ownership of the operation object.
66 ASIO_ASSUME(base != 0);
67 win_iocp_wait_op* o(static_cast<win_iocp_wait_op*>(base));
68 ptr p = { asio::detail::addressof(o->handler_), o, o };
69
70 ASIO_HANDLER_COMPLETION((*o));
71
72 // Take ownership of the operation's outstanding work.
73 handler_work<Handler, IoExecutor> w(
74 static_cast<handler_work<Handler, IoExecutor>&&>(
75 o->work_));
76
77 // The reactor may have stored a result in the operation object.
78 if (o->ec_)
79 ec = o->ec_;
80
81 // Map non-portable errors to their portable counterparts.
82 if (ec.value() == ERROR_NETNAME_DELETED)
83 {
84 if (o->cancel_token_.expired())
85 ec = asio::error::operation_aborted;
86 else
87 ec = asio::error::connection_reset;
88 }
89 else if (ec.value() == ERROR_PORT_UNREACHABLE)
90 {
91 ec = asio::error::connection_refused;
92 }
93
94 ASIO_ERROR_LOCATION(ec);
95
96 // Make a copy of the handler so that the memory can be deallocated before
97 // the upcall is made. Even if we're not about to make an upcall, a
98 // sub-object of the handler may be the true owner of the memory associated
99 // with the handler. Consequently, a local copy of the handler is required
100 // to ensure that any owning sub-object remains valid until after we have
101 // deallocated the memory here.
102 detail::binder1<Handler, asio::error_code>
103 handler(o->handler_, ec);
104 p.h = asio::detail::addressof(handler.handler_);
105 p.reset();
106
107 // Make the upcall if required.
108 if (owner)
109 {
110 fenced_block b(fenced_block::half);
111 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
112 w.complete(handler, handler.handler_);
113 ASIO_HANDLER_INVOCATION_END;
114 }
115 }
116
117private:
118 socket_ops::weak_cancel_token_type cancel_token_;
119 Handler handler_;
120 handler_work<Handler, IoExecutor> work_;
121};
