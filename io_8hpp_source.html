32template <typename Stream, typename Operation>
33std::size_t io(Stream& next_layer, stream_core& core,
34 const Operation& op, asio::error_code& ec)
35{
36 asio::error_code io_ec;
37 std::size_t bytes_transferred = 0;
38 do switch (op(core.engine_, ec, bytes_transferred))
39 {
40 case engine::want_input_and_retry:
41
42 // If the input buffer is empty then we need to read some more data from
43 // the underlying transport.
44 if (core.input_.size() == 0)
45 {
46 core.input_ = asio::buffer(core.input_buffer_,
47 next_layer.read_some(core.input_buffer_, io_ec));
48 if (!ec)
49 ec = io_ec;
50 }
51
52 // Pass the new input data to the engine.
53 core.input_ = core.engine_.put_input(core.input_);
54
55 // Try the operation again.
56 continue;
57
58 case engine::want_output_and_retry:
59
60 // Get output data from the engine and write it to the underlying
61 // transport.
62 asio::write(next_layer,
63 core.engine_.get_output(core.output_buffer_), io_ec);
64 if (!ec)
65 ec = io_ec;
66
67 // Try the operation again.
68 continue;
69
70 case engine::want_output:
71
72 // Get output data from the engine and write it to the underlying
73 // transport.
74 asio::write(next_layer,
75 core.engine_.get_output(core.output_buffer_), io_ec);
76 if (!ec)
77 ec = io_ec;
78
79 // Operation is complete. Return result to caller.
80 core.engine_.map_error_code(ec);
81 return bytes_transferred;
82
83 default:
84
85 // Operation is complete. Return result to caller.
86 core.engine_.map_error_code(ec);
87 return bytes_transferred;
88
89 } while (!ec);
90
91 // Operation failed. Return result to caller.
92 core.engine_.map_error_code(ec);
93 return 0;
94}

96template <typename Stream, typename Operation, typename Handler>
97class io_op
98 : public asio::detail::base_from_cancellation_state<Handler>
99{
100public:
101 io_op(Stream& next_layer, stream_core& core,
102 const Operation& op, Handler& handler)
103 : asio::detail::base_from_cancellation_state<Handler>(handler),
104 next_layer_(next_layer),
105 core_(core),
106 op_(op),
107 start_(0),
108 want_(engine::want_nothing),
109 bytes_transferred_(0),
110 handler_(static_cast<Handler&&>(handler))
111 {
112 }

115 : asio::detail::base_from_cancellation_state<Handler>(other),
116 next_layer_(other.next_layer_),
117 core_(other.core_),
118 op_(other.op_),
119 start_(other.start_),
120 want_(other.want_),
121 ec_(other.ec_),
122 bytes_transferred_(other.bytes_transferred_),
123 handler_(other.handler_)
124 {
125 }
126
127 io_op(io_op&& other)
128 : asio::detail::base_from_cancellation_state<Handler>(
129 static_cast<
130 asio::detail::base_from_cancellation_state<Handler>&&>(other)),
131 next_layer_(other.next_layer_),
132 core_(other.core_),
133 op_(static_cast<Operation&&>(other.op_)),
134 start_(other.start_),
135 want_(other.want_),
136 ec_(other.ec_),
137 bytes_transferred_(other.bytes_transferred_),
138 handler_(static_cast<Handler&&>(other.handler_))
139 {
140 }
141
142 void operator()(asio::error_code ec,
143 std::size_t bytes_transferred = ~std::size_t(0), int start = 0)
144 {
145 switch (start_ = start)
146 {
147 case 1: // Called after at least one async operation.
148 do
149 {
150 switch (want_ = op_(core_.engine_, ec_, bytes_transferred_))
151 {
152 case engine::want_input_and_retry:
153
154 // If the input buffer already has data in it we can pass it to the
155 // engine and then retry the operation immediately.
156 if (core_.input_.size() != 0)
157 {
158 core_.input_ = core_.engine_.put_input(core_.input_);
159 continue;
160 }
161
162 // The engine wants more data to be read from input. However, we
163 // cannot allow more than one read operation at a time on the
164 // underlying transport. The pending_read_ timer's expiry is set to
165 // pos_infin if a read is in progress, and neg_infin otherwise.
166 if (core_.expiry(core_.pending_read_) == core_.neg_infin())
167 {
168 // Prevent other read operations from being started.
169 core_.pending_read_.expires_at(core_.pos_infin());
170
171 ASIO_HANDLER_LOCATION((
172 __FILE__, __LINE__, Operation::tracking_name()));
173
174 // Start reading some data from the underlying transport.
175 next_layer_.async_read_some(
176 asio::buffer(core_.input_buffer_),
177 static_cast<io_op&&>(*this));
178 }
179 else
180 {
181 ASIO_HANDLER_LOCATION((
182 __FILE__, __LINE__, Operation::tracking_name()));
183
184 // Wait until the current read operation completes.
185 core_.pending_read_.async_wait(static_cast<io_op&&>(*this));
186 }
187
188 // Yield control until asynchronous operation completes. Control
189 // resumes at the "default:" label below.
190 return;
191
192 case engine::want_output_and_retry:
193 case engine::want_output:
194
195 // The engine wants some data to be written to the output. However, we
196 // cannot allow more than one write operation at a time on the
197 // underlying transport. The pending_write_ timer's expiry is set to
198 // pos_infin if a write is in progress, and neg_infin otherwise.
199 if (core_.expiry(core_.pending_write_) == core_.neg_infin())
200 {
201 // Prevent other write operations from being started.
202 core_.pending_write_.expires_at(core_.pos_infin());
203
204 ASIO_HANDLER_LOCATION((
205 __FILE__, __LINE__, Operation::tracking_name()));
206
207 // Start writing all the data to the underlying transport.
208 asio::async_write(next_layer_,
209 core_.engine_.get_output(core_.output_buffer_),
210 static_cast<io_op&&>(*this));
211 }
212 else
213 {
214 ASIO_HANDLER_LOCATION((
215 __FILE__, __LINE__, Operation::tracking_name()));
216
217 // Wait until the current write operation completes.
218 core_.pending_write_.async_wait(static_cast<io_op&&>(*this));
219 }
220
221 // Yield control until asynchronous operation completes. Control
222 // resumes at the "default:" label below.
223 return;
224
225 default:
226
227 // The SSL operation is done and we can invoke the handler, but we
228 // have to keep in mind that this function might be being called from
229 // the async operation's initiating function. In this case we're not
230 // allowed to call the handler directly. Instead, issue a zero-sized
231 // read so the handler runs "as-if" posted using io_context::post().
232 if (start)
233 {
234 ASIO_HANDLER_LOCATION((
235 __FILE__, __LINE__, Operation::tracking_name()));
236
237 next_layer_.async_read_some(
238 asio::buffer(core_.input_buffer_, 0),
239 static_cast<io_op&&>(*this));
240
241 // Yield control until asynchronous operation completes. Control
242 // resumes at the "default:" label below.
243 return;
244 }
245 else
246 {
247 // Continue on to run handler directly.
248 break;
249 }
250 }
251
252 default:
253 if (bytes_transferred == ~std::size_t(0))
254 bytes_transferred = 0; // Timer cancellation, no data transferred.
255 else if (!ec_)
256 ec_ = ec;
257
258 switch (want_)
259 {
260 case engine::want_input_and_retry:
261
262 // Add received data to the engine's input.
263 core_.input_ = asio::buffer(
264 core_.input_buffer_, bytes_transferred);
265 core_.input_ = core_.engine_.put_input(core_.input_);
266
267 // Release any waiting read operations.
268 core_.pending_read_.expires_at(core_.neg_infin());
269
270 // Check for cancellation before continuing.
271 if (this->cancelled() != cancellation_type::none)
272 {
273 ec_ = asio::error::operation_aborted;
274 break;
275 }
276
277 // Try the operation again.
278 continue;
279
280 case engine::want_output_and_retry:
281
282 // Release any waiting write operations.
283 core_.pending_write_.expires_at(core_.neg_infin());
284
285 // Check for cancellation before continuing.
286 if (this->cancelled() != cancellation_type::none)
287 {
288 ec_ = asio::error::operation_aborted;
289 break;
290 }
291
292 // Try the operation again.
293 continue;
294
295 case engine::want_output:
296
297 // Release any waiting write operations.
298 core_.pending_write_.expires_at(core_.neg_infin());
299
300 // Fall through to call handler.
301
302 default:
303
304 // Pass the result to the handler.
305 op_.call_handler(handler_,
306 core_.engine_.map_error_code(ec_),
307 ec_ ? 0 : bytes_transferred_);
308
309 // Our work here is done.
310 return;
311 }
312 } while (!ec_);
313
314 // Operation failed. Pass the result to the handler.
315 op_.call_handler(handler_, core_.engine_.map_error_code(ec_), 0);
316 }
317 }
318
319//private:
320 Stream& next_layer_;
321 stream_core& core_;
322 Operation op_;
323 int start_;
324 engine::want want_;
325 asio::error_code ec_;
326 std::size_t bytes_transferred_;
327 Handler handler_;
328};

330template <typename Stream, typename Operation, typename Handler>
331inline bool asio_handler_is_continuation(
332 io_op<Stream, Operation, Handler>* this_handler)
333{
