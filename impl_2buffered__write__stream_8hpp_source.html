170template <typename Stream>
171template <typename ConstBufferSequence>
172std::size_t buffered_write_stream<Stream>::write_some(
173 const ConstBufferSequence& buffers)
174{
175 using asio::buffer_size;
176 if (buffer_size(buffers) == 0)
177 return 0;
178
179 if (storage_.size() == storage_.capacity())
180 this->flush();
181
182 return this->copy(buffers);
183}

247 storage_.data() + orig_size, buffers_, length);
248 static_cast<WriteHandler&&>(handler_)(ec, bytes_copied);

367std::size_t buffered_write_stream<Stream>::copy(
368 const ConstBufferSequence& buffers)
369{
370 using asio::buffer_size;
371 std::size_t orig_size = storage_.size();
372 std::size_t space_avail = storage_.capacity() - orig_size;
373 std::size_t bytes_avail = buffer_size(buffers);
374 std::size_t length = bytes_avail < space_avail ? bytes_avail : space_avail;
375 storage_.resize(orig_size + length);
376 return asio::buffer_copy(
377 storage_.data() + orig_size, buffers, length);
