279 : true

281 &&
282 (
283 is_preferable<Prop>::value
284 ? can_prefer<T, Prop>::value
285 : true
286 )
287 &&
288 (
289 !is_requirable<Prop>::value && !is_preferable<Prop>::value
290 ? can_query<T, Prop>::value
291 : true
292 )
293 >
294 {
295 };

310 template <typename T>
311 struct find_convertible_property :
312 conditional_t<
313 is_same<T, Prop>::value || is_convertible<T, Prop>::value,
314 found,
315 not_found
316 > {};

413class shared_target_executor
414{
415public:
416 template <typename E>
417 shared_target_executor(E&& e, decay_t<E>*& target)
418 {
419 impl<decay_t<E>>* i =
420 new impl<decay_t<E>>(static_cast<E&&>(e));
421 target = &i->ex_;
422 impl_ = i;
423 }
424
425 template <typename E>
426 shared_target_executor(std::nothrow_t, E&& e, decay_t<E>*& target) noexcept
427 {
428 impl<decay_t<E>>* i =
429 new (std::nothrow) impl<decay_t<E>>(static_cast<E&&>(e));
430 target = i ? &i->ex_ : 0;
431 impl_ = i;
432 }
433
434 shared_target_executor(const shared_target_executor& other) noexcept
435 : impl_(other.impl_)
436 {
437 if (impl_)
438 asio::detail::ref_count_up(impl_->ref_count_);
439 }
440
441 shared_target_executor(shared_target_executor&& other) noexcept
442 : impl_(other.impl_)
443 {
444 other.impl_ = 0;
445 }
446
447 ~shared_target_executor()
448 {
449 if (impl_)
450 if (asio::detail::ref_count_down(impl_->ref_count_))
451 delete impl_;
452 }
453
454 void* get() const noexcept
455 {
456 return impl_ ? impl_->get() : 0;
457 }
458
459private:
460 shared_target_executor& operator=(
461 const shared_target_executor& other) = delete;
462
463 shared_target_executor& operator=(
464 shared_target_executor&& other) = delete;
