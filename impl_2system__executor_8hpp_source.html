91template <typename Blocking, typename Relationship, typename Allocator>
92template <typename Function>
93void basic_system_executor<Blocking, Relationship, Allocator>::do_execute(
94 Function&& f, execution::blocking_t::never_t) const
95{
96 system_context& ctx = detail::global<system_context>();
97
98 // Allocate and construct an operation to wrap the function.
99 typedef detail::executor_op<decay_t<Function>, Allocator> op;
100 typename op::ptr p = { detail::addressof(allocator_),
101 op::ptr::allocate(allocator_), 0 };
102 p.p = new (p.v) op(static_cast<Function&&>(f), allocator_);
103
104 if (is_same<Relationship, execution::relationship_t::continuation_t>::value)
105 {
106 ASIO_HANDLER_CREATION((ctx, *p.p,
107 "system_executor", &ctx, 0, "execute(blk=never,rel=cont)"));
108 }
109 else
110 {
111 ASIO_HANDLER_CREATION((ctx, *p.p,
112 "system_executor", &ctx, 0, "execute(blk=never,rel=fork)"));
113 }
114
115 ctx.scheduler_.post_immediate_completion(p.p,
116 is_same<Relationship, execution::relationship_t::continuation_t>::value);
117 p.v = p.p = 0;
118}
