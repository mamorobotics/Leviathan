102template <typename Executors, typename Handler,
103 typename Return, typename R, typename... Args>
104class co_composed_state_return_overload<
105 Executors, Handler, Return, R(Args...)>
106{
107public:
108 using derived_type = co_composed_state<Executors, Handler, Return>;
109 using promise_type = co_composed_promise<Executors, Handler, Return>;
110 using return_type = std::tuple<Args...>;
111
112 void on_cancellation_complete_with(Args... args)
113 {
114 derived_type& state = *static_cast<derived_type*>(this);
115 state.return_value_ = std::make_tuple(std::move(args)...);
116 state.cancellation_on_suspend_fn(
117 [](void* p)
118 {
119 auto& promise = *static_cast<promise_type*>(p);
120
121 co_composed_handler_base<Executors, Handler,
122 Return> composed_handler(promise);
123
124 Handler handler(std::move(promise.state().handler_));
125 return_type result(
126 std::move(std::get<return_type>(promise.state().return_value_)));
127
128 co_composed_handler_base<Executors, Handler,
129 Return>(std::move(composed_handler));
130
131 std::apply(std::move(handler), std::move(result));
132 });
133 }
134};

359
360 void check_for_cancellation_on_transform()
361 {
362 if (throw_if_cancelled_ && !!cancelled())

365

367 co_composed_promise<Executors, Handler, Return>& promise) noexcept

369 if (complete_if_cancelled_ && !!cancelled() && cancellation_on_suspend_fn_)
370 {
371 promise.state().work_.reset();

375 }

384 bool complete_if_cancelled_ = true;
