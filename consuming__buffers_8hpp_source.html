49// A buffer sequence used to represent a subsequence of the buffers.
50template <typename Buffer, std::size_t MaxBuffers>
51struct prepared_buffers
52{
53 typedef Buffer value_type;
54 typedef const Buffer* const_iterator;
55
56 enum { max_buffers = MaxBuffers < 16 ? MaxBuffers : 16 };
57
58 prepared_buffers() : count(0) {}
59 const_iterator begin() const { return elems; }
60 const_iterator end() const { return elems + count; }
61
62 Buffer elems[max_buffers];
63 std::size_t count;
64};

66// A proxy for a sub-range in a list of buffers.
67template <typename Buffer, typename Buffers, typename Buffer_Iterator>
68class consuming_buffers
69{
70public:
71 typedef prepared_buffers<Buffer, prepared_buffers_max<Buffers>::value>
72 prepared_buffers_type;
73
74 // Construct to represent the entire list of buffers.
75 explicit consuming_buffers(const Buffers& buffers)
76 : buffers_(buffers),
77 total_consumed_(0),
78 next_elem_(0),
79 next_elem_offset_(0)
80 {
81 using asio::buffer_size;
82 total_size_ = buffer_size(buffers);
83 }
84
85 // Determine if we are at the end of the buffers.
86 bool empty() const
87 {
88 return total_consumed_ >= total_size_;
89 }
90
91 // Get the buffer for a single transfer, with a size.
92 prepared_buffers_type prepare(std::size_t max_size)
93 {
94 prepared_buffers_type result;
95
96 Buffer_Iterator next = asio::buffer_sequence_begin(buffers_);
97 Buffer_Iterator end = asio::buffer_sequence_end(buffers_);
98
99 std::advance(next, next_elem_);
100 std::size_t elem_offset = next_elem_offset_;
101 while (next != end && max_size > 0 && (result.count) < result.max_buffers)
102 {
103 Buffer next_buf = Buffer(*next) + elem_offset;
104 result.elems[result.count] = asio::buffer(next_buf, max_size);
105 max_size -= result.elems[result.count].size();
106 elem_offset = 0;
107 if (result.elems[result.count].size() > 0)
108 ++result.count;
109 ++next;
110 }
111
112 return result;
113 }
114
115 // Consume the specified number of bytes from the buffers.
116 void consume(std::size_t size)
117 {
118 total_consumed_ += size;
119
120 Buffer_Iterator next = asio::buffer_sequence_begin(buffers_);
121 Buffer_Iterator end = asio::buffer_sequence_end(buffers_);
122
123 std::advance(next, next_elem_);
124 while (next != end && size > 0)
125 {
126 Buffer next_buf = Buffer(*next) + next_elem_offset_;
127 if (size < next_buf.size())
128 {
129 next_elem_offset_ += size;
130 size = 0;
131 }
132 else
133 {
134 size -= next_buf.size();
135 next_elem_offset_ = 0;
136 ++next_elem_;
137 ++next;
138 }
139 }
140 }
141
142 // Get the total number of bytes consumed from the buffers.
143 std::size_t total_consumed() const
144 {
145 return total_consumed_;
146 }
147
148private:
149 Buffers buffers_;
150 std::size_t total_size_;
151 std::size_t total_consumed_;
152 std::size_t next_elem_;
153 std::size_t next_elem_offset_;
154};

309public:

312 : buffers_(buffers),

315 }
316
317 // Determine if we are at the end of the buffers.

319 {
320 return total_consumed_ >=

322 }
323
324 // Get the buffer for a single transfer, with a size.
325 boost::array<Buffer, 2> prepare(std::size_t max_size)
326 {
327 boost::array<Buffer, 2> result = {{
328 Buffer(buffers_[0]), Buffer(buffers_[1]) }};
329 std::size_t buffer0_size = result[0].size();
330 result[0] = asio::buffer(result[0] + total_consumed_, max_size);
331 result[1] = asio::buffer(
332 result[1] + (total_consumed_ < buffer0_size
333 ? 0 : total_consumed_ - buffer0_size),
334 max_size - result[0].size());
