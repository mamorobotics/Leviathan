65template <typename Clock, typename Duration>
66std::size_t io_context::run_until(
67 const chrono::time_point<Clock, Duration>& abs_time)
68{
69 std::size_t n = 0;
70 while (this->run_one_until(abs_time))
71 if (n != (std::numeric_limits<std::size_t>::max)())
72 ++n;
73 return n;

83template <typename Clock, typename Duration>
84std::size_t io_context::run_one_until(
85 const chrono::time_point<Clock, Duration>& abs_time)
86{
87 typename Clock::time_point now = Clock::now();
88 while (now < abs_time)
89 {
90 typename Clock::duration rel_time = abs_time - now;
91 if (rel_time > chrono::seconds(1))
92 rel_time = chrono::seconds(1);
93
94 asio::error_code ec;
95 std::size_t s = impl_.wait_one(
96 static_cast<long>(chrono::duration_cast<
97 chrono::microseconds>(rel_time).count()), ec);
98 asio::detail::throw_error(ec);
99
100 if (s || impl_.stopped())
101 return s;
102
103 now = Clock::now();
104 }
105
106 return 0;
107}

116struct io_context::initiate_dispatch
117{
118 template <typename LegacyCompletionHandler>
119 void operator()(LegacyCompletionHandler&& handler,
120 io_context* self) const
121 {
122 // If you get an error on the following line it means that your handler does
123 // not meet the documented type requirements for a LegacyCompletionHandler.
124 ASIO_LEGACY_COMPLETION_HANDLER_CHECK(
125 LegacyCompletionHandler, handler) type_check;
126
127 detail::non_const_lvalue<LegacyCompletionHandler> handler2(handler);
128 if (self->impl_.can_dispatch())
129 {
130 detail::fenced_block b(detail::fenced_block::full);
131 static_cast<decay_t<LegacyCompletionHandler>&&>(handler2.value)();
132 }
133 else
134 {
135 // Allocate and construct an operation to wrap the handler.
136 typedef detail::completion_handler<
137 decay_t<LegacyCompletionHandler>, executor_type> op;
138 typename op::ptr p = { detail::addressof(handler2.value),
139 op::ptr::allocate(handler2.value), 0 };
140 p.p = new (p.v) op(handler2.value, self->get_executor());
141
142 ASIO_HANDLER_CREATION((*self, *p.p,
143 "io_context", self, 0, "dispatch"));
144
145 self->impl_.do_dispatch(p.p);
146 p.v = p.p = 0;
147 }
148 }
149};

217io_context::basic_executor_type<Allocator, Bits>::operator=(
218 const basic_executor_type& other) noexcept
219{
220 if (this != &other)
221 {
222 static_cast<Allocator&>(*this) = static_cast<const Allocator&>(other);
223 io_context* old_io_context = context_ptr();
224 target_ = other.target_;
225 if (Bits & outstanding_work_tracked)
226 {
227 if (context_ptr())
228 context_ptr()->impl_.work_started();
229 if (old_io_context)
230 old_io_context->impl_.work_finished();
231 }
232 }
233 return *this;
234}

263template <typename Allocator, uintptr_t Bits>
264template <typename Function>
265void io_context::basic_executor_type<Allocator, Bits>::execute(
266 Function&& f) const
267{
268 typedef decay_t<Function> function_type;
269
270 // Invoke immediately if the blocking.possibly property is enabled and we are
271 // already inside the thread pool.
272 if ((bits() & blocking_never) == 0 && context_ptr()->impl_.can_dispatch())
273 {
274 // Make a local, non-const copy of the function.
275 function_type tmp(static_cast<Function&&>(f));
276
277#if !defined(ASIO_NO_EXCEPTIONS)
278 try
279 {
280#endif // !defined(ASIO_NO_EXCEPTIONS)
281 detail::fenced_block b(detail::fenced_block::full);
282 static_cast<function_type&&>(tmp)();
283 return;
284#if !defined(ASIO_NO_EXCEPTIONS)
285 }
286 catch (...)
287 {
288 context_ptr()->impl_.capture_current_exception();
289 return;
290 }
291#endif // !defined(ASIO_NO_EXCEPTIONS)
292 }
293
294 // Allocate and construct an operation to wrap the function.
295 typedef detail::executor_op<function_type, Allocator, detail::operation> op;
296 typename op::ptr p = {
297 detail::addressof(static_cast<const Allocator&>(*this)),
298 op::ptr::allocate(static_cast<const Allocator&>(*this)), 0 };
299 p.p = new (p.v) op(static_cast<Function&&>(f),
300 static_cast<const Allocator&>(*this));
301
302 ASIO_HANDLER_CREATION((*context_ptr(), *p.p,
303 "io_context", context_ptr(), 0, "execute"));
304
305 context_ptr()->impl_.post_immediate_completion(p.p,
306 (bits() & relationship_continuation) != 0);
307 p.v = p.p = 0;
308}
