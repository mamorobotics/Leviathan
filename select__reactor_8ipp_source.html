84void select_reactor::shutdown()
85{
86 asio::detail::mutex::scoped_lock lock(mutex_);
87 shutdown_ = true;
88#if defined(ASIO_HAS_IOCP)
89 stop_thread_ = true;
90 if (thread_)
91 interrupter_.interrupt();
92#endif // defined(ASIO_HAS_IOCP)
93 lock.unlock();
94
95#if defined(ASIO_HAS_IOCP)
96 if (thread_)
97 {
98 thread_->join();
99 delete thread_;
100 thread_ = 0;
101 }
102#endif // defined(ASIO_HAS_IOCP)
103
104 op_queue<operation> ops;
105
106 for (int i = 0; i < max_ops; ++i)
107 op_queue_[i].get_all_operations(ops);
108
109 timer_queues_.get_all_timers(ops);
110
111 scheduler_.abandon_operations(ops);
112}

162 select_reactor::per_descriptor_data&, reactor_op* op, bool is_continuation,
163 bool, void (*on_immediate)(operation*, bool, const void*),
164 const void* immediate_arg)
165{
166 asio::detail::mutex::scoped_lock lock(mutex_);
167
168 if (shutdown_)

170 on_immediate(op, is_continuation, immediate_arg);
171 return;
172 }
173
174 bool first = op_queue_[op_type].enqueue_operation(descriptor, op);
175 scheduler_.work_started();
176 if (first)
177 interrupter_.interrupt();

221void select_reactor::run(long usec, op_queue<operation>& ops)
222{
223 asio::detail::mutex::scoped_lock lock(mutex_);
224
225#if defined(ASIO_HAS_IOCP)
226 // Check if the thread is supposed to stop.
227 if (stop_thread_)
228 return;
229#endif // defined(ASIO_HAS_IOCP)
230
231 // Set up the descriptor sets.
232 for (int i = 0; i < max_select_ops; ++i)
233 fd_sets_[i].reset();
234 fd_sets_[read_op].set(interrupter_.read_descriptor());
235 socket_type max_fd = 0;
236 bool have_work_to_do = !timer_queues_.all_empty();
237 for (int i = 0; i < max_select_ops; ++i)
238 {
239 have_work_to_do = have_work_to_do || !op_queue_[i].empty();
240 fd_sets_[i].set(op_queue_[i], ops);
241 if (fd_sets_[i].max_descriptor() > max_fd)
242 max_fd = fd_sets_[i].max_descriptor();
243 }
244
245#if defined(ASIO_WINDOWS) || defined(__CYGWIN__)
246 // Connection operations on Windows use both except and write fd_sets.
247 have_work_to_do = have_work_to_do || !op_queue_[connect_op].empty();
248 fd_sets_[write_op].set(op_queue_[connect_op], ops);
249 if (fd_sets_[write_op].max_descriptor() > max_fd)
250 max_fd = fd_sets_[write_op].max_descriptor();
251 fd_sets_[except_op].set(op_queue_[connect_op], ops);
252 if (fd_sets_[except_op].max_descriptor() > max_fd)
253 max_fd = fd_sets_[except_op].max_descriptor();
254#endif // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
255
256 // We can return immediately if there's no work to do and the reactor is
257 // not supposed to block.
258 if (!usec && !have_work_to_do)
259 return;
260
261 // Determine how long to block while waiting for events.
262 timeval tv_buf = { 0, 0 };
263 timeval* tv = usec ? get_timeout(usec, tv_buf) : &tv_buf;
264
265 lock.unlock();
266
267 // Block on the select call until descriptors become ready.
268 asio::error_code ec;
269 int retval = socket_ops::select(static_cast<int>(max_fd + 1),
270 fd_sets_[read_op], fd_sets_[write_op], fd_sets_[except_op], tv, ec);
271
272 // Reset the interrupter.
273 if (retval > 0 && fd_sets_[read_op].is_set(interrupter_.read_descriptor()))
274 {
275 if (!interrupter_.reset())
276 {
277 lock.lock();
278#if defined(ASIO_HAS_IOCP)
279 stop_thread_ = true;
280 scheduler_.post_immediate_completion(&restart_reactor_, false);
281#else // defined(ASIO_HAS_IOCP)
282 interrupter_.recreate();
283#endif // defined(ASIO_HAS_IOCP)
284 }
285 --retval;
286 }
287
288 lock.lock();
289
290 // Dispatch all ready operations.
291 if (retval > 0)
292 {
293#if defined(ASIO_WINDOWS) || defined(__CYGWIN__)
294 // Connection operations on Windows use both except and write fd_sets.
295 fd_sets_[except_op].perform(op_queue_[connect_op], ops);
296 fd_sets_[write_op].perform(op_queue_[connect_op], ops);
297#endif // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
298
299 // Exception operations must be processed first to ensure that any
300 // out-of-band data is read before normal data.
301 for (int i = max_select_ops - 1; i >= 0; --i)
302 fd_sets_[i].perform(op_queue_[i], ops);
303 }
304 timer_queues_.get_ready_timers(ops);
305}

313void select_reactor::run_thread()
314{
315 asio::detail::mutex::scoped_lock lock(mutex_);
316 while (!stop_thread_)
317 {
318 lock.unlock();
319 op_queue<operation> ops;
320 run(-1, ops);
321 scheduler_.post_deferred_completions(ops);
322 lock.lock();
323 }

326void select_reactor::restart_reactor::do_complete(void* owner, operation* base,
327 const asio::error_code& /*ec*/, std::size_t /*bytes_transferred*/)
328{
329 if (owner)
330 {
331 select_reactor* reactor = static_cast<restart_reactor*>(base)->reactor_;
332
333 if (reactor->thread_)
334 {
335 reactor->thread_->join();
336 delete reactor->thread_;
337 reactor->thread_ = 0;
338 }
339
340 asio::detail::mutex::scoped_lock lock(reactor->mutex_);
341 reactor->interrupter_.recreate();
342 reactor->stop_thread_ = false;
343 lock.unlock();
344
345 asio::detail::signal_blocker sb;
346 reactor->thread_ =
347 new asio::detail::thread(thread_function(reactor));
348 }
349}

364timeval* select_reactor::get_timeout(long usec, timeval& tv)
365{
366 // By default we will wait no longer than 5 minutes. This will ensure that
367 // any changes to the system clock are detected after no longer than this.
368 const long max_usec = 5 * 60 * 1000 * 1000;
369 usec = timer_queues_.wait_duration_usec(
370 (usec < 0 || max_usec < usec) ? max_usec : usec);
371 tv.tv_sec = usec / 1000000;
372 tv.tv_usec = usec % 1000000;
373 return &tv;
374}

376void select_reactor::cancel_ops_unlocked(socket_type descriptor,
377 const asio::error_code& ec)
378{
379 bool need_interrupt = false;
380 op_queue<operation> ops;
381 for (int i = 0; i < max_ops; ++i)
382 need_interrupt = op_queue_[i].cancel_operations(
383 descriptor, ops, ec) || need_interrupt;
384 scheduler_.post_deferred_completions(ops);
385 if (need_interrupt)
386 interrupter_.interrupt();
387}
