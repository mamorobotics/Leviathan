171 // Start an asynchronous operation to wait for a signal to be delivered.
172 template <typename Handler, typename IoExecutor>
173 void async_wait(implementation_type& impl,
174 Handler& handler, const IoExecutor& io_ex)
175 {
176 associated_cancellation_slot_t<Handler> slot
177 = asio::get_associated_cancellation_slot(handler);
178
179 // Allocate and construct an operation to wrap the handler.
180 typedef signal_handler<Handler, IoExecutor> op;
181 typename op::ptr p = { asio::detail::addressof(handler),
182 op::ptr::allocate(handler), 0 };
183 p.p = new (p.v) op(handler, io_ex);
184
185 // Optionally register for per-operation cancellation.
186 if (slot.is_connected())
187 {
188 p.p->cancellation_key_ =
189 &slot.template emplace<signal_op_cancellation>(this, &impl);
190 }
191
192 ASIO_HANDLER_CREATION((scheduler_.context(),
193 *p.p, "signal_set", &impl, 0, "async_wait"));
194
195 start_wait_op(impl, p.p);
196 p.v = p.p = 0;
197 }
