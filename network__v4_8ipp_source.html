45network_v4::network_v4(const address_v4& addr, const address_v4& mask)
46 : address_(addr),
47 prefix_length_(0)
48{
49 address_v4::bytes_type mask_bytes = mask.to_bytes();
50 bool finished = false;
51 for (std::size_t i = 0; i < mask_bytes.size(); ++i)
52 {
53 if (finished)
54 {
55 if (mask_bytes[i])
56 {
57 std::invalid_argument ex("non-contiguous netmask");
58 asio::detail::throw_exception(ex);
59 }
60 continue;
61 }
62 else
63 {
64 switch (mask_bytes[i])
65 {
66 case 255:
67 prefix_length_ += 8;
68 break;
69 case 254: // prefix_length_ += 7
70 prefix_length_ += 1;
71 case 252: // prefix_length_ += 6
72 prefix_length_ += 1;
73 case 248: // prefix_length_ += 5
74 prefix_length_ += 1;
75 case 240: // prefix_length_ += 4
76 prefix_length_ += 1;
77 case 224: // prefix_length_ += 3
78 prefix_length_ += 1;
79 case 192: // prefix_length_ += 2
80 prefix_length_ += 1;
81 case 128: // prefix_length_ += 1
82 prefix_length_ += 1;
83 case 0: // nbits += 0
84 finished = true;
85 break;
86 default:
87 std::out_of_range ex("non-contiguous netmask");
88 asio::detail::throw_exception(ex);
89 }
90 }
91 }
92}

94address_v4 network_v4::netmask() const noexcept
95{
96 uint32_t nmbits = 0xffffffff;
97 if (prefix_length_ == 0)
98 nmbits = 0;
99 else
100 nmbits = nmbits << (32 - prefix_length_);
101 return address_v4(nmbits);
102}

111bool network_v4::is_subnet_of(const network_v4& other) const
112{
113 if (other.prefix_length_ >= prefix_length_)
114 return false; // Only real subsets are allowed.
115 const network_v4 me(address_, other.prefix_length_);
116 return other.canonical() == me.canonical();
117}

160network_v4 make_network_v4(const std::string& str,
161 asio::error_code& ec)
162{
163 std::string::size_type pos = str.find_first_of("/");
164
165 if (pos == std::string::npos)
166 {
167 ec = asio::error::invalid_argument;
168 return network_v4();
169 }
170
171 if (pos == str.size() - 1)
172 {
173 ec = asio::error::invalid_argument;
174 return network_v4();
175 }
176
177 std::string::size_type end = str.find_first_not_of("0123456789", pos + 1);
178 if (end != std::string::npos)
179 {
180 ec = asio::error::invalid_argument;
181 return network_v4();
182 }
183
184 const address_v4 addr = make_address_v4(str.substr(0, pos), ec);
185 if (ec)
186 return network_v4();
187
188 const int prefix_len = std::atoi(str.substr(pos + 1).c_str());
189 if (prefix_len < 0 || prefix_len > 32)
190 {
191 ec = asio::error::invalid_argument;
192 return network_v4();
193 }
194
195 return network_v4(addr, static_cast<unsigned short>(prefix_len));
196}
