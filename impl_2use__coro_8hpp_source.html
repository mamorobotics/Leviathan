28template <typename Allocator, typename R>

30{

33 std::allocator_arg_t, Allocator, InitArgs... args)
34 -> experimental::coro<void() noexcept, void,
35 asio::associated_executor_t<Initiation>, Allocator>
36 {

39 }
40
41 template <typename... InitArgs>
42 static auto initiate_impl(asio::detail::initiation_archetype<R()>,
43 std::allocator_arg_t, Allocator, InitArgs... args)

46
