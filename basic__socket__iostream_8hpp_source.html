31// A separate base class is used to ensure that the streambuf is initialised

34class socket_iostream_base
35{
36protected:

46 socket_iostream_base(basic_stream_socket<Protocol> s)
47 : streambuf_(std::move(s))

52 {
53 streambuf_ = std::move(other.streambuf_);
