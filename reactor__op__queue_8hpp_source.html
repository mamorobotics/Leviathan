30template <typename Descriptor>
31class reactor_op_queue
32 : private noncopyable
33{
34public:
35 typedef Descriptor key_type;
36
37 struct mapped_type : op_queue<reactor_op>
38 {
39 mapped_type() {}
40 mapped_type(const mapped_type&) {}
41 void operator=(const mapped_type&) {}
42 };
43
44 typedef typename hash_map<key_type, mapped_type>::value_type value_type;
45 typedef typename hash_map<key_type, mapped_type>::iterator iterator;
46
47 // Constructor.
48 reactor_op_queue()
49 : operations_()
50 {
51 }
52
53 // Obtain iterators to all registered descriptors.
54 iterator begin() { return operations_.begin(); }
55 iterator end() { return operations_.end(); }
56
57 // Add a new operation to the queue. Returns true if this is the only
58 // operation for the given descriptor, in which case the reactor's event
59 // demultiplexing function call may need to be interrupted and restarted.
60 bool enqueue_operation(Descriptor descriptor, reactor_op* op)
61 {
62 std::pair<iterator, bool> entry =
63 operations_.insert(value_type(descriptor, mapped_type()));
64 entry.first->second.push(op);
65 return entry.second;
66 }
67
68 // Cancel all operations associated with the descriptor identified by the
69 // supplied iterator. Any operations pending for the descriptor will be
70 // cancelled. Returns true if any operations were cancelled, in which case
71 // the reactor's event demultiplexing function may need to be interrupted and
72 // restarted.
73 bool cancel_operations(iterator i, op_queue<operation>& ops,
74 const asio::error_code& ec =
75 asio::error::operation_aborted)
76 {
77 if (i != operations_.end())
78 {
79 while (reactor_op* op = i->second.front())
80 {
81 op->ec_ = ec;
82 i->second.pop();
83 ops.push(op);
84 }
85 operations_.erase(i);
86 return true;
87 }
88
89 return false;
90 }
91
92 // Cancel all operations associated with the descriptor. Any operations
93 // pending for the descriptor will be cancelled. Returns true if any
94 // operations were cancelled, in which case the reactor's event
95 // demultiplexing function may need to be interrupted and restarted.
96 bool cancel_operations(Descriptor descriptor, op_queue<operation>& ops,
97 const asio::error_code& ec =
98 asio::error::operation_aborted)
99 {
100 return this->cancel_operations(operations_.find(descriptor), ops, ec);
101 }
102
103 // Cancel operations associated with the descriptor identified by the
104 // supplied iterator, and the specified cancellation key. Any operations
105 // pending for the descriptor with the key will be cancelled. Returns true if
106 // any operations were cancelled, in which case the reactor's event
107 // demultiplexing function may need to be interrupted and restarted.
108 bool cancel_operations_by_key(iterator i, op_queue<operation>& ops,
109 void* cancellation_key, const asio::error_code& ec =
110 asio::error::operation_aborted)
111 {
112 bool result = false;
113 if (i != operations_.end())
114 {
115 op_queue<reactor_op> other_ops;
116 while (reactor_op* op = i->second.front())
117 {
118 i->second.pop();
119 if (op->cancellation_key_ == cancellation_key)
120 {
121 op->ec_ = ec;
122 ops.push(op);
123 result = true;
124 }
125 else
126 other_ops.push(op);
127 }
128 i->second.push(other_ops);
129 if (i->second.empty())
130 operations_.erase(i);
131 }
132 return result;
133 }
134
135 // Cancel all operations associated with the descriptor. Any operations
136 // pending for the descriptor will be cancelled. Returns true if any
137 // operations were cancelled, in which case the reactor's event
138 // demultiplexing function may need to be interrupted and restarted.
139 bool cancel_operations_by_key(Descriptor descriptor, op_queue<operation>& ops,
140 void* cancellation_key, const asio::error_code& ec =
141 asio::error::operation_aborted)
142 {
143 return this->cancel_operations_by_key(
144 operations_.find(descriptor), ops, cancellation_key, ec);
145 }
146
147 // Whether there are no operations in the queue.
148 bool empty() const
149 {
150 return operations_.empty();
151 }
152
153 // Determine whether there are any operations associated with the descriptor.
154 bool has_operation(Descriptor descriptor) const
155 {
156 return operations_.find(descriptor) != operations_.end();
157 }
158
159 // Perform the operations corresponding to the descriptor identified by the
160 // supplied iterator. Returns true if there are still unfinished operations
161 // queued for the descriptor.
162 bool perform_operations(iterator i, op_queue<operation>& ops)
163 {
164 if (i != operations_.end())
165 {
166 while (reactor_op* op = i->second.front())
167 {
168 if (op->perform())
169 {
170 i->second.pop();
171 ops.push(op);
172 }
173 else
174 {
175 return true;
176 }
177 }
178 operations_.erase(i);
179 }
180 return false;
181 }
182
183 // Perform the operations corresponding to the descriptor. Returns true if
184 // there are still unfinished operations queued for the descriptor.
185 bool perform_operations(Descriptor descriptor, op_queue<operation>& ops)
186 {
187 return this->perform_operations(operations_.find(descriptor), ops);
188 }
189
190 // Get all operations owned by the queue.
191 void get_all_operations(op_queue<operation>& ops)
192 {
193 iterator i = operations_.begin();
194 while (i != operations_.end())
195 {
196 iterator op_iter = i++;
197 ops.push(op_iter->second);
198 operations_.erase(op_iter);
199 }
200 }
201
202private:
203 // The operations that are currently executing asynchronously.
204 hash_map<key_type, mapped_type> operations_;
205};
