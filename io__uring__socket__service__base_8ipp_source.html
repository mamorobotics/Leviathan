80void io_uring_socket_service_base::destroy(
81 io_uring_socket_service_base::base_implementation_type& impl)
82{
83 if (impl.socket_ != invalid_socket)
84 {
85 ASIO_HANDLER_OPERATION((io_uring_service_.context(),
86 "socket", &impl, impl.socket_, "close"));
87
88 io_uring_service_.deregister_io_object(impl.io_object_data_);
89 asio::error_code ignored_ec;
90 socket_ops::close(impl.socket_, impl.state_, true, ignored_ec);
91 io_uring_service_.cleanup_io_object(impl.io_object_data_);
92 }
93}

95asio::error_code io_uring_socket_service_base::close(
96 io_uring_socket_service_base::base_implementation_type& impl,
97 asio::error_code& ec)
98{
99 if (is_open(impl))
100 {
101 ASIO_HANDLER_OPERATION((io_uring_service_.context(),
102 "socket", &impl, impl.socket_, "close"));
103
104 io_uring_service_.deregister_io_object(impl.io_object_data_);
105 socket_ops::close(impl.socket_, impl.state_, false, ec);
106 io_uring_service_.cleanup_io_object(impl.io_object_data_);
107 }
108 else
109 {
110 ec = success_ec_;
111 }
112
113 // The descriptor is closed by the OS even if close() returns an error.
114 //
115 // (Actually, POSIX says the state of the descriptor is unspecified. On
116 // Linux the descriptor is apparently closed anyway; e.g. see
117 // http://lkml.org/lkml/2005/9/10/129
118 construct(impl);
119
120 return ec;
121}

123socket_type io_uring_socket_service_base::release(
124 io_uring_socket_service_base::base_implementation_type& impl,
125 asio::error_code& ec)
126{
127 if (!is_open(impl))
128 {
129 ec = asio::error::bad_descriptor;
130 return invalid_socket;
131 }
132
133 ASIO_HANDLER_OPERATION((io_uring_service_.context(),
134 "socket", &impl, impl.socket_, "release"));
135
136 io_uring_service_.deregister_io_object(impl.io_object_data_);
137 io_uring_service_.cleanup_io_object(impl.io_object_data_);
138 socket_type sock = impl.socket_;
139 construct(impl);
140 ec = success_ec_;

162asio::error_code io_uring_socket_service_base::do_open(
163 io_uring_socket_service_base::base_implementation_type& impl,
164 int af, int type, int protocol, asio::error_code& ec)
165{
166 if (is_open(impl))
167 {
168 ec = asio::error::already_open;
169 return ec;
170 }
171
172 socket_holder sock(socket_ops::socket(af, type, protocol, ec));
173 if (sock.get() == invalid_socket)
174 return ec;
175
176 io_uring_service_.register_io_object(impl.io_object_data_);
177
178 impl.socket_ = sock.release();
179 switch (type)
180 {
181 case SOCK_STREAM: impl.state_ = socket_ops::stream_oriented; break;
182 case SOCK_DGRAM: impl.state_ = socket_ops::datagram_oriented; break;
183 default: impl.state_ = 0; break;
184 }
185 ec = success_ec_;
186 return ec;
187}

189asio::error_code io_uring_socket_service_base::do_assign(
190 io_uring_socket_service_base::base_implementation_type& impl, int type,
191 const io_uring_socket_service_base::native_handle_type& native_socket,
192 asio::error_code& ec)
193{
194 if (is_open(impl))
195 {
196 ec = asio::error::already_open;
197 return ec;
198 }
199
200 io_uring_service_.register_io_object(impl.io_object_data_);
201
202 impl.socket_ = native_socket;
203 switch (type)
204 {
205 case SOCK_STREAM: impl.state_ = socket_ops::stream_oriented; break;
206 case SOCK_DGRAM: impl.state_ = socket_ops::datagram_oriented; break;
207 default: impl.state_ = 0; break;
208 }
209 impl.state_ |= socket_ops::possible_dup;
210 ec = success_ec_;
211 return ec;
212}
