51 static status do_perform(reactor_op* base)
52 {
53 ASIO_ASSUME(base != 0);
54 reactive_socket_recvfrom_op_base* o(
55 static_cast<reactive_socket_recvfrom_op_base*>(base));
56
57 typedef buffer_sequence_adapter<asio::mutable_buffer,
58 MutableBufferSequence> bufs_type;
59
60 std::size_t addr_len = o->sender_endpoint_.capacity();
61 status result;
62 if (bufs_type::is_single_buffer)
63 {
64 result = socket_ops::non_blocking_recvfrom1(
65 o->socket_, bufs_type::first(o->buffers_).data(),
66 bufs_type::first(o->buffers_).size(), o->flags_,
67 o->sender_endpoint_.data(), &addr_len,
68 o->ec_, o->bytes_transferred_) ? done : not_done;
69 }
70 else
71 {
72 bufs_type bufs(o->buffers_);
73 result = socket_ops::non_blocking_recvfrom(o->socket_,
74 bufs.buffers(), bufs.count(), o->flags_,
75 o->sender_endpoint_.data(), &addr_len,
76 o->ec_, o->bytes_transferred_) ? done : not_done;
77 }
78
79 if (result && !o->ec_)
80 o->sender_endpoint_.resize(addr_len);
81
82 ASIO_HANDLER_REACTOR_OPERATION((*o, "non_blocking_recvfrom",
83 o->ec_, o->bytes_transferred_));
84
85 return result;
86 }
87
88private:
89 socket_type socket_;
90 int protocol_type_;
91 MutableBufferSequence buffers_;
92 Endpoint& sender_endpoint_;
93 socket_base::message_flags flags_;
94};

120 static void do_complete(void* owner, operation* base,
121 const asio::error_code& /*ec*/,
122 std::size_t /*bytes_transferred*/)
123 {
124 // Take ownership of the handler object.
125 ASIO_ASSUME(base != 0);
126 reactive_socket_recvfrom_op* o(
127 static_cast<reactive_socket_recvfrom_op*>(base));
128 ptr p = { asio::detail::addressof(o->handler_), o, o };
129
130 ASIO_HANDLER_COMPLETION((*o));
131
132 // Take ownership of the operation's outstanding work.
133 handler_work<Handler, IoExecutor> w(
134 static_cast<handler_work<Handler, IoExecutor>&&>(
135 o->work_));
136
137 ASIO_ERROR_LOCATION(o->ec_);
138
139 // Make a copy of the handler so that the memory can be deallocated before
140 // the upcall is made. Even if we're not about to make an upcall, a
141 // sub-object of the handler may be the true owner of the memory associated
142 // with the handler. Consequently, a local copy of the handler is required
143 // to ensure that any owning sub-object remains valid until after we have
144 // deallocated the memory here.
145 detail::binder2<Handler, asio::error_code, std::size_t>
146 handler(o->handler_, o->ec_, o->bytes_transferred_);
147 p.h = asio::detail::addressof(handler.handler_);
148 p.reset();
149
150 // Make the upcall if required.
151 if (owner)
152 {
153 fenced_block b(fenced_block::half);
154 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
155 w.complete(handler, handler.handler_);
156 ASIO_HANDLER_INVOCATION_END;
157 }
158 }
