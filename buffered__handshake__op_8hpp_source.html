29class buffered_handshake_op
30{
31public:

37 buffered_handshake_op(stream_base::handshake_type type,
38 const ConstBufferSequence& buffers)
39 : type_(type),
40 buffers_(buffers),
41 total_buffer_size_(asio::buffer_size(buffers_))
42 {
43 }

46 asio::error_code& ec,

54 template <typename Handler>
55 void call_handler(Handler& handler,
56 const asio::error_code& ec,
57 const std::size_t& bytes_transferred) const
58 {
59 static_cast<Handler&&>(handler)(ec, bytes_transferred);
60 }
61
62private:
63 template <typename Iterator>
64 engine::want process(engine& eng,
65 asio::error_code& ec,
66 std::size_t& bytes_transferred,
67 Iterator begin, Iterator end) const
68 {
69 Iterator iter = begin;
70 std::size_t accumulated_size = 0;
71
72 for (;;)
73 {
74 engine::want want = eng.handshake(type_, ec);
75 if (want != engine::want_input_and_retry
76 || bytes_transferred == total_buffer_size_)
77 return want;
78
79 // Find the next buffer piece to be fed to the engine.
80 while (iter != end)
81 {
82 const_buffer buffer(*iter);
83
84 // Skip over any buffers which have already been consumed by the engine.
85 if (bytes_transferred >= accumulated_size + buffer.size())
86 {
87 accumulated_size += buffer.size();
88 ++iter;
89 continue;
90 }
91
92 // The current buffer may have been partially consumed by the engine on
93 // a previous iteration. If so, adjust the buffer to point to the
94 // unused portion.
95 if (bytes_transferred > accumulated_size)
96 buffer = buffer + (bytes_transferred - accumulated_size);
97
98 // Pass the buffer to the engine, and update the bytes transferred to
99 // reflect the total number of bytes consumed so far.
100 bytes_transferred += buffer.size();
101 buffer = eng.put_input(buffer);
102 bytes_transferred -= buffer.size();
103 break;
104 }
105 }
106 }
107
108 stream_base::handshake_type type_;
109 ConstBufferSequence buffers_;
110 std::size_t total_buffer_size_;
111};
