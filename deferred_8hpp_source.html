43// Helper trait for getting the completion signatures of the tail in a sequence
44// when invoked with the specified arguments.

46template <typename Tail, typename... Signatures>

49template <typename Tail, typename R, typename... Args, typename... Signatures>
50struct deferred_sequence_signatures<Tail, R(Args...), Signatures...>
51 : completion_signature_of<decltype(declval<Tail>()(declval<Args>()...))>
52{
53 static_assert(
54 !is_same<decltype(declval<Tail>()(declval<Args>()...)), void>::value,
55 "deferred functions must produce a deferred return type");

58// Completion handler for the head component of a deferred sequence.
59template <typename Handler, typename Tail>
60class deferred_sequence_handler
61{
62public:
63 template <typename H, typename T>
64 explicit deferred_sequence_handler(H&& handler, T&& tail)
65 : handler_(static_cast<H&&>(handler)),

68 }
69

71 void operator()(Args&&... args)
72 {

76 }

83template <typename Head, typename Tail, typename... Signatures>
84class deferred_sequence_base
85{
86private:
87 struct initiate
88 {
89 template <typename Handler>
90 void operator()(Handler&& handler, Head head, Tail&& tail)
91 {
92 static_cast<Head&&>(head)(
93 deferred_sequence_handler<decay_t<Handler>, decay_t<Tail>>(
94 static_cast<Handler&&>(handler), static_cast<Tail&&>(tail)));
95 }
96 };
97
98 Head head_;
99 Tail tail_;
100
101public:
102 template <typename H, typename T>
103 constexpr explicit deferred_sequence_base(H&& head, T&& tail)
104 : head_(static_cast<H&&>(head)),
105 tail_(static_cast<T&&>(tail))
106 {
107 }
108
109 template <ASIO_COMPLETION_TOKEN_FOR(Signatures...) CompletionToken>
110 auto operator()(CompletionToken&& token) &&
111 -> decltype(
112 async_initiate<CompletionToken, Signatures...>(
113 initiate(), token, static_cast<Head&&>(this->head_),
114 static_cast<Tail&&>(this->tail_)))
115 {
116 return async_initiate<CompletionToken, Signatures...>(initiate(),
117 token, static_cast<Head&&>(head_), static_cast<Tail&&>(tail_));
118 }
119
120 template <ASIO_COMPLETION_TOKEN_FOR(Signatures...) CompletionToken>
121 auto operator()(CompletionToken&& token) const &
122 -> decltype(
123 async_initiate<CompletionToken, Signatures...>(
124 initiate(), token, this->head_, this->tail_))
125 {
126 return async_initiate<CompletionToken, Signatures...>(
127 initiate(), token, head_, tail_);
128 }
129};

131// Two-step application of variadic Signatures to determine correct base type.

133template <typename Head, typename Tail>
134struct deferred_sequence_types
135{
136 template <typename... Signatures>
137 struct op1
138 {
139 typedef deferred_sequence_base<Head, Tail, Signatures...> type;
140 };
141
142 template <typename... Signatures>
143 struct op2
144 {
145 typedef typename deferred_sequence_signatures<Tail, Signatures...>::template
146 apply<op1>::type::type type;
147 };
148
149 typedef typename completion_signature_of<Head>::template
150 apply<op2>::type::type base;
151};
