104// An executor that adapts the system_executor to capture any exeption thrown
105// by a submitted function object and save it into a promise.
106template <typename T, typename Blocking = execution::blocking_t::possibly_t>
107class promise_executor
108{
109public:
110 explicit promise_executor(const shared_ptr<std::promise<T>>& p)
111 : p_(p)
112 {
113 }
114
115 execution_context& query(execution::context_t) const noexcept
116 {
117 return asio::query(system_executor(), execution::context);
118 }
119
120 static constexpr Blocking query(execution::blocking_t)
121 {
122 return Blocking();
123 }
124
125 promise_executor<T, execution::blocking_t::possibly_t>
126 require(execution::blocking_t::possibly_t) const
127 {
128 return promise_executor<T, execution::blocking_t::possibly_t>(p_);
129 }
130
131 promise_executor<T, execution::blocking_t::never_t>
132 require(execution::blocking_t::never_t) const
133 {
134 return promise_executor<T, execution::blocking_t::never_t>(p_);
135 }
136
137 template <typename F>
138 void execute(F&& f) const
139 {
140 asio::require(system_executor(), Blocking()).execute(
141 promise_invoker<T, F>(p_, static_cast<F&&>(f)));
142 }
143
144#if !defined(ASIO_NO_TS_EXECUTORS)
145 execution_context& context() const noexcept
146 {
147 return system_executor().context();
148 }
149
150 void on_work_started() const noexcept {}
151 void on_work_finished() const noexcept {}
152
153 template <typename F, typename A>
154 void dispatch(F&& f, const A&) const
155 {
156 promise_invoker<T, F>(p_, static_cast<F&&>(f))();
157 }
158
159 template <typename F, typename A>
160 void post(F&& f, const A& a) const
161 {
162 system_executor().post(
163 promise_invoker<T, F>(p_, static_cast<F&&>(f)), a);
164 }
165
166 template <typename F, typename A>
167 void defer(F&& f, const A& a) const
168 {
169 system_executor().defer(
170 promise_invoker<T, F>(p_, static_cast<F&&>(f)), a);
171 }
172#endif // !defined(ASIO_NO_TS_EXECUTORS)
173
174 friend bool operator==(const promise_executor& a,
175 const promise_executor& b) noexcept
176 {
177 return a.p_ == b.p_;
178 }
179
180 friend bool operator!=(const promise_executor& a,
181 const promise_executor& b) noexcept
182 {
183 return a.p_ != b.p_;
184 }
185
186private:
187 shared_ptr<std::promise<T>> p_;
188};

190// The base class for all completion handlers that create promises.
191template <typename T>
192class promise_creator
193{
194public:
195 typedef promise_executor<T> executor_type;
196
197 executor_type get_executor() const noexcept
198 {
199 return executor_type(p_);
200 }
201
202 typedef std::future<T> future_type;
203
204 future_type get_future()
205 {
206 return p_->get_future();
207 }
208
209protected:
210 template <typename Allocator>
211 void create_promise(const Allocator& a)
212 {
213 ASIO_REBIND_ALLOC(Allocator, char) b(a);
214 p_ = std::allocate_shared<std::promise<T>>(b, std::allocator_arg, b);
215 }
216
217 shared_ptr<std::promise<T>> p_;
218};
