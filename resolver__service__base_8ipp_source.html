57void resolver_service_base::base_shutdown()
58{
59 if (work_scheduler_.get())
60 {
61 work_scheduler_->work_finished();
62 work_scheduler_->stop();
63 if (work_thread_.get())
64 {
65 work_thread_->join();
66 work_thread_.reset();
67 }
68 work_scheduler_.reset();
69 }
70}

72void resolver_service_base::base_notify_fork(
73 execution_context::fork_event fork_ev)
74{
75 if (work_thread_.get())
76 {
77 if (fork_ev == execution_context::fork_prepare)
78 {
79 work_scheduler_->stop();
80 work_thread_->join();
81 work_thread_.reset();
82 }
83 }
84 else if (fork_ev != execution_context::fork_prepare)
85 {
86 work_scheduler_->restart();
87 }
88}

127void resolver_service_base::start_resolve_op(resolve_op* op)
128{
129 if (ASIO_CONCURRENCY_HINT_IS_LOCKING(SCHEDULER,
130 scheduler_.concurrency_hint()))
131 {
132 start_work_thread();
133 scheduler_.work_started();
134 work_scheduler_->post_immediate_completion(op, false);
135 }
136 else
137 {
138 op->ec_ = asio::error::operation_not_supported;
139 scheduler_.post_immediate_completion(op, false);
140 }
141}

143void resolver_service_base::start_work_thread()
144{
145 asio::detail::mutex::scoped_lock lock(mutex_);
146 if (!work_thread_.get())
147 {
148 work_thread_.reset(new asio::detail::thread(
149 work_scheduler_runner(*work_scheduler_)));
150 }
151}
