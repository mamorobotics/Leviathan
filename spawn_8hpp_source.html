38// Base class for all spawn()-ed thread implementations.
39class spawned_thread_base
40{
41public:
42 spawned_thread_base()
43 : owner_(0),
44 has_context_switched_(false),
45 throw_if_cancelled_(false),
46 terminal_(false)
47 {
48 }
49
50 virtual ~spawned_thread_base() {}
51 virtual void resume() = 0;
52 virtual void suspend_with(void (*fn)(void*), void* arg) = 0;
53 virtual void destroy() = 0;
54
55 void attach(spawned_thread_base** owner)
56 {
57 owner_ = owner;
58 *owner_ = this;
59 }
60
61 void detach()
62 {
63 if (owner_)
64 *owner_ = 0;
65 owner_ = 0;
66 }

68 void suspend()

70 suspend_with(0, 0);
71 }
