27template <typename Executor, typename T>
28class awaitable_handler_base
29 : public awaitable_thread<Executor>
30{
31public:
32 typedef void result_type;
33 typedef awaitable<T, Executor> awaitable_type;
34
35 // Construct from the entry point of a new thread of execution.
36 awaitable_handler_base(awaitable<awaitable_thread_entry_point, Executor> a,
37 const Executor& ex, cancellation_slot pcs, cancellation_state cs)
38 : awaitable_thread<Executor>(std::move(a), ex, pcs, cs)
39 {
40 }
41
42 // Transfer ownership from another awaitable_thread.
43 explicit awaitable_handler_base(awaitable_thread<Executor>* h)
44 : awaitable_thread<Executor>(std::move(*h))
45 {
46 }
47
48protected:
49 awaitable_frame<T, Executor>* frame() noexcept
50 {
51 return static_cast<awaitable_frame<T, Executor>*>(
52 this->entry_point()->top_of_stack_);
53 }
54};

256template <typename Executor, typename R, typename... Args>
257class async_result<use_awaitable_t<Executor>, R(Args...)>
258{
259public:
260 typedef typename detail::awaitable_handler<
261 Executor, decay_t<Args>...> handler_type;
262 typedef typename handler_type::awaitable_type return_type;
263
264 template <typename Initiation, typename... InitArgs>
265#if defined(__APPLE_CC__) && (__clang_major__ == 13)
266 __attribute__((noinline))
267#endif // defined(__APPLE_CC__) && (__clang_major__ == 13)
268 static handler_type* do_init(
269 detail::awaitable_frame_base<Executor>* frame, Initiation& initiation,
270 use_awaitable_t<Executor> u, InitArgs&... args)
271 {
272 (void)u;
273 ASIO_HANDLER_LOCATION((u.file_name_, u.line_, u.function_name_));
274 handler_type handler(frame->detach_thread());
275 std::move(initiation)(std::move(handler), std::move(args)...);
276 return nullptr;
277 }
278
279 template <typename Initiation, typename... InitArgs>
280 static return_type initiate(Initiation initiation,
281 use_awaitable_t<Executor> u, InitArgs... args)
282 {
283 co_await [&] (auto* frame)
284 {
285 return do_init(frame, initiation, u, args...);
286 };
287
288 for (;;) {} // Never reached.
289#if defined(_MSC_VER)
290 co_return dummy_return<typename return_type::value_type>();
291#endif // defined(_MSC_VER)
292 }
293};
