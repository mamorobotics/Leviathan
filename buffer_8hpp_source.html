473// Helper trait to detect single buffers.
474template <typename BufferSequence>
475struct buffer_sequence_cardinality :
476 conditional_t<
477 is_same<BufferSequence, mutable_buffer>::value
478#if !defined(ASIO_NO_DEPRECATED)
479 || is_same<BufferSequence, mutable_buffers_1>::value
480 || is_same<BufferSequence, const_buffers_1>::value
481#endif // !defined(ASIO_NO_DEPRECATED)
482 || is_same<BufferSequence, const_buffer>::value,
483 one_buffer, multiple_buffers> {};
484
485template <typename Iterator>
486inline std::size_t buffer_size(one_buffer,
487 Iterator begin, Iterator) noexcept
488{
489 return const_buffer(*begin).size();
490}
491
492template <typename Iterator>
493inline std::size_t buffer_size(multiple_buffers,
494 Iterator begin, Iterator end) noexcept
495{
496 std::size_t total_buffer_size = 0;
497
498 Iterator iter = begin;
499 for (; iter != end; ++iter)
500 {
501 const_buffer b(*iter);
502 total_buffer_size += b.size();
503 }
504
505 return total_buffer_size;
506}
