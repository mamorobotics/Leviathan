53// Spawned thread implementation using Boost.Coroutine.
54class spawned_coroutine_thread : public spawned_thread_base
55{
56public:
57#if defined(BOOST_COROUTINES_UNIDIRECT) || defined(BOOST_COROUTINES_V2)
58 typedef boost::coroutines::pull_coroutine<void> callee_type;
59 typedef boost::coroutines::push_coroutine<void> caller_type;
60#else
61 typedef boost::coroutines::coroutine<void()> callee_type;
62 typedef boost::coroutines::coroutine<void()> caller_type;
63#endif
64
65 spawned_coroutine_thread(caller_type& caller)
66 : caller_(caller),
67 on_suspend_fn_(0),
68 on_suspend_arg_(0)
69 {
70 }
71
72 template <typename F>
73 static spawned_thread_base* spawn(F&& f,
74 const boost::coroutines::attributes& attributes,
75 cancellation_slot parent_cancel_slot = cancellation_slot(),
76 cancellation_state cancel_state = cancellation_state())
77 {
78 spawned_coroutine_thread* spawned_thread = 0;
79 callee_type callee(entry_point<decay_t<F>>(
80 static_cast<F&&>(f), &spawned_thread), attributes);
81 spawned_thread->callee_.swap(callee);
82 spawned_thread->parent_cancellation_slot_ = parent_cancel_slot;
83 spawned_thread->cancellation_state_ = cancel_state;
84 return spawned_thread;
85 }
86
87 template <typename F>
88 static spawned_thread_base* spawn(F&& f,
89 cancellation_slot parent_cancel_slot = cancellation_slot(),
90 cancellation_state cancel_state = cancellation_state())
91 {
92 return spawn(static_cast<F&&>(f), boost::coroutines::attributes(),
93 parent_cancel_slot, cancel_state);
94 }
95
96 void resume()
97 {
98 callee_();
99 if (on_suspend_fn_)
100 {
101 void (*fn)(void*) = on_suspend_fn_;
102 void* arg = on_suspend_arg_;
103 on_suspend_fn_ = 0;
104 fn(arg);
105 }
106 }
107
108 void suspend_with(void (*fn)(void*), void* arg)
109 {
110 if (throw_if_cancelled_)
111 if (!!cancellation_state_.cancelled())
112 throw_error(asio::error::operation_aborted, "yield");
113 has_context_switched_ = true;
114 on_suspend_fn_ = fn;
115 on_suspend_arg_ = arg;
116 caller_();
117 }
118
119 void destroy()
120 {
121 callee_type callee;
122 callee.swap(callee_);
123 if (terminal_)
124 callee();
125 }
126
127private:
128 template <typename Function>
129 class entry_point
130 {
131 public:
132 template <typename F>
133 entry_point(F&& f,
134 spawned_coroutine_thread** spawned_thread_out)
135 : function_(static_cast<F&&>(f)),
136 spawned_thread_out_(spawned_thread_out)
137 {
138 }
139
140 void operator()(caller_type& caller)
141 {
142 Function function(static_cast<Function&&>(function_));
143 spawned_coroutine_thread spawned_thread(caller);
144 *spawned_thread_out_ = &spawned_thread;
145 spawned_thread_out_ = 0;
146 spawned_thread.suspend();
147#if !defined(ASIO_NO_EXCEPTIONS)
148 try
149#endif // !defined(ASIO_NO_EXCEPTIONS)
150 {
151 function(&spawned_thread);
152 spawned_thread.terminal_ = true;
153 spawned_thread.suspend();
154 }
155#if !defined(ASIO_NO_EXCEPTIONS)
156 catch (const boost::coroutines::detail::forced_unwind&)
157 {
158 throw;
159 }
160 catch (...)
161 {
162 exception_ptr ex = current_exception();
163 spawned_thread.terminal_ = true;
164 spawned_thread.suspend_with(spawned_thread_rethrow, &ex);
165 }
166#endif // !defined(ASIO_NO_EXCEPTIONS)
167 }
168
169 private:
170 Function function_;
171 spawned_coroutine_thread** spawned_thread_out_;
172 };
173
174 caller_type& caller_;
175 callee_type callee_;
176 void (*on_suspend_fn_)(void*);
177 void* on_suspend_arg_;
178};

386// Base class for all completion handlers associated with a spawned thread.
387template <typename Executor>
388class spawn_handler_base
389{
390public:
391 typedef Executor executor_type;
392 typedef cancellation_slot cancellation_slot_type;
393
394 spawn_handler_base(const basic_yield_context<Executor>& yield)
395 : yield_(yield),
396 spawned_thread_(yield.spawned_thread_)
397 {
398 spawned_thread_->detach();
399 }
400
401 spawn_handler_base(spawn_handler_base&& other) noexcept
402 : yield_(other.yield_),
403 spawned_thread_(other.spawned_thread_)
404
405 {
406 other.spawned_thread_ = 0;
407 }
408
409 ~spawn_handler_base()
410 {
411 if (spawned_thread_)
412 (post)(yield_.executor_, spawned_thread_destroyer(spawned_thread_));
413 }
414
415 executor_type get_executor() const noexcept
416 {
417 return yield_.executor_;
418 }
419
420 cancellation_slot_type get_cancellation_slot() const noexcept
421 {
422 return spawned_thread_->get_cancellation_slot();
423 }
424
425 void resume()
426 {
427 spawned_thread_resumer resumer(spawned_thread_);
428 spawned_thread_ = 0;
429 resumer();
430 }
431
432protected:
433 const basic_yield_context<Executor>& yield_;
434 spawned_thread_base* spawned_thread_;
435};

765template <typename Executor, typename Signature>
766class async_result<basic_yield_context<Executor>, Signature>
767{

770 typedef typename handler_type::return_type return_type;

772#if defined(ASIO_HAS_VARIADIC_LAMBDA_CAPTURES)

774 template <typename Initiation, typename... InitArgs>

802
803 template <std::size_t... I>
804 void do_invoke(detail::index_sequence<I...>)
805 {
806 static_cast<Initiation&&>(init_)(
807 handler_type(yield_, result_),

826 suspend_with_helper<Initiation, InitArgs...>{
827 result, static_cast<Initiation&&>(init), yield,

829 static_cast<InitArgs&&>(init_args)...)});
