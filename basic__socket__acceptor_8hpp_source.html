231 basic_socket_acceptor(const executor_type& ex,
232 const endpoint_type& endpoint, bool reuse_addr = true)
233 : impl_(0, ex)
234 {
235 asio::error_code ec;
236 const protocol_type protocol = endpoint.protocol();
237 impl_.get_service().open(impl_.get_implementation(), protocol, ec);
238 asio::detail::throw_error(ec, "open");
239 if (reuse_addr)
240 {
241 impl_.get_service().set_option(impl_.get_implementation(),
242 socket_base::reuse_address(true), ec);
243 asio::detail::throw_error(ec, "set_option");
244 }
245 impl_.get_service().bind(impl_.get_implementation(), endpoint, ec);
246 asio::detail::throw_error(ec, "bind");
247 impl_.get_service().listen(impl_.get_implementation(),
248 socket_base::max_listen_connections, ec);
249 asio::detail::throw_error(ec, "listen");
250 }
