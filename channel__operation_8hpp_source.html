37// Base class for all channel operations. A function pointer is used instead of
38// virtual functions to avoid the associated overhead.
39class channel_operation ASIO_INHERIT_TRACKED_HANDLER
40{
41public:
42 template <typename Executor, typename = void, typename = void>
43 class handler_work_base;
44
45 template <typename Handler, typename IoExecutor, typename = void>
46 class handler_work;
47
48 void destroy()
49 {
50 func_(this, destroy_op, 0);
51 }
52
53protected:
54 enum action
55 {
56 destroy_op = 0,
57 immediate_op = 1,
58 post_op = 2,
59 dispatch_op = 3,
60 cancel_op = 4,
61 close_op = 5
62 };
63
64 typedef void (*func_type)(channel_operation*, action, void*);
65
66 channel_operation(func_type func)
67 : next_(0),
68 func_(func),
69 cancellation_key_(0)
70 {
71 }
72
73 // Prevents deletion through this type.
74 ~channel_operation()
75 {
76 }
77
78 friend class asio::detail::op_queue_access;
79 channel_operation* next_;
80 func_type func_;
81
82public:
83 // The operation key used for targeted cancellation.
84 void* cancellation_key_;
85};

107 template <typename IoExecutor, typename Function, typename Handler>
108 void post(const IoExecutor& io_exec, Function& function, Handler&)

116 {
