168 template <typename Y, typename R, typename E, typename A>
169 auto await_suspend(coroutine_handle<coro_promise<Y, R, E, A>> h)
170 -> coroutine_handle<>
171 {
172 auto& hp = h.promise();
173
174 if constexpr (!coro_promise<Y, R, E, A>::is_noexcept)
175 {
176 if ((hp.cancel->state.cancelled() != cancellation_type::none)
177 && hp.cancel->throw_if_cancelled_)
178 {
179 asio::detail::throw_error(
180 asio::error::operation_aborted, "coro-cancelled");
181 }
182 }
183
184 if (hp.get_executor() == coro.get_executor())
185 {
186 coro.coro_->awaited_from = h;
187 coro.coro_->reset_error();
188 coro.coro_->input_ = std::move(value);
189 coro.coro_->cancel = hp.cancel;
190 return coro.coro_->get_handle();
191 }
192 else
193 {
194 coro.coro_->awaited_from =
195 dispatch_coroutine(
196 asio::prefer(hp.get_executor(),
197 execution::outstanding_work.tracked),
198 [h]() mutable { h.resume(); }).handle;
199
200 coro.coro_->reset_error();
201 coro.coro_->input_ = std::move(value);
202
203 struct cancel_handler
204 {
205 using src = std::pair<cancellation_signal,
206 detail::coro_cancellation_source>;
207
208 std::shared_ptr<src> st = std::make_shared<src>();
209
210 cancel_handler(E e, coro_t& coro) : e(e), coro_(coro.coro_)
211 {
212 st->second.state =
213 cancellation_state(st->second.slot = st->first.slot());
214 }
215
216 E e;
217 typename coro_t::promise_type* coro_;
218
219 void operator()(cancellation_type ct)
220 {
221 asio::dispatch(e, [ct, st = st]() mutable
222 {
223 auto & [sig, state] = *st;
224 sig.emit(ct);
225 });
226 }
227 };
228
229 if (hp.cancel->state.slot().is_connected())
230 {
231 hp.cancel->state.slot().template emplace<cancel_handler>(
232 coro.get_executor(), coro);
233 }
234
235 auto hh = detail::coroutine_handle<
236 typename coro_t::promise_type>::from_promise(*coro.coro_);
237
238 return dispatch_coroutine(
239 coro.coro_->get_executor(), [hh]() mutable { hh.resume(); }).handle;
240 }
241 }
242
243 auto await_resume() -> typename coro_t::result_type
244 {
245 coro.coro_->cancel = nullptr;
246 coro.coro_->rethrow_if();
247 return std::move(coro.coro_->result_);
248 }
249 };
250
251 template <typename CompletionToken>
252 auto async_resume(CompletionToken&& token) &&
253 {
254 return coro.async_resume(std::move(value),
255 std::forward<CompletionToken>(token));
256 }
257
258 auto operator co_await() &&
259 {
260 return awaitable_t{std::move(value), coro};
261 }

958 detail::coroutine_handle<detail::coro_promise<Y, R, E, A>> h)
959 -> detail::coroutine_handle<>
960 {
961 auto& hp = h.promise();
962
963 if constexpr (!detail::coro_promise<Y, R, E, A>::is_noexcept)
964 {
965 if ((hp.cancel->state.cancelled() != cancellation_type::none)
966 && hp.cancel->throw_if_cancelled_)
967 {
968 asio::detail::throw_error(
969 asio::error::operation_aborted, "coro-cancelled");
970 }
971 }
972
973 if (hp.get_executor() == coro_.get_executor())
974 {
975 coro_.coro_->awaited_from = h;
976 coro_.coro_->cancel = hp.cancel;
977 coro_.coro_->reset_error();
978
979 return coro_.coro_->get_handle();
980 }
981 else
982 {
983 coro_.coro_->awaited_from = detail::dispatch_coroutine(
984 asio::prefer(hp.get_executor(),
985 execution::outstanding_work.tracked),
986 [h]() mutable
987 {
988 h.resume();
989 }).handle;
990
991 coro_.coro_->reset_error();
992
993 struct cancel_handler
994 {
995 std::shared_ptr<std::pair<cancellation_signal,
996 detail::coro_cancellation_source>> st = std::make_shared<
997 std::pair<cancellation_signal, detail::coro_cancellation_source>>();
998
999 cancel_handler(E e, coro& coro) : e(e), coro_(coro.coro_)
1000 {
1001 st->second.state = cancellation_state(
1002 st->second.slot = st->first.slot());
1003 }
1004
1005 E e;
1006 typename coro::promise_type* coro_;
1007
1008 void operator()(cancellation_type ct)
1009 {
1010 asio::dispatch(e,
1011 [ct, st = st]() mutable
1012 {
1013 auto & [sig, state] = *st;
1014 sig.emit(ct);
1015 });
1016 }
1017 };
1018
1019 if (hp.cancel->state.slot().is_connected())
1020 {
1021 hp.cancel->state.slot().template emplace<cancel_handler>(
1022 coro_.get_executor(), coro_);
1023 }
1024
1025 auto hh = detail::coroutine_handle<
1026 detail::coro_promise<Yield, Return, Executor, Allocator>>::from_promise(
1027 *coro_.coro_);
1028
1029 return detail::dispatch_coroutine(
1030 coro_.coro_->get_executor(),
1031 [hh]() mutable { hh.resume(); }).handle;
1032 }
1033 }
1034
1035 auto await_resume() -> result_type
1036 {
1037 coro_.coro_->cancel = nullptr;
1038 coro_.coro_->rethrow_if();
1039 if constexpr (!std::is_void_v<result_type>)
1040 return std::move(coro_.coro_->result_);
1041 }
