152 typename implementation_type<Traits,
153 Signatures...>::payload_type&& payload,
154 Handler& handler, const IoExecutor& io_ex)
155 {
156 associated_cancellation_slot_t<Handler> slot
157 = asio::get_associated_cancellation_slot(handler);
158
159 // Allocate and construct an operation to wrap the handler.
160 typedef channel_send_op<
161 typename implementation_type<Traits, Signatures...>::payload_type,
162 Handler, IoExecutor> op;
163 typename op::ptr p = { asio::detail::addressof(handler),
164 op::ptr::allocate(handler), 0 };
165 p.p = new (p.v) op(static_cast<typename implementation_type<
166 Traits, Signatures...>::payload_type&&>(payload), handler, io_ex);
167
168 // Optionally register for per-operation cancellation.
169 if (slot.is_connected())
170 {
171 p.p->cancellation_key_ =
172 &slot.template emplace<op_cancellation<Traits, Signatures...>>(
173 this, &impl);
174 }
175
176 ASIO_HANDLER_CREATION((this->context(), *p.p,
177 "channel", &impl, 0, "async_send"));
178
179 start_send_op(impl, p.p);
180 p.v = p.p = 0;
181 }

291// The implementation for a specific value type.
292template <typename Mutex>
293template <typename Traits, typename... Signatures>
294struct channel_service<Mutex>::implementation_type : base_implementation_type
295{
296 // The traits type associated with the channel.
297 typedef typename Traits::template rebind<Signatures...>::other traits_type;
298
299 // Type of an element stored in the buffer.
300 typedef conditional_t<
301 has_signature<
302 typename traits_type::receive_cancelled_signature,
303 Signatures...
304 >::value,
305 conditional_t<
306 has_signature<
307 typename traits_type::receive_closed_signature,
308 Signatures...
309 >::value,
310 channel_payload<Signatures...>,
311 channel_payload<
312 Signatures...,
313 typename traits_type::receive_closed_signature
314 >
315 >,
316 conditional_t<
317 has_signature<
318 typename traits_type::receive_closed_signature,
319 Signatures...,
320 typename traits_type::receive_cancelled_signature
321 >::value,
322 channel_payload<
323 Signatures...,
324 typename traits_type::receive_cancelled_signature
325 >,
326 channel_payload<
327 Signatures...,
328 typename traits_type::receive_cancelled_signature,
329 typename traits_type::receive_closed_signature
330 >
331 >
332 > payload_type;
333
334 // Move from another buffer.
335 void buffer_move_from(implementation_type& other)
336 {
337 buffer_ = static_cast<
338 typename traits_type::template container<payload_type>::type&&>(
339 other.buffer_);
340 other.buffer_clear();
341 }
342
343 // Get number of buffered elements.
344 std::size_t buffer_size() const
345 {
346 return buffer_.size();
347 }
348
349 // Push a new value to the back of the buffer.
350 void buffer_push(payload_type payload)
351 {
352 buffer_.push_back(static_cast<payload_type&&>(payload));
353 }
354
355 // Push new values to the back of the buffer.
356 std::size_t buffer_push_n(std::size_t count, payload_type payload)
357 {
358 std::size_t i = 0;
359 for (; i < count && buffer_.size() < this->max_buffer_size_; ++i)
360 buffer_.push_back(payload);
361 return i;
362 }
363
364 // Get the element at the front of the buffer.
365 payload_type buffer_front()
366 {
367 return static_cast<payload_type&&>(buffer_.front());
368 }
369
370 // Pop a value from the front of the buffer.
371 void buffer_pop()
372 {
373 buffer_.pop_front();
374 }
375
376 // Clear all buffered values.
377 void buffer_clear()
378 {
379 buffer_.clear();
380 }
381
382private:
383 // Buffered values.
384 typename traits_type::template container<payload_type>::type buffer_;
385};

529 typename traits_type::receive_closed_signature
530 >
531 >
532 > payload_type;
533
534 // Construct with empty buffer.
535 implementation_type()
536 : size_(0)
537 {
538 first_.count_ = 0;
539 }
540
541 // Move from another buffer.
542 void buffer_move_from(implementation_type& other)
543 {
544 size_ = other.buffer_;
545 other.size_ = 0;
546 first_ = other.first_;
547 other.first.count_ = 0;
548 rest_ = static_cast<
549 typename traits_type::template container<buffered_value>::type&&>(
550 other.rest_);
551 other.buffer_clear();
552 }
553
554 // Get number of buffered elements.
555 std::size_t buffer_size() const
556 {
557 return size_;
558 }
559
560 // Push a new value to the back of the buffer.
561 void buffer_push(payload_type payload)
562 {
563 buffered_value& last = rest_.empty() ? first_ : rest_.back();
564 if (last.count_ == 0)
565 {
566 value_handler handler{last.value_};
567 payload.receive(handler);
568 last.count_ = 1;
569 }
570 else
571 {
572 asio::error_code value{last.value_};
573 value_handler handler{value};
574 payload.receive(handler);
575 if (last.value_ == value)
576 ++last.count_;
577 else
578 rest_.push_back({value, 1});
579 }
580 ++size_;
581 }
582
583 // Push new values to the back of the buffer.
584 std::size_t buffer_push_n(std::size_t count, payload_type payload)
585 {
586 std::size_t available = this->max_buffer_size_ - size_;
587 count = (count < available) ? count : available;
588 if (count > 0)
589 {
590 buffered_value& last = rest_.empty() ? first_ : rest_.back();
591 if (last.count_ == 0)
592 {
593 payload.receive(value_handler{last.value_});
594 last.count_ = count;
595 }
596 else
597 {
598 asio::error_code value{last.value_};
599 payload.receive(value_handler{value});
600 if (last.value_ == value)
601 last.count_ += count;
602 else
603 rest_.push_back({value, count});
604 }
605 size_ += count;
606 }
607 return count;
608 }
609
610 // Get the element at the front of the buffer.
611 payload_type buffer_front()
612 {
613 return payload_type({0, first_.value_});
614 }
615
616 // Pop a value from the front of the buffer.
617 void buffer_pop()
618 {
619 --size_;
620 if (--first_.count_ == 0 && !rest_.empty())
621 {
622 first_ = rest_.front();
623 rest_.pop_front();
624 }
625 }
626
627 // Clear all values from the buffer.
628 void buffer_clear()
629 {
630 size_ = 0;
631 first_.count_ == 0;
632 rest_.clear();
633 }
634
635private:
636 struct buffered_value
637 {
638 asio::error_code value_;
639 std::size_t count_;
640 };
641
642 struct value_handler
643 {
644 asio::error_code& target_;
645
646 template <typename... Args>
647 void operator()(const asio::error_code& value, Args&&...)
648 {
649 target_ = value;
650 }
651 };
652
653 buffered_value& last_value()
654 {
655 return rest_.empty() ? first_ : rest_.back();
656 }
657
658 // Total number of buffered values.
659 std::size_t size_;
660
661 // The first buffered value is maintained as a separate data member to avoid
662 // allocating space in the container in the common case.
663 buffered_value first_;
664
665 // The rest of the buffered values.
666 typename traits_type::template container<buffered_value>::type rest_;
667};
