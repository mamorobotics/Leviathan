204 impl()

214 }
215
216 void operator()(cancellation_type_t in)
217 {
218 this->cancelled_ = in_filter_(in);
219 cancellation_type_t out = out_filter_(this->cancelled_);
220 if (out != cancellation_type::none)
221 this->signal_.emit(out);
222 }
