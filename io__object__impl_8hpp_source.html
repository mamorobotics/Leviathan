31template <typename IoObjectService,
32 typename Executor = io_context::executor_type>
33class io_object_impl
34{
35public:
36 // The type of the service that will be used to provide I/O operations.
37 typedef IoObjectService service_type;
38
39 // The underlying implementation type of I/O object.
40 typedef typename service_type::implementation_type implementation_type;
41
42 // The type of the executor associated with the object.
43 typedef Executor executor_type;
44
45 // Construct an I/O object using an executor.
46 explicit io_object_impl(int, const executor_type& ex)
47 : service_(&asio::use_service<IoObjectService>(
48 io_object_impl::get_context(ex))),
49 executor_(ex)
50 {
51 service_->construct(implementation_);
52 }
53
54 // Construct an I/O object using an execution context.
55 template <typename ExecutionContext>
56 explicit io_object_impl(int, int, ExecutionContext& context)

60 service_->construct(implementation_);
61 }

63 // Move-construct an I/O object.

66 executor_(other.get_executor())

68 service_->move_construct(implementation_, other.implementation_);
69 }
70
71 // Perform converting move-construction of an I/O object on the same service.
72 template <typename Executor1>
73 io_object_impl(io_object_impl<IoObjectService, Executor1>&& other)
74 : service_(&other.get_service()),
75 executor_(other.get_executor())
76 {
77 service_->move_construct(implementation_, other.get_implementation());

79
80 // Perform converting move-construction of an I/O object on another service.
81 template <typename IoObjectService1, typename Executor1>
82 io_object_impl(io_object_impl<IoObjectService1, Executor1>&& other)
83 : service_(&asio::use_service<IoObjectService>(
84 io_object_impl::get_context(other.get_executor()))),

86 {
87 service_->converting_move_construct(implementation_,

90
91 // Destructor.

96
97 // Move-assign an I/O object.
98 io_object_impl& operator=(io_object_impl&& other)
99 {
100 if (this != &other)
101 {
102 service_->move_assign(implementation_,
103 *other.service_, other.implementation_);
104 executor_.~executor_type();
105 new (&executor_) executor_type(other.executor_);
106 service_ = other.service_;
107 }
108 return *this;
109 }

141private:
142 // Helper function to get an executor's context.
143 template <typename T>
144 static execution_context& get_context(const T& t,
145 enable_if_t<execution::is_executor<T>::value>* = 0)

147 return asio::query(t, execution::context);

149
150 // Helper function to get an executor's context.

170};
