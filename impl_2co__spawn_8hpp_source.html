136template <typename T, typename Handler, typename Executor, typename Function>
137awaitable<awaitable_thread_entry_point, Executor> co_spawn_entry_point(
138 awaitable<T, Executor>*, co_spawn_state<Handler, Executor, Function> s)
139{
140 (void) co_await co_spawn_dispatch{};
141
142 (co_await awaitable_thread_has_context_switched{}) = false;
143 std::exception_ptr e = nullptr;
144 bool done = false;
145 try
146 {
147 T t = co_await s.function();
148
149 done = true;
150
151 bool switched = (co_await awaitable_thread_has_context_switched{});
152 if (!switched)
153 (void) co_await co_spawn_post();
154
155 (dispatch)(s.handler_work.get_executor(),
156 [handler = std::move(s.handler), t = std::move(t)]() mutable
157 {
158 std::move(handler)(std::exception_ptr(), std::move(t));
159 });
160
161 co_return;
162 }
163 catch (...)
164 {
165 if (done)
166 throw;
167
168 e = std::current_exception();
169 }
170
171 bool switched = (co_await awaitable_thread_has_context_switched{});
172 if (!switched)
173 (void) co_await co_spawn_post();
174
175 (dispatch)(s.handler_work.get_executor(),
176 [handler = std::move(s.handler), e]() mutable
177 {
178 std::move(handler)(e, T());
179 });
180}

285template <typename Executor>
286class initiate_co_spawn
287{
288public:
289 typedef Executor executor_type;
290
291 template <typename OtherExecutor>
292 explicit initiate_co_spawn(const OtherExecutor& ex)
293 : ex_(ex)
294 {
295 }
296
297 executor_type get_executor() const noexcept
298 {
299 return ex_;
300 }
301
302 template <typename Handler, typename F>
303 void operator()(Handler&& handler, F&& f) const
304 {
305 typedef result_of_t<F()> awaitable_type;
306 typedef decay_t<Handler> handler_type;
307 typedef decay_t<F> function_type;
308 typedef co_spawn_cancellation_handler<
309 handler_type, Executor> cancel_handler_type;
310
311 auto slot = asio::get_associated_cancellation_slot(handler);
312 cancel_handler_type* cancel_handler = slot.is_connected()
313 ? &slot.template emplace<cancel_handler_type>(handler, ex_)
314 : nullptr;
315
316 cancellation_slot proxy_slot(
317 cancel_handler
318 ? cancel_handler->slot()
319 : cancellation_slot());
320
321 cancellation_state cancel_state(proxy_slot);
322
323 auto a = (co_spawn_entry_point)(static_cast<awaitable_type*>(nullptr),
324 co_spawn_state<handler_type, Executor, function_type>(
325 std::forward<Handler>(handler), ex_, std::forward<F>(f)));
326 awaitable_handler<executor_type, void>(std::move(a),
327 ex_, proxy_slot, cancel_state).launch();
328 }
329
330private:
331 Executor ex_;
332};
