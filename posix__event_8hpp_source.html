45 // Signal the event. (Retained for backward compatibility.)
46 template <typename Lock>
47 void signal(Lock& lock)
48 {
49 this->signal_all(lock);
50 }
51
52 // Signal all waiters.
53 template <typename Lock>
54 void signal_all(Lock& lock)
55 {
56 ASIO_ASSERT(lock.locked());
57 (void)lock;
58 state_ |= 1;
59 ::pthread_cond_broadcast(&cond_); // Ignore EINVAL.
60 }
61
62 // Unlock the mutex and signal one waiter.
63 template <typename Lock>
64 void unlock_and_signal_one(Lock& lock)
65 {
66 ASIO_ASSERT(lock.locked());
67 state_ |= 1;
68 bool have_waiters = (state_ > 1);
69 lock.unlock();
70 if (have_waiters)
71 ::pthread_cond_signal(&cond_); // Ignore EINVAL.
72 }
73
74 // Unlock the mutex and signal one waiter who may destroy us.
75 template <typename Lock>
76 void unlock_and_signal_one_for_destruction(Lock& lock)
77 {
78 ASIO_ASSERT(lock.locked());
79 state_ |= 1;
80 bool have_waiters = (state_ > 1);
81 if (have_waiters)
82 ::pthread_cond_signal(&cond_); // Ignore EINVAL.
83 lock.unlock();
84 }
85
86 // If there's a waiter, unlock the mutex and signal it.
87 template <typename Lock>
88 bool maybe_unlock_and_signal_one(Lock& lock)
89 {
90 ASIO_ASSERT(lock.locked());
91 state_ |= 1;
92 if (state_ > 1)
93 {
94 lock.unlock();
95 ::pthread_cond_signal(&cond_); // Ignore EINVAL.
96 return true;
97 }
98 return false;
99 }
100
101 // Reset the event.
102 template <typename Lock>
103 void clear(Lock& lock)
104 {
105 ASIO_ASSERT(lock.locked());
106 (void)lock;
107 state_ &= ~std::size_t(1);
108 }
109
110 // Wait for the event to become signalled.
111 template <typename Lock>
112 void wait(Lock& lock)
113 {
114 ASIO_ASSERT(lock.locked());
115 while ((state_ & 1) == 0)
116 {
117 state_ += 2;
118 ::pthread_cond_wait(&cond_, &lock.mutex().mutex_); // Ignore EINVAL.
119 state_ -= 2;
120 }
121 }
122
123 // Timed wait for the event to become signalled.
124 template <typename Lock>
125 bool wait_for_usec(Lock& lock, long usec)
126 {
127 ASIO_ASSERT(lock.locked());
128 if ((state_ & 1) == 0)
129 {
130 state_ += 2;
131 timespec ts;
132#if (defined(__MACH__) && defined(__APPLE__)) \
133 || (defined(__ANDROID__) && (__ANDROID_API__ < 21) \
134 && defined(HAVE_PTHREAD_COND_TIMEDWAIT_RELATIVE))
135 ts.tv_sec = usec / 1000000;
136 ts.tv_nsec = (usec % 1000000) * 1000;
137 ::pthread_cond_timedwait_relative_np(
138 &cond_, &lock.mutex().mutex_, &ts); // Ignore EINVAL.
139#else // (defined(__MACH__) && defined(__APPLE__))
140 // || (defined(__ANDROID__) && (__ANDROID_API__ < 21)
141 // && defined(HAVE_PTHREAD_COND_TIMEDWAIT_RELATIVE))
142 if (::clock_gettime(CLOCK_MONOTONIC, &ts) == 0)
143 {
144 ts.tv_sec += usec / 1000000;
145 ts.tv_nsec += (usec % 1000000) * 1000;
146 ts.tv_sec += ts.tv_nsec / 1000000000;
147 ts.tv_nsec = ts.tv_nsec % 1000000000;
148 ::pthread_cond_timedwait(&cond_,
149 &lock.mutex().mutex_, &ts); // Ignore EINVAL.
150 }
151#endif // (defined(__MACH__) && defined(__APPLE__))
152 // || (defined(__ANDROID__) && (__ANDROID_API__ < 21)
153 // && defined(HAVE_PTHREAD_COND_TIMEDWAIT_RELATIVE))
154 state_ -= 2;
155 }
156 return (state_ & 1) != 0;
157 }
158
159private:
160 ::pthread_cond_t cond_;
161 std::size_t state_;
162};
