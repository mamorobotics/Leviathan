65bool operator<(const endpoint& e1, const endpoint& e2)
66{
67 if (e1.protocol() < e2.protocol())
68 return true;
69
70 if (e1.protocol() > e2.protocol())
71 return false;
72
73 using namespace std; // For memcmp.
74 std::size_t compare_size = e1.size() < e2.size() ? e1.size() : e2.size();
75 int compare_result = memcmp(e1.data(), e2.data(), compare_size);
76
77 if (compare_result < 0)
78 return true;
79
80 if (compare_result > 0)
81 return false;
82
83 return e1.size() < e2.size();
84}
