83 // Take ownership of the handler object.
84 ASIO_ASSUME(base != 0);

87 ptr p = { asio::detail::addressof(o->handler_), o, o };
88
89 ASIO_HANDLER_COMPLETION((*o));
90
91 // Take ownership of the operation's outstanding work.
92 handler_work<Handler, IoExecutor> w(
93 static_cast<handler_work<Handler, IoExecutor>&&>(
94 o->work_));
95
96 ASIO_ERROR_LOCATION(o->ec_);
97
98 // Make a copy of the handler so that the memory can be deallocated before
99 // the upcall is made. Even if we're not about to make an upcall, a
100 // sub-object of the handler may be the true owner of the memory associated
101 // with the handler. Consequently, a local copy of the handler is required
102 // to ensure that any owning sub-object remains valid until after we have
103 // deallocated the memory here.
104 detail::binder1<Handler, asio::error_code>
105 handler(o->handler_, o->ec_);
106 p.h = asio::detail::addressof(handler.handler_);
107 p.reset();
108
109 // Make the upcall if required.
110 if (owner)
111 {
112 fenced_block b(fenced_block::half);
113 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
114 w.complete(handler, handler.handler_);
115 ASIO_HANDLER_INVOCATION_END;
116 }
117 }
