106 void cancel(cancellation_type level = cancellation_type::all)
107 {
108 if (impl_ && !impl_->done)
109 {
110 asio::dispatch(impl_->executor,
111 [level, impl = impl_]{ impl->cancel.emit(level); });
112 }
113 }
114
116 bool completed() const noexcept
117 {
118 return impl_ && impl_->done;

160 {
161 std::shared_ptr<detail::promise_impl<

164 template <typename WaitHandler>
165 void operator()(WaitHandler&& handler) const
166 {
167 const auto alloc = get_associated_allocator(
168 handler, self_->get_allocator());
169
170 auto cancel = get_associated_cancellation_slot(handler);
171
172 if (self_->done)
173 {
174 auto exec = asio::get_associated_executor(
175 handler, self_->get_executor());
176
177 asio::post(exec,
178 [self = std::move(self_),
179 handler = std::forward<WaitHandler>(handler)]() mutable
180 {
181 self->apply(std::move(handler));
182 });
183 }
184 else
185 {
186 if (cancel.is_connected())
187 {
188 struct cancel_handler
189 {
190 std::weak_ptr<detail::promise_impl<
191 void(Ts...), Executor, Allocator>> self;
192
193 cancel_handler(
194 std::weak_ptr<detail::promise_impl<
195 void(Ts...), Executor, Allocator>> self)
196 : self(std::move(self))
197 {
198 }
199
200 void operator()(cancellation_type level) const
201 {
202 if (auto p = self.lock())
203 {
204 p->cancel.emit(level);
205 p->cancel_();
206 }
207 }
208 };
209 cancel.template emplace<cancel_handler>(self_);
210 }
211
212 self_->set_completion(alloc, std::forward<WaitHandler>(handler));
213 }
