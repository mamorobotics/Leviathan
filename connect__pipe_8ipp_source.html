45void create_pipe(native_pipe_handle p[2], asio::error_code& ec)
46{
47#if defined(ASIO_HAS_IOCP)
48 using namespace std; // For sprintf and memcmp.
49
50 static long counter1 = 0;
51 static long counter2 = 0;
52
53 long n1 = ::InterlockedIncrement(&counter1);
54 long n2 = (static_cast<unsigned long>(n1) % 0x10000000) == 0
55 ? ::InterlockedIncrement(&counter2)
56 : ::InterlockedExchangeAdd(&counter2, 0);
57
58 wchar_t pipe_name[128];
59#if defined(ASIO_HAS_SECURE_RTL)
60 swprintf_s(
61#else // defined(ASIO_HAS_SECURE_RTL)
62 _snwprintf(
63#endif // defined(ASIO_HAS_SECURE_RTL)
64 pipe_name, 128,
65 L"\\\\.\\pipe\\asio-A0812896-741A-484D-AF23-BE51BF620E22-%u-%ld-%ld",
66 static_cast<unsigned int>(::GetCurrentProcessId()), n1, n2);
67
68 p[0] = ::CreateNamedPipeW(pipe_name,
69 PIPE_ACCESS_INBOUND | FILE_FLAG_OVERLAPPED,
70 0, 1, 8192, 8192, 0, 0);
71
72 if (p[0] == INVALID_HANDLE_VALUE)
73 {
74 DWORD last_error = ::GetLastError();
75 ec.assign(last_error, asio::error::get_system_category());
76 return;
77 }
78
79 p[1] = ::CreateFileW(pipe_name, GENERIC_WRITE, 0,
80 0, OPEN_EXISTING, FILE_FLAG_OVERLAPPED, 0);
81
82 if (p[1] == INVALID_HANDLE_VALUE)
83 {
84 DWORD last_error = ::GetLastError();
85 ::CloseHandle(p[0]);
86 ec.assign(last_error, asio::error::get_system_category());
87 return;
88 }
89
90# if _WIN32_WINNT >= 0x601
91 unsigned char nonce[16];
92 if (::BCryptGenRandom(0, nonce, sizeof(nonce),
93 BCRYPT_USE_SYSTEM_PREFERRED_RNG) != 0)
94 {
95 ec = asio::error::connection_aborted;
96 ::CloseHandle(p[0]);
97 ::CloseHandle(p[1]);
98 return;
99 }
100
101 DWORD bytes_written = 0;
102 BOOL ok = ::WriteFile(p[1], nonce, sizeof(nonce), &bytes_written, 0);
103 if (!ok || bytes_written != sizeof(nonce))
104 {
105 ec = asio::error::connection_aborted;
106 ::CloseHandle(p[0]);
107 ::CloseHandle(p[1]);
108 return;
109 }
110
111 unsigned char nonce_check[sizeof(nonce)];
112 DWORD bytes_read = 0;
113 ok = ::ReadFile(p[0], nonce_check, sizeof(nonce), &bytes_read, 0);
114 if (!ok || bytes_read != sizeof(nonce)
115 || memcmp(nonce, nonce_check, sizeof(nonce)) != 0)
116 {
117 ec = asio::error::connection_aborted;
118 ::CloseHandle(p[0]);
119 ::CloseHandle(p[1]);
120 return;
121 }
122#endif // _WIN32_WINNT >= 0x601
123
124 asio::error::clear(ec);
125#else // defined(ASIO_HAS_IOCP)
126 int result = ::pipe(p);
127 detail::descriptor_ops::get_last_error(ec, result != 0);
128#endif // defined(ASIO_HAS_IOCP)
129}
