63 static void do_complete(void* owner, operation* base,
64 const asio::error_code& /*ec*/,
65 std::size_t /*bytes_transferred*/)
66 {
67 // Take ownership of the handler object.
68 ASIO_ASSUME(base != 0);
69 io_uring_null_buffers_op* o(static_cast<io_uring_null_buffers_op*>(base));
70 ptr p = { asio::detail::addressof(o->handler_), o, o };
71
72 ASIO_HANDLER_COMPLETION((*o));
73
74 // Take ownership of the operation's outstanding work.
75 handler_work<Handler, IoExecutor> w(
76 static_cast<handler_work<Handler, IoExecutor>&&>(
77 o->work_));
78
79 ASIO_ERROR_LOCATION(o->ec_);
80
81 // Make a copy of the handler so that the memory can be deallocated before
82 // the upcall is made. Even if we're not about to make an upcall, a
83 // sub-object of the handler may be the true owner of the memory associated
84 // with the handler. Consequently, a local copy of the handler is required
85 // to ensure that any owning sub-object remains valid until after we have
86 // deallocated the memory here.
87 detail::binder2<Handler, asio::error_code, std::size_t>
88 handler(o->handler_, o->ec_, o->bytes_transferred_);
89 p.h = asio::detail::addressof(handler.handler_);
90 p.reset();
91
92 // Make the upcall if required.
93 if (owner)
94 {
95 fenced_block b(fenced_block::half);
96 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
97 w.complete(handler, handler.handler_);
98 ASIO_HANDLER_INVOCATION_END;
99 }
100 }
101
102private:
103 Handler handler_;
104 handler_work<Handler, IoExecutor> work_;
105 int descriptor_;
106 int poll_flags_;
107};
