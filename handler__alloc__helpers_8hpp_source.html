30inline void* default_allocate(std::size_t s,
31 std::size_t align = ASIO_DEFAULT_ALIGN)
32{
33#if !defined(ASIO_DISABLE_SMALL_BLOCK_RECYCLING)
34 return asio::detail::thread_info_base::allocate(
35 asio::detail::thread_context::top_of_thread_call_stack(),
36 s, align);
37#else // !defined(ASIO_DISABLE_SMALL_BLOCK_RECYCLING)
38 return asio::aligned_new(align, s);
39#endif // !defined(ASIO_DISABLE_SMALL_BLOCK_RECYCLING)
40}

42inline void default_deallocate(void* p, std::size_t s)
43{
44#if !defined(ASIO_DISABLE_SMALL_BLOCK_RECYCLING)
45 asio::detail::thread_info_base::deallocate(
46 asio::detail::thread_context::top_of_thread_call_stack(), p, s);
47#else // !defined(ASIO_DISABLE_SMALL_BLOCK_RECYCLING)
48 (void)s;
49 asio::aligned_delete(p);
50#endif // !defined(ASIO_DISABLE_SMALL_BLOCK_RECYCLING)
51}
