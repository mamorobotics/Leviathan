58void endpoint::resize(std::size_t new_size)
59{
60 if (new_size > sizeof(asio::detail::sockaddr_un_type))
61 {
62 asio::error_code ec(asio::error::invalid_argument);
63 asio::detail::throw_error(ec);
64 }
65 else if (new_size == 0)
66 {
67 path_length_ = 0;
68 }
69 else
70 {
71 path_length_ = new_size
72 - offsetof(asio::detail::sockaddr_un_type, sun_path);
73
74 // The path returned by the operating system may be NUL-terminated.
75 if (path_length_ > 0 && data_.local.sun_path[path_length_ - 1] == 0)
76 --path_length_;
77 }
78}

106void endpoint::init(const char* path_name, std::size_t path_length)
107{
108 if (path_length > sizeof(data_.local.sun_path) - 1)
109 {
110 // The buffer is not large enough to store this address.
111 asio::error_code ec(asio::error::name_too_long);
112 asio::detail::throw_error(ec);
113 }
114
115 using namespace std; // For memset and memcpy.
116 memset(&data_.local, 0, sizeof(asio::detail::sockaddr_un_type));
117 data_.local.sun_family = AF_UNIX;
118 if (path_length > 0)
119 memcpy(data_.local.sun_path, path_name, path_length);
120 path_length_ = path_length;
121}
