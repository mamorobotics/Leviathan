36 // According to the OpenSSL documentation, this is the buffer size that is
37 // sufficient to hold the largest possible TLS record.
38 enum { max_tls_record_size = 17 * 1024 };
39
40 template <typename Executor>
41 stream_core(SSL_CTX* context, const Executor& ex)
42 : engine_(context),
43 pending_read_(ex),
44 pending_write_(ex),
45 output_buffer_space_(max_tls_record_size),
46 output_buffer_(asio::buffer(output_buffer_space_)),
47 input_buffer_space_(max_tls_record_size),
48 input_buffer_(asio::buffer(input_buffer_space_))
49 {
50 pending_read_.expires_at(neg_infin());
51 pending_write_.expires_at(neg_infin());
52 }
