31// Class to adapt a as_single_t as a completion handler.
32template <typename Handler>
33class as_single_handler
34{
35public:
36 typedef void result_type;
37
38 template <typename CompletionToken>
39 as_single_handler(as_single_t<CompletionToken> e)
40 : handler_(static_cast<CompletionToken&&>(e.token_))
41 {
42 }
43
44 template <typename RedirectedHandler>
45 as_single_handler(RedirectedHandler&& h)
46 : handler_(static_cast<RedirectedHandler&&>(h))
47 {
48 }
49
50 void operator()()
51 {
52 static_cast<Handler&&>(handler_)();
53 }
54
55 template <typename Arg>
56 void operator()(Arg&& arg)
57 {
58 static_cast<Handler&&>(handler_)(static_cast<Arg&&>(arg));
59 }
60
61 template <typename... Args>
62 void operator()(Args&&... args)
63 {
64 static_cast<Handler&&>(handler_)(
65 std::make_tuple(static_cast<Args&&>(args)...));
66 }
67
68//private:
69 Handler handler_;
70};
