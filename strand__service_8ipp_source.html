53void strand_service::shutdown()
54{

69void strand_service::construct(strand_service::implementation_type& impl)
70{
71 asio::detail::mutex::scoped_lock lock(mutex_);
72
73 std::size_t salt = salt_++;
74#if defined(ASIO_ENABLE_SEQUENTIAL_STRAND_ALLOCATION)
75 std::size_t index = salt;
76#else // defined(ASIO_ENABLE_SEQUENTIAL_STRAND_ALLOCATION)
77 std::size_t index = reinterpret_cast<std::size_t>(&impl);
78 index += (reinterpret_cast<std::size_t>(&impl) >> 3);
79 index ^= salt + 0x9e3779b9 + (index << 6) + (index >> 2);
80#endif // defined(ASIO_ENABLE_SEQUENTIAL_STRAND_ALLOCATION)
81 index = index % num_implementations;
82
83 if (!implementations_[index].get())
84 implementations_[index].reset(new strand_impl);
85 impl = implementations_[index].get();
86}

111void strand_service::do_dispatch(implementation_type& impl, operation* op)
112{
113 // If we are running inside the io_context, and no other handler already
114 // holds the strand lock, then the handler can run immediately.
115 bool can_dispatch = io_context_impl_.can_dispatch();
116 impl->mutex_.lock();
117 if (can_dispatch && !impl->locked_)
118 {
119 // Immediate invocation is allowed.
120 impl->locked_ = true;
121 impl->mutex_.unlock();
122
123 // Indicate that this strand is executing on the current thread.
124 call_stack<strand_impl>::context ctx(impl);
125
126 // Ensure the next handler, if any, is scheduled on block exit.
127 on_dispatch_exit on_exit = { &io_context_impl_, impl };
128 (void)on_exit;
129
130 op->complete(&io_context_impl_, asio::error_code(), 0);
131 return;
132 }
133
134 if (impl->locked_)
135 {
136 // Some other handler already holds the strand lock. Enqueue for later.
137 impl->waiting_queue_.push(op);
138 impl->mutex_.unlock();
139 }
140 else
141 {
142 // The handler is acquiring the strand lock and so is responsible for
143 // scheduling the strand.
144 impl->locked_ = true;
145 impl->mutex_.unlock();
146 impl->ready_queue_.push(op);
147 io_context_impl_.post_immediate_completion(impl, false);
148 }
149}

151void strand_service::do_post(implementation_type& impl,
152 operation* op, bool is_continuation)
153{
154 impl->mutex_.lock();
155 if (impl->locked_)
156 {
157 // Some other handler already holds the strand lock. Enqueue for later.
158 impl->waiting_queue_.push(op);
159 impl->mutex_.unlock();
160 }

163 // The handler is acquiring the strand lock and so is responsible for

165 impl->locked_ = true;
166 impl->mutex_.unlock();
167 impl->ready_queue_.push(op);
168 io_context_impl_.post_immediate_completion(impl, is_continuation);

172void strand_service::do_complete(void* owner, operation* base,
173 const asio::error_code& ec, std::size_t /*bytes_transferred*/)
174{
175 if (owner)
176 {
177 strand_impl* impl = static_cast<strand_impl*>(base);
178
179 // Indicate that this strand is executing on the current thread.
180 call_stack<strand_impl>::context ctx(impl);
181
182 // Ensure the next handler, if any, is scheduled on block exit.
183 on_do_complete_exit on_exit;
184 on_exit.owner_ = static_cast<io_context_impl*>(owner);
185 on_exit.impl_ = impl;
186
187 // Run all ready handlers. No lock is required since the ready queue is
188 // accessed only within the strand.
189 while (operation* o = impl->ready_queue_.front())
190 {
191 impl->ready_queue_.pop();
192 o->complete(owner, ec, 0);
193 }
194 }
195}
