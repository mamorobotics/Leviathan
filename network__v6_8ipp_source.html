45ASIO_DECL address_v6 network_v6::network() const noexcept
46{
47 address_v6::bytes_type bytes(address_.to_bytes());
48 for (std::size_t i = 0; i < 16; ++i)
49 {
50 if (prefix_length_ <= i * 8)
51 bytes[i] = 0;
52 else if (prefix_length_ < (i + 1) * 8)
53 bytes[i] &= 0xFF00 >> (prefix_length_ % 8);
54 }
55 return address_v6(bytes, address_.scope_id());
56}

58address_v6_range network_v6::hosts() const noexcept
59{
60 address_v6::bytes_type begin_bytes(address_.to_bytes());
61 address_v6::bytes_type end_bytes(address_.to_bytes());
62 for (std::size_t i = 0; i < 16; ++i)
63 {
64 if (prefix_length_ <= i * 8)
65 {
66 begin_bytes[i] = 0;
67 end_bytes[i] = 0xFF;
68 }
69 else if (prefix_length_ < (i + 1) * 8)
70 {
71 begin_bytes[i] &= 0xFF00 >> (prefix_length_ % 8);
72 end_bytes[i] |= 0xFF >> (prefix_length_ % 8);
73 }
74 }
75 return address_v6_range(
76 address_v6_iterator(address_v6(begin_bytes, address_.scope_id())),
77 ++address_v6_iterator(address_v6(end_bytes, address_.scope_id())));
78}

80bool network_v6::is_subnet_of(const network_v6& other) const
81{
82 if (other.prefix_length_ >= prefix_length_)
83 return false; // Only real subsets are allowed.
84 const network_v6 me(address_, other.prefix_length_);
85 return other.canonical() == me.canonical();
86}

129network_v6 make_network_v6(const std::string& str,
130 asio::error_code& ec)
131{
132 std::string::size_type pos = str.find_first_of("/");
133
134 if (pos == std::string::npos)
135 {
136 ec = asio::error::invalid_argument;
137 return network_v6();
138 }
139
140 if (pos == str.size() - 1)
141 {
142 ec = asio::error::invalid_argument;
143 return network_v6();
144 }
145
146 std::string::size_type end = str.find_first_not_of("0123456789", pos + 1);
147 if (end != std::string::npos)
148 {
149 ec = asio::error::invalid_argument;
150 return network_v6();
151 }
152
153 const address_v6 addr = make_address_v6(str.substr(0, pos), ec);
154 if (ec)
155 return network_v6();
156
157 const int prefix_len = std::atoi(str.substr(pos + 1).c_str());
158 if (prefix_len < 0 || prefix_len > 128)
159 {
160 ec = asio::error::invalid_argument;
161 return network_v6();
162 }
163
164 return network_v6(addr, static_cast<unsigned short>(prefix_len));
165}
