38template <typename Executor1, typename Executor2>
39ASIO_SYNC_OP_VOID connect_pipe(basic_readable_pipe<Executor1>& read_end,
40 basic_writable_pipe<Executor2>& write_end, asio::error_code& ec)
41{
42 detail::native_pipe_handle p[2];
43 detail::create_pipe(p, ec);
44 if (ec)
45 ASIO_SYNC_OP_VOID_RETURN(ec);
46
47 read_end.assign(p[0], ec);
48 if (ec)
49 {
50 detail::close_pipe(p[0]);
51 detail::close_pipe(p[1]);
52 ASIO_SYNC_OP_VOID_RETURN(ec);
53 }
54
55 write_end.assign(p[1], ec);
56 if (ec)
57 {
58 asio::error_code temp_ec;
59 read_end.close(temp_ec);
60 detail::close_pipe(p[1]);
61 ASIO_SYNC_OP_VOID_RETURN(ec);
62 }
63
64 ASIO_SYNC_OP_VOID_RETURN(ec);
65}
