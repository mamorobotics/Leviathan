33bool rfc2818_verification::operator()(
34 bool preverified, verify_context& ctx) const
35{
36 using namespace std; // For memcmp.
37
38 // Don't bother looking at certificates that have failed pre-verification.
39 if (!preverified)
40 return false;
41
42 // We're only interested in checking the certificate at the end of the chain.
43 int depth = X509_STORE_CTX_get_error_depth(ctx.native_handle());
44 if (depth > 0)
45 return true;
46
47 // Try converting the host name to an address. If it is an address then we
48 // need to look for an IP address in the certificate rather than a host name.
49 asio::error_code ec;
50 ip::address address = ip::make_address(host_, ec);
51 bool is_address = !ec;
52
53 X509* cert = X509_STORE_CTX_get_current_cert(ctx.native_handle());
54
55 // Go through the alternate names in the certificate looking for matching DNS
56 // or IP address entries.
57 GENERAL_NAMES* gens = static_cast<GENERAL_NAMES*>(
58 X509_get_ext_d2i(cert, NID_subject_alt_name, 0, 0));
59 for (int i = 0; i < sk_GENERAL_NAME_num(gens); ++i)
60 {
61 GENERAL_NAME* gen = sk_GENERAL_NAME_value(gens, i);
62 if (gen->type == GEN_DNS && !is_address)
63 {
64 ASN1_IA5STRING* domain = gen->d.dNSName;
65 if (domain->type == V_ASN1_IA5STRING && domain->data && domain->length)
66 {
67 const char* pattern = reinterpret_cast<const char*>(domain->data);
68 std::size_t pattern_length = domain->length;
69 if (match_pattern(pattern, pattern_length, host_.c_str()))
70 {
71 GENERAL_NAMES_free(gens);
72 return true;
73 }
74 }
75 }
76 else if (gen->type == GEN_IPADD && is_address)
77 {
78 ASN1_OCTET_STRING* ip_address = gen->d.iPAddress;
79 if (ip_address->type == V_ASN1_OCTET_STRING && ip_address->data)
80 {
81 if (address.is_v4() && ip_address->length == 4)
82 {
83 ip::address_v4::bytes_type bytes = address.to_v4().to_bytes();
84 if (memcmp(bytes.data(), ip_address->data, 4) == 0)
85 {
86 GENERAL_NAMES_free(gens);
87 return true;
88 }
89 }
90 else if (address.is_v6() && ip_address->length == 16)
91 {
92 ip::address_v6::bytes_type bytes = address.to_v6().to_bytes();
93 if (memcmp(bytes.data(), ip_address->data, 16) == 0)
94 {
95 GENERAL_NAMES_free(gens);
96 return true;
97 }
98 }
99 }
100 }
101 }
102 GENERAL_NAMES_free(gens);
103
104 // No match in the alternate names, so try the common names. We should only
105 // use the "most specific" common name, which is the last one in the list.
106 X509_NAME* name = X509_get_subject_name(cert);
107 int i = -1;
108 ASN1_STRING* common_name = 0;
109 while ((i = X509_NAME_get_index_by_NID(name, NID_commonName, i)) >= 0)
110 {
111 X509_NAME_ENTRY* name_entry = X509_NAME_get_entry(name, i);
112 common_name = X509_NAME_ENTRY_get_data(name_entry);
113 }
114 if (common_name && common_name->data && common_name->length)
115 {
116 const char* pattern = reinterpret_cast<const char*>(common_name->data);
117 std::size_t pattern_length = common_name->length;
118 if (match_pattern(pattern, pattern_length, host_.c_str()))
119 return true;
120 }
121
122 return false;
123}

125bool rfc2818_verification::match_pattern(const char* pattern,
126 std::size_t pattern_length, const char* host)
127{
128 using namespace std; // For tolower.
129
130 const char* p = pattern;
131 const char* p_end = p + pattern_length;
132 const char* h = host;
133
134 while (p != p_end && *h)
135 {
136 if (*p == '*')
137 {
138 ++p;
139 while (*h && *h != '.')
140 if (match_pattern(p, p_end - p, h++))
141 return true;
142 }
143 else if (tolower(*p) == tolower(*h))
144 {
145 ++p;
146 ++h;
147 }
148 else
149 {
150 return false;
151 }
152 }
153
154 return p == p_end && !*h;
155}
