88 static void do_complete(void* owner, operation* base,
89 const asio::error_code& /*ec*/,
90 std::size_t /*bytes_transferred*/)
91 {
92 // Take ownership of the handler object.
93 ASIO_ASSUME(base != 0);
94 io_uring_socket_connect_op* o
95 (static_cast<io_uring_socket_connect_op*>(base));
96 ptr p = { asio::detail::addressof(o->handler_), o, o };
97
98 ASIO_HANDLER_COMPLETION((*o));
99
100 // Take ownership of the operation's outstanding work.
101 handler_work<Handler, IoExecutor> w(
102 static_cast<handler_work<Handler, IoExecutor>&&>(
103 o->work_));
104
105 ASIO_ERROR_LOCATION(o->ec_);
106
107 // Make a copy of the handler so that the memory can be deallocated before
108 // the upcall is made. Even if we're not about to make an upcall, a
109 // sub-object of the handler may be the true owner of the memory associated
110 // with the handler. Consequently, a local copy of the handler is required
111 // to ensure that any owning sub-object remains valid until after we have
112 // deallocated the memory here.
113 detail::binder1<Handler, asio::error_code>
114 handler(o->handler_, o->ec_);
115 p.h = asio::detail::addressof(handler.handler_);
116 p.reset();
117
118 // Make the upcall if required.
119 if (owner)
120 {
121 fenced_block b(fenced_block::half);
122 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
123 w.complete(handler, handler.handler_);
124 ASIO_HANDLER_INVOCATION_END;
125 }
126 }
127
128private:
129 Handler handler_;
130 handler_work<Handler, IoExecutor> work_;
131};
