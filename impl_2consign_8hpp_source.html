30// Class to adapt a consign_t as a completion handler.
31template <typename Handler, typename... Values>
32class consign_handler
33{
34public:
35 typedef void result_type;
36
37 template <typename H>
38 consign_handler(H&& handler, std::tuple<Values...> values)
39 : handler_(static_cast<H&&>(handler)),
40 values_(static_cast<std::tuple<Values...>&&>(values))
41 {
42 }
43
44 template <typename... Args>
45 void operator()(Args&&... args)
46 {
47 static_cast<Handler&&>(handler_)(static_cast<Args&&>(args)...);
48 }
49
50//private:
51 Handler handler_;
52 std::tuple<Values...> values_;
53};
