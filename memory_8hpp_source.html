73inline void* aligned_new(std::size_t align, std::size_t size)
74{
75#if defined(ASIO_HAS_STD_ALIGNED_ALLOC)
76 align = (align < ASIO_DEFAULT_ALIGN) ? ASIO_DEFAULT_ALIGN : align;
77 size = (size % align == 0) ? size : size + (align - size % align);
78 void* ptr = std::aligned_alloc(align, size);
79 if (!ptr)
80 {
81 std::bad_alloc ex;
82 asio::detail::throw_exception(ex);
83 }
84 return ptr;
85#elif defined(ASIO_HAS_BOOST_ALIGN)
86 align = (align < ASIO_DEFAULT_ALIGN) ? ASIO_DEFAULT_ALIGN : align;
87 size = (size % align == 0) ? size : size + (align - size % align);
88 void* ptr = boost::alignment::aligned_alloc(align, size);
89 if (!ptr)
90 {
91 std::bad_alloc ex;
92 asio::detail::throw_exception(ex);
93 }
94 return ptr;
95#elif defined(ASIO_MSVC)
96 align = (align < ASIO_DEFAULT_ALIGN) ? ASIO_DEFAULT_ALIGN : align;
97 size = (size % align == 0) ? size : size + (align - size % align);
98 void* ptr = _aligned_malloc(size, align);
99 if (!ptr)
100 {
101 std::bad_alloc ex;
102 asio::detail::throw_exception(ex);
103 }
104 return ptr;
105#else // defined(ASIO_MSVC)
106 (void)align;
107 return ::operator new(size);
108#endif // defined(ASIO_MSVC)
109}
