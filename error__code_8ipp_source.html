45 std::string message(int value) const
46 {
47#if defined(ASIO_WINDOWS_RUNTIME) || defined(ASIO_WINDOWS_APP)
48 std::wstring wmsg(128, wchar_t());
49 for (;;)
50 {
51 DWORD wlength = ::FormatMessageW(FORMAT_MESSAGE_FROM_SYSTEM
52 | FORMAT_MESSAGE_IGNORE_INSERTS, 0, value,
53 MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
54 &wmsg[0], static_cast<DWORD>(wmsg.size()), 0);
55 if (wlength == 0 && ::GetLastError() == ERROR_INSUFFICIENT_BUFFER)
56 {
57 wmsg.resize(wmsg.size() + wmsg.size() / 2);
58 continue;
59 }
60 if (wlength && wmsg[wlength - 1] == '\n')
61 --wlength;
62 if (wlength && wmsg[wlength - 1] == '\r')
63 --wlength;
64 if (wlength)
65 {
66 std::string msg(wlength * 2, char());
67 int length = ::WideCharToMultiByte(CP_ACP, 0,
68 wmsg.c_str(), static_cast<int>(wlength),
69 &msg[0], static_cast<int>(wlength * 2), 0, 0);
70 if (length <= 0)
71 return "asio.system error";
72 msg.resize(static_cast<std::size_t>(length));
73 return msg;
74 }
75 else
76 return "asio.system error";
77 }
78#elif defined(ASIO_WINDOWS) || defined(__CYGWIN__)
79 char* msg = 0;
80 DWORD length = ::FormatMessageA(FORMAT_MESSAGE_ALLOCATE_BUFFER
81 | FORMAT_MESSAGE_FROM_SYSTEM
82 | FORMAT_MESSAGE_IGNORE_INSERTS, 0, value,
83 MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT), (char*)&msg, 0, 0);
84 detail::local_free_on_block_exit local_free_obj(msg);
85 if (length && msg[length - 1] == '\n')
86 msg[--length] = '\0';
87 if (length && msg[length - 1] == '\r')
88 msg[--length] = '\0';
89 if (length)
90 return msg;
91 else
92 return "asio.system error";
93#else // defined(ASIO_WINDOWS_DESKTOP) || defined(__CYGWIN__)
94#if !defined(__sun)
95 if (value == ECANCELED)
96 return "Operation aborted.";
97#endif // !defined(__sun)
98#if defined(__sun) || defined(__QNX__) || defined(__SYMBIAN32__)
99 using namespace std;
100 return strerror(value);
101#else
102 char buf[256] = "";
103 using namespace std;
104 return strerror_result(strerror_r(value, buf, sizeof(buf)), buf);
105#endif
106#endif // defined(ASIO_WINDOWS_DESKTOP) || defined(__CYGWIN__)
107 }
