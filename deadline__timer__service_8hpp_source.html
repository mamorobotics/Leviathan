46template <typename Time_Traits>
47class deadline_timer_service
48 : public execution_context_service_base<deadline_timer_service<Time_Traits>>

50public:

52 typedef typename Time_Traits::time_type time_type;

57 // The implementation type of the timer. This type is dependent on the
58 // underlying implementation of the timer service.
59 struct implementation_type
60 : private asio::detail::noncopyable

62 time_type expiry;
63 bool might_have_pending_waits;
64 typename timer_queue<Time_Traits>::per_timer_data timer_data;

67 // Constructor.
68 deadline_timer_service(execution_context& context)
69 : execution_context_service_base<

102 // Move-construct a new timer implementation.
103 void move_construct(implementation_type& impl,
104 implementation_type& other_impl)
105 {
106 scheduler_.move_timer(timer_queue_, impl.timer_data, other_impl.timer_data);
107
108 impl.expiry = other_impl.expiry;

111 impl.might_have_pending_waits = other_impl.might_have_pending_waits;

115 // Move-assign from another timer implementation.
116 void move_assign(implementation_type& impl,
117 deadline_timer_service& other_service,
118 implementation_type& other_impl)
119 {
120 if (this != &other_service)
121 if (impl.might_have_pending_waits)
122 scheduler_.cancel_timer(timer_queue_, impl.timer_data);
123
124 other_service.scheduler_.move_timer(other_service.timer_queue_,
125 impl.timer_data, other_impl.timer_data);
126
127 impl.expiry = other_impl.expiry;
128 other_impl.expiry = time_type();
129
130 impl.might_have_pending_waits = other_impl.might_have_pending_waits;
131 other_impl.might_have_pending_waits = false;
132 }
133
134 // Move-construct a new timer implementation.

138 move_construct(impl, other_impl);
139 }

149 // Cancel any asynchronous wait operations associated with the timer.
150 std::size_t cancel(implementation_type& impl, asio::error_code& ec)
151 {
152 if (!impl.might_have_pending_waits)
153 {
154 ec = asio::error_code();
155 return 0;
156 }
157
158 ASIO_HANDLER_OPERATION((scheduler_.context(),
159 "deadline_timer", &impl, 0, "cancel"));
160
161 std::size_t count = scheduler_.cancel_timer(timer_queue_, impl.timer_data);
162 impl.might_have_pending_waits = false;
163 ec = asio::error_code();
164 return count;
165 }
166
167 // Cancels one asynchronous wait operation associated with the timer.
168 std::size_t cancel_one(implementation_type& impl,
169 asio::error_code& ec)
170 {
171 if (!impl.might_have_pending_waits)
172 {
173 ec = asio::error_code();
174 return 0;
175 }
176
177 ASIO_HANDLER_OPERATION((scheduler_.context(),
178 "deadline_timer", &impl, 0, "cancel_one"));
179
180 std::size_t count = scheduler_.cancel_timer(
181 timer_queue_, impl.timer_data, 1);
182 if (count == 0)
183 impl.might_have_pending_waits = false;
184 ec = asio::error_code();
185 return count;
186 }

189 time_type expiry(const implementation_type& impl) const
190 {
191 return impl.expiry;
192 }

194 // Get the expiry time for the timer as an absolute time.

200 // Get the expiry time for the timer relative to now.
201 duration_type expires_from_now(const implementation_type& impl) const
202 {
203 return Time_Traits::subtract(this->expiry(impl), Time_Traits::now());
204 }
205
206 // Set the expiry time for the timer as an absolute time.
207 std::size_t expires_at(implementation_type& impl,
208 const time_type& expiry_time, asio::error_code& ec)
209 {
210 std::size_t count = cancel(impl, ec);
211 impl.expiry = expiry_time;
212 ec = asio::error_code();
213 return count;
214 }
215
216 // Set the expiry time for the timer relative to now.
217 std::size_t expires_after(implementation_type& impl,
218 const duration_type& expiry_time, asio::error_code& ec)
219 {
220 return expires_at(impl,
221 Time_Traits::add(Time_Traits::now(), expiry_time), ec);
222 }
223
224 // Set the expiry time for the timer relative to now.
225 std::size_t expires_from_now(implementation_type& impl,
226 const duration_type& expiry_time, asio::error_code& ec)
227 {
228 return expires_at(impl,
229 Time_Traits::add(Time_Traits::now(), expiry_time), ec);
230 }

232 // Perform a blocking wait on the timer.
233 void wait(implementation_type& impl, asio::error_code& ec)
234 {
235 time_type now = Time_Traits::now();
236 ec = asio::error_code();
237 while (Time_Traits::less_than(now, impl.expiry) && !ec)
238 {
239 this->do_wait(Time_Traits::to_posix_duration(
240 Time_Traits::subtract(impl.expiry, now)), ec);
241 now = Time_Traits::now();
242 }
243 }

245 // Start an asynchronous wait on the timer.
246 template <typename Handler, typename IoExecutor>
247 void async_wait(implementation_type& impl,
248 Handler& handler, const IoExecutor& io_ex)
249 {
250 associated_cancellation_slot_t<Handler> slot
251 = asio::get_associated_cancellation_slot(handler);
252
253 // Allocate and construct an operation to wrap the handler.
254 typedef wait_handler<Handler, IoExecutor> op;
255 typename op::ptr p = { asio::detail::addressof(handler),
256 op::ptr::allocate(handler), 0 };
257 p.p = new (p.v) op(handler, io_ex);
258
259 // Optionally register for per-operation cancellation.
260 if (slot.is_connected())
261 {
262 p.p->cancellation_key_ =
263 &slot.template emplace<op_cancellation>(this, &impl.timer_data);
264 }
265
266 impl.might_have_pending_waits = true;
267
268 ASIO_HANDLER_CREATION((scheduler_.context(),
269 *p.p, "deadline_timer", &impl, 0, "async_wait"));
270
271 scheduler_.schedule_timer(timer_queue_, impl.expiry, impl.timer_data, p.p);
272 p.v = p.p = 0;
273 }

275private:
276 // Helper function to wait given a duration type. The duration type should
277 // either be of type boost::posix_time::time_duration, or implement the
278 // required subset of its interface.
279 template <typename Duration>
280 void do_wait(const Duration& timeout, asio::error_code& ec)
281 {
282#if defined(ASIO_WINDOWS_RUNTIME)
283 std::this_thread::sleep_for(
284 std::chrono::seconds(timeout.total_seconds())
285 + std::chrono::microseconds(timeout.total_microseconds()));
286 ec = asio::error_code();
287#else // defined(ASIO_WINDOWS_RUNTIME)
288 ::timeval tv;
289 tv.tv_sec = timeout.total_seconds();
290 tv.tv_usec = timeout.total_microseconds() % 1000000;
291 socket_ops::select(0, 0, 0, 0, &tv, ec);
292#endif // defined(ASIO_WINDOWS_RUNTIME)
293 }

295 // Helper class used to implement per-operation cancellation.
296 class op_cancellation
297 {
298 public:
299 op_cancellation(deadline_timer_service* s,
300 typename timer_queue<Time_Traits>::per_timer_data* p)
301 : service_(s),
302 timer_data_(p)
303 {
304 }
305
306 void operator()(cancellation_type_t type)
307 {
308 if (!!(type &
309 (cancellation_type::terminal
310 | cancellation_type::partial
311 | cancellation_type::total)))
312 {
313 service_->scheduler_.cancel_timer_by_key(
314 service_->timer_queue_, timer_data_, this);
315 }
316 }
317
318 private:
319 deadline_timer_service* service_;
320 typename timer_queue<Time_Traits>::per_timer_data* timer_data_;
321 };

325

327 timer_scheduler& scheduler_;
328};
