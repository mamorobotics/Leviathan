82void reactive_socket_service_base::destroy(
83 reactive_socket_service_base::base_implementation_type& impl)
84{
85 if (impl.socket_ != invalid_socket)
86 {
87 ASIO_HANDLER_OPERATION((reactor_.context(),
88 "socket", &impl, impl.socket_, "close"));
89
90 reactor_.deregister_descriptor(impl.socket_, impl.reactor_data_,
91 (impl.state_ & socket_ops::possible_dup) == 0);
92
93 asio::error_code ignored_ec;
94 socket_ops::close(impl.socket_, impl.state_, true, ignored_ec);
95
96 reactor_.cleanup_descriptor_data(impl.reactor_data_);
97 }
98}

100asio::error_code reactive_socket_service_base::close(
101 reactive_socket_service_base::base_implementation_type& impl,
102 asio::error_code& ec)
103{
104 if (is_open(impl))
105 {
106 ASIO_HANDLER_OPERATION((reactor_.context(),
107 "socket", &impl, impl.socket_, "close"));
108
109 reactor_.deregister_descriptor(impl.socket_, impl.reactor_data_,
110 (impl.state_ & socket_ops::possible_dup) == 0);
111
112 socket_ops::close(impl.socket_, impl.state_, false, ec);
113
114 reactor_.cleanup_descriptor_data(impl.reactor_data_);
115 }
116 else
117 {
118 ec = asio::error_code();
119 }
120
121 // The descriptor is closed by the OS even if close() returns an error.
122 //
123 // (Actually, POSIX says the state of the descriptor is unspecified. On
124 // Linux the descriptor is apparently closed anyway; e.g. see
125 // http://lkml.org/lkml/2005/9/10/129
126 // We'll just have to assume that other OSes follow the same behaviour. The
127 // known exception is when Windows's closesocket() function fails with
128 // WSAEWOULDBLOCK, but this case is handled inside socket_ops::close().
129 construct(impl);
130
131 return ec;
132}

134socket_type reactive_socket_service_base::release(
135 reactive_socket_service_base::base_implementation_type& impl,
136 asio::error_code& ec)
137{
138 if (!is_open(impl))
139 {
140 ec = asio::error::bad_descriptor;
141 return invalid_socket;
142 }
143
144 ASIO_HANDLER_OPERATION((reactor_.context(),
145 "socket", &impl, impl.socket_, "release"));
146
147 reactor_.deregister_descriptor(impl.socket_, impl.reactor_data_, false);
148 reactor_.cleanup_descriptor_data(impl.reactor_data_);
149 socket_type sock = impl.socket_;
150 construct(impl);
151 ec = asio::error_code();
152 return sock;
153}

173asio::error_code reactive_socket_service_base::do_open(
174 reactive_socket_service_base::base_implementation_type& impl,
175 int af, int type, int protocol, asio::error_code& ec)
176{
177 if (is_open(impl))
178 {
179 ec = asio::error::already_open;
180 return ec;
181 }
182
183 socket_holder sock(socket_ops::socket(af, type, protocol, ec));
184 if (sock.get() == invalid_socket)
185 return ec;
186
187 if (int err = reactor_.register_descriptor(sock.get(), impl.reactor_data_))
188 {
189 ec = asio::error_code(err,
190 asio::error::get_system_category());
191 return ec;
192 }
193
194 impl.socket_ = sock.release();
195 switch (type)
196 {
197 case SOCK_STREAM: impl.state_ = socket_ops::stream_oriented; break;
198 case SOCK_DGRAM: impl.state_ = socket_ops::datagram_oriented; break;
199 default: impl.state_ = 0; break;
200 }
201 ec = asio::error_code();
202 return ec;
203}

205asio::error_code reactive_socket_service_base::do_assign(
206 reactive_socket_service_base::base_implementation_type& impl, int type,
207 const reactive_socket_service_base::native_handle_type& native_socket,
208 asio::error_code& ec)
209{
210 if (is_open(impl))
211 {
212 ec = asio::error::already_open;
213 return ec;
214 }
215
216 if (int err = reactor_.register_descriptor(
217 native_socket, impl.reactor_data_))
218 {
219 ec = asio::error_code(err,
220 asio::error::get_system_category());
221 return ec;
222 }
223
224 impl.socket_ = native_socket;
225 switch (type)
226 {
227 case SOCK_STREAM: impl.state_ = socket_ops::stream_oriented; break;
228 case SOCK_DGRAM: impl.state_ = socket_ops::datagram_oriented; break;
229 default: impl.state_ = 0; break;
230 }
231 impl.state_ |= socket_ops::possible_dup;
232 ec = asio::error_code();
233 return ec;
234}

236void reactive_socket_service_base::do_start_op(
237 reactive_socket_service_base::base_implementation_type& impl, int op_type,
238 reactor_op* op, bool is_continuation, bool is_non_blocking, bool noop,
239 void (*on_immediate)(operation* op, bool, const void*),
240 const void* immediate_arg)
241{
242 if (!noop)
243 {
244 if ((impl.state_ & socket_ops::non_blocking)
245 || socket_ops::set_internal_non_blocking(
246 impl.socket_, impl.state_, true, op->ec_))
247 {
248 reactor_.start_op(op_type, impl.socket_, impl.reactor_data_, op,
249 is_continuation, is_non_blocking, on_immediate, immediate_arg);
250 return;
251 }
252 }
253
254 on_immediate(op, is_continuation, immediate_arg);
255}

257void reactive_socket_service_base::do_start_accept_op(
258 reactive_socket_service_base::base_implementation_type& impl,
259 reactor_op* op, bool is_continuation, bool peer_is_open,
260 void (*on_immediate)(operation* op, bool, const void*),
261 const void* immediate_arg)
262{
263 if (!peer_is_open)
264 {
265 do_start_op(impl, reactor::read_op, op, is_continuation,
266 true, false, on_immediate, immediate_arg);
267 }
268 else
269 {
270 op->ec_ = asio::error::already_open;
271 on_immediate(op, is_continuation, immediate_arg);
272 }
273}

275void reactive_socket_service_base::do_start_connect_op(
276 reactive_socket_service_base::base_implementation_type& impl,
277 reactor_op* op, bool is_continuation, const void* addr, size_t addrlen,
278 void (*on_immediate)(operation* op, bool, const void*),
279 const void* immediate_arg)
280{
281 if ((impl.state_ & socket_ops::non_blocking)
282 || socket_ops::set_internal_non_blocking(
283 impl.socket_, impl.state_, true, op->ec_))
284 {
285 if (socket_ops::connect(impl.socket_, addr, addrlen, op->ec_) != 0)
286 {
287 if (op->ec_ == asio::error::in_progress
288 || op->ec_ == asio::error::would_block)
289 {
290 op->ec_ = asio::error_code();
291 reactor_.start_op(reactor::connect_op, impl.socket_, impl.reactor_data_,
292 op, is_continuation, false, on_immediate, immediate_arg);
293 return;
294 }
295 }
296 }
297
298 on_immediate(op, is_continuation, immediate_arg);
299}
