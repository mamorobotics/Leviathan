371template <>
372struct hash<asio::ip::address_v6>
373{
374 std::size_t operator()(const asio::ip::address_v6& addr)
375 const noexcept
376 {
377 const asio::ip::address_v6::bytes_type bytes = addr.to_bytes();
378 std::size_t result = static_cast<std::size_t>(addr.scope_id());
379 combine_4_bytes(result, &bytes[0]);
380 combine_4_bytes(result, &bytes[4]);
381 combine_4_bytes(result, &bytes[8]);
382 combine_4_bytes(result, &bytes[12]);
383 return result;
384 }
385
386private:
387 static void combine_4_bytes(std::size_t& seed, const unsigned char* bytes)
388 {
389 const std::size_t bytes_hash =
390 (static_cast<std::size_t>(bytes[0]) << 24) |
391 (static_cast<std::size_t>(bytes[1]) << 16) |
392 (static_cast<std::size_t>(bytes[2]) << 8) |
393 (static_cast<std::size_t>(bytes[3]));
394 seed ^= bytes_hash + 0x9e3779b9 + (seed << 6) + (seed >> 2);
395 }
396};
