119 // Create results from an addrinfo list returned by getaddrinfo.
120 static basic_resolver_results create(
121 asio::detail::addrinfo_type* address_info,
122 const std::string& host_name, const std::string& service_name)
123 {
124 basic_resolver_results results;
125 if (!address_info)
126 return results;
127
128 std::string actual_host_name = host_name;
129 if (address_info->ai_canonname)
130 actual_host_name = address_info->ai_canonname;
131
132 results.values_.reset(new values_type);
133
134 while (address_info)
135 {
136 if (address_info->ai_family == ASIO_OS_DEF(AF_INET)
137 || address_info->ai_family == ASIO_OS_DEF(AF_INET6))
138 {
139 using namespace std; // For memcpy.
140 typename InternetProtocol::endpoint endpoint;
141 endpoint.resize(static_cast<std::size_t>(address_info->ai_addrlen));
142 memcpy(endpoint.data(), address_info->ai_addr,
143 address_info->ai_addrlen);
144 results.values_->push_back(
145 basic_resolver_entry<InternetProtocol>(endpoint,
146 actual_host_name, service_name));
147 }
148 address_info = address_info->ai_next;
149 }
150
151 return results;
152 }
153
154 // Create results from an endpoint, host name and service name.
155 static basic_resolver_results create(const endpoint_type& endpoint,
156 const std::string& host_name, const std::string& service_name)
157 {
158 basic_resolver_results results;

166 // Create results from a sequence of endpoints, host and service name.
167 template <typename EndpointIterator>
168 static basic_resolver_results create(
169 EndpointIterator begin, EndpointIterator end,
170 const std::string& host_name, const std::string& service_name)
171 {
172 basic_resolver_results results;
173 if (begin != end)
174 {
175 results.values_.reset(new values_type);
176 for (EndpointIterator ep_iter = begin; ep_iter != end; ++ep_iter)
177 {
178 results.values_->push_back(
179 basic_resolver_entry<InternetProtocol>(
180 *ep_iter, host_name, service_name));
181 }
182 }
183 return results;
184 }

186# if defined(ASIO_WINDOWS_RUNTIME)
187 // Create results from a Windows Runtime list of EndpointPair objects.
188 static basic_resolver_results create(
189 Windows::Foundation::Collections::IVectorView<
190 Windows::Networking::EndpointPair^>^ endpoints,
191 const asio::detail::addrinfo_type& hints,
192 const std::string& host_name, const std::string& service_name)
193 {
194 basic_resolver_results results;
195 if (endpoints->Size)
196 {
197 results.values_.reset(new values_type);
198 for (unsigned int i = 0; i < endpoints->Size; ++i)
199 {
200 auto pair = endpoints->GetAt(i);
201
202 if (hints.ai_family == ASIO_OS_DEF(AF_INET)
203 && pair->RemoteHostName->Type
204 != Windows::Networking::HostNameType::Ipv4)
205 continue;
206
207 if (hints.ai_family == ASIO_OS_DEF(AF_INET6)
208 && pair->RemoteHostName->Type
209 != Windows::Networking::HostNameType::Ipv6)
210 continue;
211
212 results.values_->push_back(
213 basic_resolver_entry<InternetProtocol>(
214 typename InternetProtocol::endpoint(
215 ip::make_address(
216 asio::detail::winrt_utils::string(
217 pair->RemoteHostName->CanonicalName)),
218 asio::detail::winrt_utils::integer(
219 pair->RemoteServiceName)),
220 host_name, service_name));
221 }
222 }
223 return results;
224 }
225# endif // defined(ASIO_WINDOWS_RUNTIME)
226#endif // !defined(GENERATING_DOCUMENTATION)
