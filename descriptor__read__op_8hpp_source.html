36template <typename MutableBufferSequence>
37class descriptor_read_op_base : public reactor_op
38{
39public:
40 descriptor_read_op_base(const asio::error_code& success_ec,
41 int descriptor, const MutableBufferSequence& buffers,
42 func_type complete_func)
43 : reactor_op(success_ec,
44 &descriptor_read_op_base::do_perform, complete_func),
45 descriptor_(descriptor),
46 buffers_(buffers)
47 {
48 }
49
50 static status do_perform(reactor_op* base)
51 {
52 ASIO_ASSUME(base != 0);
53 descriptor_read_op_base* o(static_cast<descriptor_read_op_base*>(base));
54
55 typedef buffer_sequence_adapter<asio::mutable_buffer,
56 MutableBufferSequence> bufs_type;
57
58 status result;
59 if (bufs_type::is_single_buffer)
60 {
61 result = descriptor_ops::non_blocking_read1(o->descriptor_,
62 bufs_type::first(o->buffers_).data(),
63 bufs_type::first(o->buffers_).size(),
64 o->ec_, o->bytes_transferred_) ? done : not_done;
65 }
66 else
67 {
68 bufs_type bufs(o->buffers_);
69 result = descriptor_ops::non_blocking_read(o->descriptor_,
70 bufs.buffers(), bufs.count(), o->ec_, o->bytes_transferred_)
71 ? done : not_done;
72 }
73
74 ASIO_HANDLER_REACTOR_OPERATION((*o, "non_blocking_read",
75 o->ec_, o->bytes_transferred_));
76
77 return result;
78 }
79
80private:
81 int descriptor_;
82 MutableBufferSequence buffers_;
83};

105 static void do_complete(void* owner, operation* base,
106 const asio::error_code& /*ec*/,
107 std::size_t /*bytes_transferred*/)
108 {
109 // Take ownership of the handler object.
110 ASIO_ASSUME(base != 0);
111 descriptor_read_op* o(static_cast<descriptor_read_op*>(base));
112 ptr p = { asio::detail::addressof(o->handler_), o, o };
113
114 ASIO_HANDLER_COMPLETION((*o));
115
116 // Take ownership of the operation's outstanding work.
117 handler_work<Handler, IoExecutor> w(
118 static_cast<handler_work<Handler, IoExecutor>&&>(
119 o->work_));
120
121 ASIO_ERROR_LOCATION(o->ec_);
122
123 // Make a copy of the handler so that the memory can be deallocated before
124 // the upcall is made. Even if we're not about to make an upcall, a
125 // sub-object of the handler may be the true owner of the memory associated
126 // with the handler. Consequently, a local copy of the handler is required
127 // to ensure that any owning sub-object remains valid until after we have
128 // deallocated the memory here.
129 detail::binder2<Handler, asio::error_code, std::size_t>
130 handler(o->handler_, o->ec_, o->bytes_transferred_);
131 p.h = asio::detail::addressof(handler.handler_);
132 p.reset();
133
134 // Make the upcall if required.
135 if (owner)
136 {
137 fenced_block b(fenced_block::half);
138 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
139 w.complete(handler, handler.handler_);
140 ASIO_HANDLER_INVOCATION_END;
141 }
142 }
