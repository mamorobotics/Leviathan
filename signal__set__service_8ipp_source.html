38struct signal_state
39{
40 // Mutex used for protecting global state.
41 static_mutex mutex_;
42
43 // The read end of the pipe used for signal notifications.
44 int read_descriptor_;
45
46 // The write end of the pipe used for signal notifications.
47 int write_descriptor_;
48
49 // Whether the signal state has been prepared for a fork.
50 bool fork_prepared_;
51
52 // The head of a linked list of all signal_set_service instances.
53 class signal_set_service* service_list_;
54
55 // A count of the number of objects that are registered for each signal.
56 std::size_t registration_count_[max_signal_number];
57
58 // The flags used for each registered signal.
59 signal_set_base::flags_t flags_[max_signal_number];
60};

70void asio_signal_handler(int signal_number)
71{
72#if defined(ASIO_WINDOWS) \
73 || defined(ASIO_WINDOWS_RUNTIME) \
74 || defined(__CYGWIN__)
75 signal_set_service::deliver_signal(signal_number);
76#else // defined(ASIO_WINDOWS)
77 // || defined(ASIO_WINDOWS_RUNTIME)
78 // || defined(__CYGWIN__)
79 int saved_errno = errno;
80 signal_state* state = get_signal_state();
81 signed_size_type result = ::write(state->write_descriptor_,
82 &signal_number, sizeof(signal_number));
83 (void)result;
84 errno = saved_errno;
85#endif // defined(ASIO_WINDOWS)
86 // || defined(ASIO_WINDOWS_RUNTIME)
87 // || defined(__CYGWIN__)
88
89#if defined(ASIO_HAS_SIGNAL) && !defined(ASIO_HAS_SIGACTION)
90 ::signal(signal_number, asio_signal_handler);
91#endif // defined(ASIO_HAS_SIGNAL) && !defined(ASIO_HAS_SIGACTION)
92}

207void signal_set_service::shutdown()
208{
209 remove_service(this);
210
211 op_queue<operation> ops;
212
213 for (int i = 0; i < max_signal_number; ++i)
214 {
215 registration* reg = registrations_[i];
216 while (reg)
217 {
218 ops.push(*reg->queue_);
219 reg = reg->next_in_table_;
220 }
221 }
222
223 scheduler_.abandon_operations(ops);
224}

226void signal_set_service::notify_fork(execution_context::fork_event fork_ev)
227{
228#if !defined(ASIO_WINDOWS) \
229 && !defined(ASIO_WINDOWS_RUNTIME) \
230 && !defined(__CYGWIN__)
231 signal_state* state = get_signal_state();
232 static_mutex::scoped_lock lock(state->mutex_);
233
234 switch (fork_ev)
235 {
236 case execution_context::fork_prepare:
237 {
238 int read_descriptor = state->read_descriptor_;
239 state->fork_prepared_ = true;
240 lock.unlock();
241# if defined(ASIO_HAS_IO_URING_AS_DEFAULT)
242 (void)read_descriptor;
243 io_uring_service_.deregister_io_object(io_object_data_);
244 io_uring_service_.cleanup_io_object(io_object_data_);
245# else // defined(ASIO_HAS_IO_URING_AS_DEFAULT)
246 reactor_.deregister_internal_descriptor(read_descriptor, reactor_data_);
247 reactor_.cleanup_descriptor_data(reactor_data_);
248# endif // defined(ASIO_HAS_IO_URING_AS_DEFAULT)
249 }
250 break;
251 case execution_context::fork_parent:
252 if (state->fork_prepared_)
253 {
254 int read_descriptor = state->read_descriptor_;
255 state->fork_prepared_ = false;
256 lock.unlock();
257# if defined(ASIO_HAS_IO_URING_AS_DEFAULT)
258 (void)read_descriptor;
259 io_uring_service_.register_internal_io_object(io_object_data_,
260 io_uring_service::read_op, new pipe_read_op);
261# else // defined(ASIO_HAS_IO_URING_AS_DEFAULT)
262 reactor_.register_internal_descriptor(reactor::read_op,
263 read_descriptor, reactor_data_, new pipe_read_op);
264# endif // defined(ASIO_HAS_IO_URING_AS_DEFAULT)
265 }
266 break;
267 case execution_context::fork_child:
268 if (state->fork_prepared_)
269 {
270 asio::detail::signal_blocker blocker;
271 close_descriptors();
272 open_descriptors();
273 int read_descriptor = state->read_descriptor_;
274 state->fork_prepared_ = false;
275 lock.unlock();
276# if defined(ASIO_HAS_IO_URING_AS_DEFAULT)
277 (void)read_descriptor;
278 io_uring_service_.register_internal_io_object(io_object_data_,
279 io_uring_service::read_op, new pipe_read_op);
280# else // defined(ASIO_HAS_IO_URING_AS_DEFAULT)
281 reactor_.register_internal_descriptor(reactor::read_op,
282 read_descriptor, reactor_data_, new pipe_read_op);
283# endif // defined(ASIO_HAS_IO_URING_AS_DEFAULT)
284 }
285 break;
286 default:
287 break;
288 }
289#else // !defined(ASIO_WINDOWS)
290 // && !defined(ASIO_WINDOWS_RUNTIME)
291 // && !defined(__CYGWIN__)
292 (void)fork_ev;
293#endif // !defined(ASIO_WINDOWS)
294 // && !defined(ASIO_WINDOWS_RUNTIME)
295 // && !defined(__CYGWIN__)
296}

312asio::error_code signal_set_service::add(
313 signal_set_service::implementation_type& impl, int signal_number,
314 signal_set_base::flags_t f, asio::error_code& ec)
315{
316 // Check that the signal number is valid.
317 if (signal_number < 0 || signal_number >= max_signal_number)
318 {
319 ec = asio::error::invalid_argument;
320 return ec;
321 }
322
323 // Check that the specified flags are supported.
324#if !defined(ASIO_HAS_SIGACTION)
325 if (f != signal_set_base::flags::dont_care)
326 {
327 ec = asio::error::operation_not_supported;
328 return ec;
329 }
330#endif // !defined(ASIO_HAS_SIGACTION)
331
332 signal_state* state = get_signal_state();
333 static_mutex::scoped_lock lock(state->mutex_);
334
335 // Find the appropriate place to insert the registration.
336 registration** insertion_point = &impl.signals_;
337 registration* next = impl.signals_;
338 while (next && next->signal_number_ < signal_number)
339 {
340 insertion_point = &next->next_in_set_;
341 next = next->next_in_set_;
342 }
343
344 // Only do something if the signal is not already registered.
345 if (next == 0 || next->signal_number_ != signal_number)
346 {
347 registration* new_registration = new registration;
348
349#if defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
350 // Register for the signal if we're the first.
351 if (state->registration_count_[signal_number] == 0)
352 {
353# if defined(ASIO_HAS_SIGACTION)
354 using namespace std; // For memset.
355 struct sigaction sa;
356 memset(&sa, 0, sizeof(sa));
357 sa.sa_handler = asio_signal_handler;
358 sigfillset(&sa.sa_mask);
359 if (f != signal_set_base::flags::dont_care)
360 sa.sa_flags = static_cast<int>(f);
361 if (::sigaction(signal_number, &sa, 0) == -1)
362# else // defined(ASIO_HAS_SIGACTION)
363 if (::signal(signal_number, asio_signal_handler) == SIG_ERR)
364# endif // defined(ASIO_HAS_SIGACTION)
365 {
366# if defined(ASIO_WINDOWS) || defined(__CYGWIN__)
367 ec = asio::error::invalid_argument;
368# else // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
369 ec = asio::error_code(errno,
370 asio::error::get_system_category());
371# endif // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
372 delete new_registration;
373 return ec;
374 }
375# if defined(ASIO_HAS_SIGACTION)
376 state->flags_[signal_number] = f;
377# endif // defined(ASIO_HAS_SIGACTION)
378 }
379# if defined(ASIO_HAS_SIGACTION)
380 // Otherwise check to see if the flags have changed.
381 else if (f != signal_set_base::flags::dont_care)
382 {
383 if (f != state->flags_[signal_number])
384 {
385 using namespace std; // For memset.
386 if (state->flags_[signal_number] != signal_set_base::flags::dont_care)
387 {
388 ec = asio::error::invalid_argument;
389 delete new_registration;
390 return ec;
391 }
392 struct sigaction sa;
393 memset(&sa, 0, sizeof(sa));
394 sa.sa_handler = asio_signal_handler;
395 sigfillset(&sa.sa_mask);
396 sa.sa_flags = static_cast<int>(f);
397 if (::sigaction(signal_number, &sa, 0) == -1)
398 {
399 ec = asio::error_code(errno,
400 asio::error::get_system_category());
401 delete new_registration;
402 return ec;
403 }
404 state->flags_[signal_number] = f;
405 }
406 }
407# endif // defined(ASIO_HAS_SIGACTION)
408#endif // defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
409
410 // Record the new registration in the set.
411 new_registration->signal_number_ = signal_number;
412 new_registration->queue_ = &impl.queue_;
413 new_registration->next_in_set_ = next;
414 *insertion_point = new_registration;
415
416 // Insert registration into the registration table.
417 new_registration->next_in_table_ = registrations_[signal_number];
418 if (registrations_[signal_number])
419 registrations_[signal_number]->prev_in_table_ = new_registration;
420 registrations_[signal_number] = new_registration;
421
422 ++state->registration_count_[signal_number];
423 }
424
425 ec = asio::error_code();
426 return ec;
427}

429asio::error_code signal_set_service::remove(
430 signal_set_service::implementation_type& impl,
431 int signal_number, asio::error_code& ec)
432{
433 // Check that the signal number is valid.
434 if (signal_number < 0 || signal_number >= max_signal_number)
435 {
436 ec = asio::error::invalid_argument;
437 return ec;
438 }
439
440 signal_state* state = get_signal_state();
441 static_mutex::scoped_lock lock(state->mutex_);
442
443 // Find the signal number in the list of registrations.
444 registration** deletion_point = &impl.signals_;
445 registration* reg = impl.signals_;
446 while (reg && reg->signal_number_ < signal_number)
447 {
448 deletion_point = &reg->next_in_set_;
449 reg = reg->next_in_set_;
450 }
451
452 if (reg != 0 && reg->signal_number_ == signal_number)
453 {
454#if defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
455 // Set signal handler back to the default if we're the last.
456 if (state->registration_count_[signal_number] == 1)
457 {
458# if defined(ASIO_HAS_SIGACTION)
459 using namespace std; // For memset.
460 struct sigaction sa;
461 memset(&sa, 0, sizeof(sa));
462 sa.sa_handler = SIG_DFL;
463 if (::sigaction(signal_number, &sa, 0) == -1)
464# else // defined(ASIO_HAS_SIGACTION)
465 if (::signal(signal_number, SIG_DFL) == SIG_ERR)
466# endif // defined(ASIO_HAS_SIGACTION)
467 {
468# if defined(ASIO_WINDOWS) || defined(__CYGWIN__)
469 ec = asio::error::invalid_argument;
470# else // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
471 ec = asio::error_code(errno,
472 asio::error::get_system_category());
473# endif // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
474 return ec;
475 }
476# if defined(ASIO_HAS_SIGACTION)
477 state->flags_[signal_number] = signal_set_base::flags_t();
478# endif // defined(ASIO_HAS_SIGACTION)
479 }
480#endif // defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
481
482 // Remove the registration from the set.
483 *deletion_point = reg->next_in_set_;
484
485 // Remove the registration from the registration table.
486 if (registrations_[signal_number] == reg)
487 registrations_[signal_number] = reg->next_in_table_;
488 if (reg->prev_in_table_)
489 reg->prev_in_table_->next_in_table_ = reg->next_in_table_;
490 if (reg->next_in_table_)
491 reg->next_in_table_->prev_in_table_ = reg->prev_in_table_;
492
493 --state->registration_count_[signal_number];
494
495 delete reg;
496 }
497
498 ec = asio::error_code();
499 return ec;
500}

502asio::error_code signal_set_service::clear(
503 signal_set_service::implementation_type& impl,
504 asio::error_code& ec)
505{
506 signal_state* state = get_signal_state();
507 static_mutex::scoped_lock lock(state->mutex_);
508
509 while (registration* reg = impl.signals_)
510 {
511#if defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
512 // Set signal handler back to the default if we're the last.
513 if (state->registration_count_[reg->signal_number_] == 1)
514 {
515# if defined(ASIO_HAS_SIGACTION)
516 using namespace std; // For memset.
517 struct sigaction sa;
518 memset(&sa, 0, sizeof(sa));
519 sa.sa_handler = SIG_DFL;
520 if (::sigaction(reg->signal_number_, &sa, 0) == -1)
521# else // defined(ASIO_HAS_SIGACTION)
522 if (::signal(reg->signal_number_, SIG_DFL) == SIG_ERR)
523# endif // defined(ASIO_HAS_SIGACTION)
524 {
525# if defined(ASIO_WINDOWS) || defined(__CYGWIN__)
526 ec = asio::error::invalid_argument;
527# else // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
528 ec = asio::error_code(errno,
529 asio::error::get_system_category());
530# endif // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
531 return ec;
532 }
533# if defined(ASIO_HAS_SIGACTION)
534 state->flags_[reg->signal_number_] = signal_set_base::flags_t();
535# endif // defined(ASIO_HAS_SIGACTION)
536 }
537#endif // defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
538
539 // Remove the registration from the registration table.
540 if (registrations_[reg->signal_number_] == reg)
541 registrations_[reg->signal_number_] = reg->next_in_table_;
542 if (reg->prev_in_table_)
543 reg->prev_in_table_->next_in_table_ = reg->next_in_table_;
544 if (reg->next_in_table_)
545 reg->next_in_table_->prev_in_table_ = reg->prev_in_table_;
546
547 --state->registration_count_[reg->signal_number_];
548
549 impl.signals_ = reg->next_in_set_;
550 delete reg;
551 }
552
553 ec = asio::error_code();
554 return ec;
555}

557asio::error_code signal_set_service::cancel(
558 signal_set_service::implementation_type& impl,
559 asio::error_code& ec)
560{
561 ASIO_HANDLER_OPERATION((scheduler_.context(),
562 "signal_set", &impl, 0, "cancel"));
563
564 op_queue<operation> ops;
565 {
566 signal_state* state = get_signal_state();
567 static_mutex::scoped_lock lock(state->mutex_);
568
569 while (signal_op* op = impl.queue_.front())
570 {
571 op->ec_ = asio::error::operation_aborted;
572 impl.queue_.pop();
573 ops.push(op);
574 }
575 }
576
577 scheduler_.post_deferred_completions(ops);
578
579 ec = asio::error_code();
580 return ec;

583void signal_set_service::cancel_ops_by_key(
584 signal_set_service::implementation_type& impl, void* cancellation_key)
585{
586 op_queue<operation> ops;

589 signal_state* state = get_signal_state();

591
592 while (signal_op* op = impl.queue_.front())
593 {
594 impl.queue_.pop();
595 if (op->cancellation_key_ == cancellation_key)
596 {
597 op->ec_ = asio::error::operation_aborted;
598 ops.push(op);

602 }
603
604 impl.queue_.push(other_ops);
605 }

607 scheduler_.post_deferred_completions(ops);

610void signal_set_service::deliver_signal(int signal_number)
611{
612 signal_state* state = get_signal_state();
613 static_mutex::scoped_lock lock(state->mutex_);
614
615 signal_set_service* service = state->service_list_;
616 while (service)
617 {
618 op_queue<operation> ops;
619
620 registration* reg = service->registrations_[signal_number];
621 while (reg)
622 {
623 if (reg->queue_->empty())
624 {
625 ++reg->undelivered_;
626 }
627 else
628 {
629 while (signal_op* op = reg->queue_->front())
630 {
631 op->signal_number_ = signal_number;
632 reg->queue_->pop();
633 ops.push(op);
634 }
635 }
636
637 reg = reg->next_in_table_;
638 }
639
640 service->scheduler_.post_deferred_completions(ops);
641
642 service = service->next_;
643 }

646void signal_set_service::add_service(signal_set_service* service)
647{
648 signal_state* state = get_signal_state();
649 static_mutex::scoped_lock lock(state->mutex_);
650
651#if !defined(ASIO_WINDOWS) && !defined(__CYGWIN__)
652 // If this is the first service to be created, open a new pipe.
653 if (state->service_list_ == 0)
654 open_descriptors();
655#endif // !defined(ASIO_WINDOWS) && !defined(__CYGWIN__)
656
657 // If a scheduler_ object is thread-unsafe then it must be the only
658 // scheduler used to create signal_set objects.
659 if (state->service_list_ != 0)
660 {
661 if (!ASIO_CONCURRENCY_HINT_IS_LOCKING(SCHEDULER,
662 service->scheduler_.concurrency_hint())
663 || !ASIO_CONCURRENCY_HINT_IS_LOCKING(SCHEDULER,
664 state->service_list_->scheduler_.concurrency_hint()))
665 {
666 std::logic_error ex(
667 "Thread-unsafe execution context objects require "
668 "exclusive access to signal handling.");
669 asio::detail::throw_exception(ex);
670 }
671 }
672
673 // Insert service into linked list of all services.
674 service->next_ = state->service_list_;
675 service->prev_ = 0;
676 if (state->service_list_)
677 state->service_list_->prev_ = service;
678 state->service_list_ = service;
679
680#if !defined(ASIO_WINDOWS) \
681 && !defined(ASIO_WINDOWS_RUNTIME) \
682 && !defined(__CYGWIN__)
683 // Register for pipe readiness notifications.
684 int read_descriptor = state->read_descriptor_;
685 lock.unlock();
686# if defined(ASIO_HAS_IO_URING_AS_DEFAULT)
687 (void)read_descriptor;
688 service->io_uring_service_.register_internal_io_object(
689 service->io_object_data_, io_uring_service::read_op, new pipe_read_op);
690# else // defined(ASIO_HAS_IO_URING_AS_DEFAULT)
691 service->reactor_.register_internal_descriptor(reactor::read_op,

699void signal_set_service::remove_service(signal_set_service* service)
700{
701 signal_state* state = get_signal_state();
702 static_mutex::scoped_lock lock(state->mutex_);
703
704 if (service->next_ || service->prev_ || state->service_list_ == service)
705 {
706#if !defined(ASIO_WINDOWS) \
707 && !defined(ASIO_WINDOWS_RUNTIME) \
708 && !defined(__CYGWIN__)
709 // Disable the pipe readiness notifications.
710 int read_descriptor = state->read_descriptor_;
711 lock.unlock();
712# if defined(ASIO_HAS_IO_URING_AS_DEFAULT)
713 (void)read_descriptor;
714 service->io_uring_service_.deregister_io_object(service->io_object_data_);
715 service->io_uring_service_.cleanup_io_object(service->io_object_data_);
716 lock.lock();
717# else // defined(ASIO_HAS_IO_URING_AS_DEFAULT)
718 service->reactor_.deregister_internal_descriptor(
719 read_descriptor, service->reactor_data_);
720 service->reactor_.cleanup_descriptor_data(service->reactor_data_);
721 lock.lock();
722# endif // defined(ASIO_HAS_IO_URING_AS_DEFAULT)
723#endif // !defined(ASIO_WINDOWS)
724 // && !defined(ASIO_WINDOWS_RUNTIME)
725 // && !defined(__CYGWIN__)
726
727 // Remove service from linked list of all services.
728 if (state->service_list_ == service)
729 state->service_list_ = service->next_;
730 if (service->prev_)
731 service->prev_->next_ = service->next_;
732 if (service->next_)
733 service->next_->prev_= service->prev_;
734 service->next_ = 0;
735 service->prev_ = 0;
736
737#if !defined(ASIO_WINDOWS) && !defined(__CYGWIN__)
738 // If this is the last service to be removed, close the pipe.
739 if (state->service_list_ == 0)
740 close_descriptors();
741#endif // !defined(ASIO_WINDOWS) && !defined(__CYGWIN__)
742 }

796void signal_set_service::start_wait_op(
797 signal_set_service::implementation_type& impl, signal_op* op)
798{
799 scheduler_.work_started();
800
801 signal_state* state = get_signal_state();
802 static_mutex::scoped_lock lock(state->mutex_);
803
804 registration* reg = impl.signals_;
805 while (reg)
806 {
807 if (reg->undelivered_ > 0)
808 {
809 --reg->undelivered_;
810 op->signal_number_ = reg->signal_number_;
811 scheduler_.post_deferred_completion(op);
812 return;
813 }
814
815 reg = reg->next_in_set_;
816 }
817
818 impl.queue_.push(op);
819}
