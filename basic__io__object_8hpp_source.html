27 // Type trait used to determine whether a service supports move.
28 template <typename IoObjectService>
29 class service_has_move
30 {
31 private:
32 typedef IoObjectService service_type;
33 typedef typename service_type::implementation_type implementation_type;
34
35 template <typename T, typename U>
36 static auto asio_service_has_move_eval(T* t, U* u)
37 -> decltype(t->move_construct(*u, *u), char());
38 static char (&asio_service_has_move_eval(...))[2];
39
40 public:
41 static const bool value =
42 sizeof(asio_service_has_move_eval(
43 static_cast<service_type*>(0),
44 static_cast<implementation_type*>(0))) == 1;
