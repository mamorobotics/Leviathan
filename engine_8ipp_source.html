32engine::engine(SSL_CTX* context)
33 : ssl_(::SSL_new(context))
34{
35 if (!ssl_)
36 {
37 asio::error_code ec(
38 static_cast<int>(::ERR_get_error()),
39 asio::error::get_ssl_category());
40 asio::detail::throw_error(ec, "engine");
41 }
42
43#if (OPENSSL_VERSION_NUMBER < 0x10000000L)
44 accept_mutex().init();
45#endif // (OPENSSL_VERSION_NUMBER < 0x10000000L)
46
47 ::SSL_set_mode(ssl_, SSL_MODE_ENABLE_PARTIAL_WRITE);
48 ::SSL_set_mode(ssl_, SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
49#if defined(SSL_MODE_RELEASE_BUFFERS)
50 ::SSL_set_mode(ssl_, SSL_MODE_RELEASE_BUFFERS);
51#endif // defined(SSL_MODE_RELEASE_BUFFERS)
52
53 ::BIO* int_bio = 0;
54 ::BIO_new_bio_pair(&int_bio, 0, &ext_bio_, 0);
55 ::SSL_set_bio(ssl_, int_bio, int_bio);
56}

149int engine::verify_callback_function(int preverified, X509_STORE_CTX* ctx)
150{
151 if (ctx)
152 {
153 if (SSL* ssl = static_cast<SSL*>(
154 ::X509_STORE_CTX_get_ex_data(
155 ctx, ::SSL_get_ex_data_X509_STORE_CTX_idx())))
156 {
157 if (SSL_get_app_data(ssl))
158 {
159 verify_callback_base* callback =
160 static_cast<verify_callback_base*>(
161 SSL_get_app_data(ssl));
162
163 verify_context verify_ctx(ctx);
164 return callback->call(preverified != 0, verify_ctx) ? 1 : 0;
165 }
166 }
167 }
168
169 return 0;
170}

231const asio::error_code& engine::map_error_code(
232 asio::error_code& ec) const
233{
234 // We only want to map the error::eof code.
235 if (ec != asio::error::eof)
236 return ec;
237
238 // If there's data yet to be read, it's an error.
239 if (BIO_wpending(ext_bio_))
240 {
241 ec = asio::ssl::error::stream_truncated;
242 return ec;
243 }
244
245 // SSL v2 doesn't provide a protocol-level shutdown, so an eof on the
246 // underlying transport is passed through.
247#if (OPENSSL_VERSION_NUMBER < 0x10100000L)
248 if (SSL_version(ssl_) == SSL2_VERSION)
249 return ec;
250#endif // (OPENSSL_VERSION_NUMBER < 0x10100000L)
251
252 // Otherwise, the peer should have negotiated a proper shutdown.
253 if ((::SSL_get_shutdown(ssl_) & SSL_RECEIVED_SHUTDOWN) == 0)
254 {
255 ec = asio::ssl::error::stream_truncated;
256 }
257
258 return ec;
259}

269engine::want engine::perform(int (engine::* op)(void*, std::size_t),
270 void* data, std::size_t length, asio::error_code& ec,
271 std::size_t* bytes_transferred)
272{
273 std::size_t pending_output_before = ::BIO_ctrl_pending(ext_bio_);
274 ::ERR_clear_error();
275 int result = (this->*op)(data, length);
276 int ssl_error = ::SSL_get_error(ssl_, result);
277 int sys_error = static_cast<int>(::ERR_get_error());
278 std::size_t pending_output_after = ::BIO_ctrl_pending(ext_bio_);
279
280 if (ssl_error == SSL_ERROR_SSL)
281 {
282 ec = asio::error_code(sys_error,
283 asio::error::get_ssl_category());
284 return pending_output_after > pending_output_before
285 ? want_output : want_nothing;
286 }
287
288 if (ssl_error == SSL_ERROR_SYSCALL)
289 {
290 if (sys_error == 0)
291 {
292 ec = asio::ssl::error::unspecified_system_error;
293 }
294 else
295 {
296 ec = asio::error_code(sys_error,
297 asio::error::get_ssl_category());
298 }
299 return pending_output_after > pending_output_before
300 ? want_output : want_nothing;
301 }
302
303 if (result > 0 && bytes_transferred)
304 *bytes_transferred = static_cast<std::size_t>(result);
305
306 if (ssl_error == SSL_ERROR_WANT_WRITE)
307 {
308 ec = asio::error_code();
309 return want_output_and_retry;
310 }
311 else if (pending_output_after > pending_output_before)
312 {
313 ec = asio::error_code();
314 return result > 0 ? want_output : want_output_and_retry;
315 }
316 else if (ssl_error == SSL_ERROR_WANT_READ)
317 {
318 ec = asio::error_code();
319 return want_input_and_retry;
320 }
321 else if (ssl_error == SSL_ERROR_ZERO_RETURN)
322 {
323 ec = asio::error::eof;
324 return want_nothing;
325 }
326 else if (ssl_error == SSL_ERROR_NONE)
327 {
328 ec = asio::error_code();
329 return want_nothing;
330 }
331 else
332 {
333 ec = asio::ssl::error::unexpected_result;
334 return want_nothing;
335 }
336}
