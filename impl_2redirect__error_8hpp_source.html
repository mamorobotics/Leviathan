30// Class to adapt a redirect_error_t as a completion handler.
31template <typename Handler>
32class redirect_error_handler
33{
34public:
35 typedef void result_type;
36
37 template <typename CompletionToken>
38 redirect_error_handler(redirect_error_t<CompletionToken> e)
39 : ec_(e.ec_),
40 handler_(static_cast<CompletionToken&&>(e.token_))
41 {
42 }
43
44 template <typename RedirectedHandler>
45 redirect_error_handler(asio::error_code& ec,
46 RedirectedHandler&& h)
47 : ec_(ec),
48 handler_(static_cast<RedirectedHandler&&>(h))
49 {
50 }
51
52 void operator()()
53 {
54 static_cast<Handler&&>(handler_)();
55 }
56
57 template <typename Arg, typename... Args>
58 enable_if_t<
59 !is_same<decay_t<Arg>, asio::error_code>::value
60 >
61 operator()(Arg&& arg, Args&&... args)
62 {
63 static_cast<Handler&&>(handler_)(
64 static_cast<Arg&&>(arg),
65 static_cast<Args&&>(args)...);
66 }
67
68 template <typename... Args>
69 void operator()(const asio::error_code& ec, Args&&... args)
70 {
71 ec_ = ec;
72 static_cast<Handler&&>(handler_)(static_cast<Args&&>(args)...);
73 }
74
75//private:
76 asio::error_code& ec_;
77 Handler handler_;
78};
